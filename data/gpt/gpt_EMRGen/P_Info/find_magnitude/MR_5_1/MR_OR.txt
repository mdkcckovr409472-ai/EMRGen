int z5check(double result1, double result2) {
    // Core evaluation logic (direct scalar comparison)

    // Step 1: Define base tolerance
    const double epsilon = 1e-5;

    // Step 2: Algebraic rearrangement for 'result1 >= result2' is already in the form A >= B
    double A = result1;
    double B = result2;

    // Step 3: NaN and Inf checks
    if (std::isnan(A) || std::isnan(B)) return 0;
    if (std::isinf(A) || std::isinf(B)) {
        if (A >= B) return 1;
        else return 0;
    }

    // Step 4: Calculate dynamic tolerance
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

    // Step 5: Tolerance-based evaluation for >=
    if (A < B - tol) return 0;

    return 1;
}
