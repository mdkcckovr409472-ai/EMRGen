#include <math.h>
#include <stddef.h>

int z5check(double result1, double result2) {
    // 1. 防御性编程 (针对浮点数的特殊检查)
    // 检查 NaN
    if (isnan(result1) || isnan(result2)) return 0;

    // 2. 核心判断逻辑 (标量直接比较)
    // 根据条件 return.result1 <= return.result2，移项后为 result1 <= result2
    // 提取移项后的操作数
    double A = result1;
    double B = result2;

    // 处理无穷大情况（动态容差对Inf无效）
    if (isinf(A) || isinf(B)) {
        // 直接进行数学比较
        if (A <= B) return 1;
        else return 0;
    }

    // 3. 浮点数比较的特殊处理
    double epsilon = 1e-5;
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

    // 根据移项后的条件 result1 <= result2 进行带容差的比较
    // 如果违反条件 (A > B + tol)，则返回0
    if (A > B + tol) return 0;

    // 4. 满足条件，返回 1
    return 1;
}
