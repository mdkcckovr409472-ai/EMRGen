int check(TestCase tc) {
    if (tc.n1 != tc.n2) {
        // Sizes mismatch: function returns NULL, treat MR as invalid
        return 0;
    }
    double* Os = ebeMultiply(tc.a, tc.b, tc.n1, tc.n2);
    if (Os == NULL) {
        return 0;
    }
    TestCase Tf_case = change(tc);
    if (Tf_case.b == NULL) {
        free(Os);
        return 0;
    }
    double* Of = ebeMultiply(Tf_case.a, Tf_case.b, Tf_case.n1, Tf_case.n2);
    if (Of == NULL) {
        free(Os);
        free(Tf_case.b);
        return 0;
    }
    int is_valid = 1;
    for (int i = 0; i < tc.n1; i++) {
        // Check relation: Of[i] == Os[i] - a[i]
        // Using a small epsilon for floating point comparison
        double diff = Of[i] - (Os[i] - tc.a[i]);
        if (diff > 1e-9 || diff < -1e-9) {
            is_valid = 0;
            break;
        }
    }
    free(Os);
    free(Of);
    free(Tf_case.b);
    return is_valid;
}
