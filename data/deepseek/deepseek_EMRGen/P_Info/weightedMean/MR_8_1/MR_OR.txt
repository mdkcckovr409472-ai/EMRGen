#include <cmath>
#include <cstddef>

int z8check(double result1, double result2) {
    // 1. 核心判断逻辑 (标量直接比较)
    // 条件: 2 * return.result1 - return.result2 == 0
    // 移项恢复量级: 2 * result1 == result2

    double A = 2.0 * result1;
    double B = result2;

    // 防御 NaN
    if (std::isnan(A) || std::isnan(B)) return 0;

    // 处理无穷大 (动态容差对Inf无效)
    if (std::isinf(A) || std::isinf(B)) {
        // 直接进行数学比较
        if (A == B) return 1;
        else return 0;
    }

    // 定义基础容差并计算动态容差
    double epsilon = 1e-5;
    double tol = epsilon * std::fmax(1.0, std::fmax(std::fabs(A), std::fabs(B)));

    // 带容差的相等比较 (移项后为 A == B)
    if (std::fabs(A - B) > tol) {
        return 0;
    }

    // 2. 满足条件，返回 1
    return 1;
}
