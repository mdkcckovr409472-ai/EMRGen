TestCase z30Change(TestCase t1) {
    TestCase follow_case = t1;
    // Memory handling for array1
    int* follow_case_array1 = (int*)malloc(t1.n * sizeof(int));
    memcpy(follow_case_array1, t1.array1, t1.n * sizeof(int));
    follow_case.array1 = follow_case_array1;

    // Defensive check for array length
    if (follow_case.n <= 0) {
        return follow_case;
    }

    // Divide elements at odd positions by 2 (when not zero)
    for (int i = 1; i < follow_case.n; i += 2) {
        if (follow_case.array1[i] != 0) {
            follow_case.array1[i] = follow_case.array1[i] / 2;
        }
    }

    return follow_case;
}
