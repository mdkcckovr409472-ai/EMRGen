TestCase change(TestCase tc) {
    TestCase tf = tc;
    double* sorted_elements = new double[tf.n];
    for (int i = 0; i < tf.n; ++i) {
        sorted_elements[i] = tf.elements[i];
    }
    // Sort the array in ascending order
    for (int i = 0; i < tf.n - 1; ++i) {
        for (int j = i + 1; j < tf.n; ++j) {
            if (sorted_elements[i] > sorted_elements[j]) {
                double tmp = sorted_elements[i];
                sorted_elements[i] = sorted_elements[j];
                sorted_elements[j] = tmp;
            }
        }
    }
    tf.elements = sorted_elements;
    return tf;
}
