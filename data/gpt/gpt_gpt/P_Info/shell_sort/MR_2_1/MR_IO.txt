TestCase change(TestCase tc) {
    TestCase tf = tc;
    // Sort the array in ascending order to construct the follow-up input Tf
    // Allocate a new array to avoid modifying the original input array
    int* sorted_array = (int*)malloc(tf.n * sizeof(int));
    if (!sorted_array) return tf; // allocation failed, return original
    memcpy(sorted_array, tf.a, tf.n * sizeof(int));
    // Simple std::sort usage since allowed or we implement a sort
    // As no std lib claimed, implement a simple sort (e.g. qsort or bubble)
    // Here, quick bubble sort for clarity:
    for (int i = 0; i < tf.n - 1; i++) {
        for (int j = 0; j < tf.n - i - 1; j++) {
            if (sorted_array[j] > sorted_array[j + 1]) {
                int tmp = sorted_array[j];
                sorted_array[j] = sorted_array[j + 1];
                sorted_array[j + 1] = tmp;
            }
        }
    }
    tf.a = sorted_array;
    return tf;
}
