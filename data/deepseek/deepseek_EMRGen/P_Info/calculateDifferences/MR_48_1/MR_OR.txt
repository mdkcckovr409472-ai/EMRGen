#include <math.h>
#include <stddef.h>

int z48check(double* result1, double* result2, int n) {
    // 1. 防御性编程 (空指针与长度检查)
    if (result1 == NULL || result2 == NULL) return 0;
    if (n <= 0) return 1;

    // 2. 核心判断逻辑 (字典序比较)
    // 条件: result1[] >= result2[] (lexically)
    // 字典序逻辑：从第一个元素开始比较，遇到第一个不相等的元素对即可决定结果。
    double epsilon = 1e-5;

    for (int i = 0; i < n; i++) {
        double A = result1[i];
        double B = result2[i];

        // 防御 NaN
        if (isnan(A) || isnan(B)) return 0;

        // 处理无穷大情况
        if (isinf(A) || isinf(B)) {
            // 对于字典序，在第一个不相等的位置，如果 A < B，则整个数组字典序小于，违反 >= 条件。
            if (A < B) return 0;
            // 如果 A > B，则字典序大于，满足条件，可以立即返回成功。
            if (A > B) return 1;
            // 如果 A == B (两个同号无穷大)，继续检查下一个元素。
            continue;
        }

        // 计算动态容差
        double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

        // 检查第一个不相等的位置
        if (fabs(A - B) > tol) {
            // 在第一个不相等的位置，判断大小关系
            // 条件要求 result1[] >= result2[] (lexically)
            // 如果 A < B - tol，则 result1[i] < result2[i]，字典序小于，违反条件。
            if (A < B - tol) {
                return 0;
            } else {
                // 如果 A > B + tol，则 result1[i] > result2[i]，字典序大于，满足条件，可以立即返回成功。
                return 1;
            }
        }
        // 如果 A 和 B 在容差内相等，继续检查下一个元素。
    }

    // 3. 循环结束，所有元素在容差内相等，满足 result1[] >= result2[] (lexically)
    return 1;
}
