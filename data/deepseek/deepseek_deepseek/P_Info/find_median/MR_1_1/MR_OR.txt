int check(TestCase tc) {
    int* original_a_copy = new int[tc.n];
    for (int i = 0; i < tc.n; i++) {
        original_a_copy[i] = tc.a[i];
    }
    double Os = find_median(original_a_copy, tc.n);
    delete[] original_a_copy;

    TestCase Tf_case = change(tc);
    double Of = find_median(Tf_case.a, Tf_case.n);

    int k = 5;
    int is_valid = 0;
    if (Os + k == Of) {
        is_valid = 1;
    }

    delete[] Tf_case.a;
    return is_valid;
}
