#include <math.h>
#include <stddef.h>

int z25check(double result1, double result2) {
    // 1. 防御性编程 (标量无需空指针和长度检查)

    // 2. 核心判断逻辑 (标量直接比较)
    // 条件为 return.result1 == return.result2，移项后为 result1 == result2
    // 浮点数比较需要特殊处理

    // 第一步：定义基础容差
    double epsilon = 1e-5;

    // 第二步：提取左右操作数 (已移项)
    double A = result1;
    double B = result2;

    // 第三步：防御NaN，处理无穷大
    if (isnan(A) || isnan(B)) return 0;
    
    if (isinf(A) || isinf(B)) {
        // 无穷大时退化为直接数学比较
        if (A == B) return 1;
        else return 0;
    }

    // 第四步：计算动态宽容度
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

    // 第五步：带容差的相等判定
    if (fabs(A - B) > tol) {
        return 0;
    }

    // 3. 满足条件，返回 1
    return 1;
}
