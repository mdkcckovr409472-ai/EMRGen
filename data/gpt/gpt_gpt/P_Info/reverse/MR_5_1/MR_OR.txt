int check(TestCase tc) {
    int *Os = reverse(tc.a, tc.n);
    TestCase Tf_case = change(tc);
    int *Of = reverse(Tf_case.a, Tf_case.n);
    int is_valid = 1;
    for (int i = 0; i < tc.n; i++) {
        if (Of[i] != Os[tc.n - 1 - i]) {
            is_valid = 0;
            break;
        }
    }
    return is_valid;
}
