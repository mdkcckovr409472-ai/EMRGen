int check(TestCase tc) {
    int* Os = dec_array(tc.a, tc.k, tc.n);
    if (!Os) return 0;
    TestCase Tf_case = change(tc);
    int* Of = dec_array(Tf_case.a, Tf_case.k, Tf_case.n);
    if (!Of) {
        free(Os);
        free(Tf_case.a);
        return 0;
    }

    int sum_Os = 0, sum_Of = 0;
    for (int i = 0; i < tc.n; i++) {
        sum_Os += Os[i];
        sum_Of += Of[i];
    }
    int is_valid = (sum_Os - tc.n) == sum_Of ? 1 : 0;

    free(Os);
    free(Of);
    free(Tf_case.a);

    return is_valid;
}
