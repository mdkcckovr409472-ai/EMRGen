int z33check(double result1, double result2) {
    double epsilon = 1e-5;

    // Extract A and B for the condition: result1 == result2
    // Rearranged as: A == B -> A = result1, B = result2

    double A = result1;
    double B = result2;

    if (std::isnan(A) || std::isnan(B)) return 0;

    if (std::isinf(A) || std::isinf(B)) {
        if (A == B) return 1;
        else return 0;
    }

    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

    if (fabs(A - B) > tol) return 0;

    return 1;
}
