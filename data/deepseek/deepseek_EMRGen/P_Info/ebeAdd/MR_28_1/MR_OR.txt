#include <cmath>
#include <cfloat>

int z28check(double* result1, double* result2, int n) {
    // 1. 防御性编程 (空指针与长度检查)
    if (result1 == NULL || result2 == NULL) return 0;
    if (n <= 0) return 1;

    // 2. 核心判断逻辑 (elementwise)
    for (int i = 0; i < n; i++) {
        // 浮点数比较特殊处理
        double epsilon = 1e-5;
        
        // 提取左右操作数（移项后）：result1[i] > result2[i]
        double A = result1[i];
        double B = result2[i];
        
        // 防御 NaN
        if (std::isnan(A) || std::isnan(B)) return 0;
        
        // 处理无穷大情况
        if (std::isinf(A) || std::isinf(B)) {
            if (!(A > B)) return 0;
            continue; // 当前元素检查通过，继续下一个
        }
        
        // 计算动态宽容度
        double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
        
        // 带容差的比较判定：A > B
        // 违反条件：A <= B + tol
        if (A <= B + tol) {
            return 0;
        }
    }
    
    // 3. 全部满足，返回 1
    return 1;
}
