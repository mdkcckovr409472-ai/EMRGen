int z8check(double result1, double result2) {
    // 1. Core evaluation logic (direct scalar comparison)
    double epsilon = 1e-5;

    double A = result1;
    double B = result2;

    // Defend against NaN
    if (std::isnan(A) || std::isnan(B)) return 0;

    // Defend against Inf
    if (std::isinf(A) || std::isinf(B)) {
        if (A > B) return 1; else return 0;
    }

    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

    // Check return.result1 > return.result2
    if (A <= B + tol) return 0;

    return 1;
}
