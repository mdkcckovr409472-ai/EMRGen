#include <cmath>
#include <cfloat>

int z3check(double result1, double result2) {
    // 1. 核心判断逻辑 (标量直接比较)
    // 移项处理: 2 * result1 - result2 == 0  => 2 * result1 == result2
    double A = 2.0 * result1;
    double B = result2;
    
    // 2. 浮点数特殊处理
    // 防御NaN
    if (std::isnan(A) || std::isnan(B)) return 0;
    
    // 处理无穷大情况
    if (std::isinf(A) || std::isinf(B)) {
        if (A == B) return 1;  // 移项后的相等比较
        else return 0;
    }
    
    // 动态容差计算
    double epsilon = 1e-5;
    double tol = epsilon * std::fmax(1.0, std::fmax(std::fabs(A), std::fabs(B)));
    
    // 带容差的相等比较
    if (std::fabs(A - B) > tol) {
        return 0;
    }
    
    // 3. 满足条件，返回 1
    return 1;
}
