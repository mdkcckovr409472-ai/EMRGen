TestCase z4Change(TestCase t1) {
    TestCase follow_case = t1;
    // Memory handling for expected array
    double* follow_case_expected = (double*)malloc(t1.n * sizeof(double));
    memcpy(follow_case_expected, t1.expected, t1.n * sizeof(double));
    follow_case.expected = follow_case_expected;
    // Defensive check for empty array
    if (follow_case.n <= 0) {
        return follow_case;
    }
    // Multiply all elements of expected by 2.0
    for (int i = 0; i < follow_case.n; ++i) {
        follow_case.expected[i] *= 2.0;
    }
    return follow_case;
}
