int check(TestCase tc) {
    double* Os = standardize(tc.data, tc.n);
    if (Os == NULL) return 0;
    TestCase Tf_case = change(tc);
    double* Of = standardize(Tf_case.data, Tf_case.n);
    if (Of == NULL) {
        free(Os);
        return 0;
    }
    int is_valid = 0;
    // Check if Os and Of are not identical element-wise
    for (int i = 0; i < tc.n; i++) {
        if (Os[i] != Of[i]) {
            is_valid = 1;
            break;
        }
    }
    free(Os);
    free(Of);
    return is_valid;
}
