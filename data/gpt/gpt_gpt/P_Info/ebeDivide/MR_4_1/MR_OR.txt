int check(TestCase tc) {
    if (tc.n1 != tc.n2) {
        // invalid input sizes, no further checking possible
        return 0;
    }

    double* Os = ebeDivide(tc.a, tc.b, tc.n1, tc.n2);
    if (Os == NULL) {
        // source output allocation failure, test cannot continue
        return 0;
    }

    TestCase Tf_case = change(tc);

    double* Of = ebeDivide(Tf_case.a, Tf_case.b, Tf_case.n1, Tf_case.n2);

    int is_valid = 1;

    if (Of == NULL) {
        is_valid = 0;
    } else {
        for (int i = 0; i < tc.n1; i++) {
            if (tc.b[i] == 0.0) {
                if (Of[i] != INFINITY) {
                    is_valid = 0;
                    break;
                }
            } else {
                if (Os[i] == INFINITY) {
                    // In source output, if Os[i] is INFINITY, Of[i] must be Os[i]/2, but INFINITY/2 is still INFINITY
                    // So Of[i] must be INFINITY as well
                    if (Of[i] != INFINITY) {
                        is_valid = 0;
                        break;
                    }
                } else {
                    if (Of[i] != Os[i] / 2.0) {
                        is_valid = 0;
                        break;
                    }
                }
            }
        }
    }

    if (Os != NULL) free(Os);
    if (Of != NULL) free(Of);
    if (Tf_case.a != NULL && Tf_case.a != tc.a) free(Tf_case.a);

    return is_valid;
}
