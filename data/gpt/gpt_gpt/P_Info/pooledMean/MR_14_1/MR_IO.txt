TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tf.n2 > 1) {
        double* shifted = new double[tf.n2];
        for (int i = 0; i < tf.n2 - 1; i++) {
            shifted[i] = tf.data2[i + 1];
        }
        shifted[tf.n2 - 1] = tf.data2[0];
        tf.data2 = shifted;
    }
    return tf;
}
