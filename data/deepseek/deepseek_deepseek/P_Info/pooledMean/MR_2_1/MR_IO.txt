TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tf.n1 > 0 && tf.data1 != nullptr) {
        double* reversed_data1 = new double[tf.n1];
        for (int i = 0; i < tf.n1; ++i) {
            reversed_data1[i] = tf.data1[tf.n1 - 1 - i];
        }
        tf.data1 = reversed_data1;
    }
    return tf;
}
