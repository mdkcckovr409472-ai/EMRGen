TestCase change(TestCase tc) {
    TestCase tf = tc;
    // Allocate new array for data2 with all elements subtracted by 1
    double* newData2 = new double[tf.n2];
    for (int i = 0; i < tf.n2; i++) {
        newData2[i] = tf.data2[i] - 1.0;
    }
    tf.data2 = newData2; // overwrite with new data2 array
    return tf;
}
