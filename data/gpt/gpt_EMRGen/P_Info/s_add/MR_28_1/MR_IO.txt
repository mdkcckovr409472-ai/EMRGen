TestCase z28Change(TestCase t1) {
    TestCase follow_case = t1;
    // Memory handling for array1 (deep copy)
    int* follow_case_array1 = (int*)malloc(t1.n * sizeof(int));
    memcpy(follow_case_array1, t1.array1, t1.n * sizeof(int));
    follow_case.array1 = follow_case_array1;
    
    // Defensive check: need at least 2 elements to have an odd index
    if (follow_case.n < 2) {
        return follow_case;
    }
    
    // Set every odd index element to -7
    for (int i = 1; i < follow_case.n; i += 2) {
        follow_case.array1[i] = -7;
    }
    
    return follow_case;
}
