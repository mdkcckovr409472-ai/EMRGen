#include <cmath>
#include <cfloat>

int z42check(double result1, double result2) {
    // 1. 防御性编程 (标量情况无需指针和长度检查)
    // 2. 核心判断逻辑 (标量直接比较)
    
    // 提取移项后的左右操作数
    double A = result1;
    double B = result2;
    
    // 第三步：防御NaN，处理Inf
    if (std::isnan(A) || std::isnan(B)) return 0;
    
    // 处理无穷大情况
    if (std::isinf(A) || std::isinf(B)) {
        // 移项后的条件是相等 (==)
        if (A == B) return 1;
        else return 0;
    }
    
    // 第四步：计算动态宽容度
    double epsilon = 1e-5;
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    // 第五步：带容差的相等判定
    // 移项后的条件是相等 (==)
    if (fabs(A - B) > tol) {
        return 0;
    }
    
    // 3. 满足条件，返回 1
    return 1;
}
