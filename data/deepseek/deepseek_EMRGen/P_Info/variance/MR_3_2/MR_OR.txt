#include <cmath>
#include <cfloat>

int z3check(double result1, double result2) {
    // 防御性编程：检查NaN
    if (std::isnan(result1) || std::isnan(result2)) return 0;
    
    // 处理无穷大情况（动态容差对Inf无效）
    if (std::isinf(result1) || std::isinf(result2)) {
        // 直接数学比较
        if (result1 <= result2) return 1;
        else return 0;
    }
    
    // 定义基础容差
    double epsilon = 1e-5;
    
    // 移项后的操作数：A = result1, B = result2
    double A = result1;
    double B = result2;
    
    // 计算动态容差
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    // 带容差的 <= 比较
    // 条件：A <= B
    // 违反条件：A > B + tol
    if (A > B + tol) {
        return 0;
    }
    
    // 满足条件
    return 1;
}
