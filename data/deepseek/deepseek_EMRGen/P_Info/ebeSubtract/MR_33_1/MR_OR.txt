#include <cmath>
#include <cfloat>

int z33check(double* result1, double* result2, int n) {
    // 1. 防御性编程 (空指针与长度检查)
    if (result1 == NULL || result2 == NULL) return 0;
    if (n <= 0) return 1;

    // 2. 核心判断逻辑 (elementwise)
    for (int i = 0; i < n; i++) {
        // 条件: 2 * result1[i] - 2 * result2[i] + 3 == 0
        // 移项恢复量级: 2 * result1[i] + 3 == 2 * result2[i]
        // 提取 A 和 B
        double A = 2.0 * result1[i] + 3.0;
        double B = 2.0 * result2[i];

        // 防御 NaN
        if (std::isnan(A) || std::isnan(B)) return 0;

        // 处理无穷大
        if (std::isinf(A) || std::isinf(B)) {
            if (A != B) return 0; // 对于等式，无穷大必须相等
            // 如果相等，则继续（即允许两个正无穷或两个负无穷相等）
            continue;
        }

        // 定义基础容差
        double epsilon = 1e-5;
        // 计算动态容差
        double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

        // 带容差的相等比较 (移项后为 A == B)
        if (fabs(A - B) > tol) {
            return 0; // 一旦违反条件，立即返回 0
        }
    }

    // 3. 全部满足，返回 1
    return 1;
}
