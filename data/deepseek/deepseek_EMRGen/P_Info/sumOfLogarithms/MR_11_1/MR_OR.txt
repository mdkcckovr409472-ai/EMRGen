#include <math.h>
#include <stddef.h>

int z11check(double result1, double result2) {
    // 1. 防御性编程 (标量情况无需指针和长度检查)
    // 2. 核心判断逻辑 (标量直接比较)
    // 根据规则3，对浮点数进行安全比较

    // 提取移项后的操作数 A 和 B
    // 条件: result1 < result2
    // 移项后: result1 < result2 (无需移项)
    double A = result1;
    double B = result2;

    // 防御NaN
    if (isnan(A) || isnan(B)) return 0;

    // 处理无穷大
    if (isinf(A) || isinf(B)) {
        // 无穷大时直接进行数学比较
        if (A < B) return 1;
        else return 0;
    }

    // 定义基础容差
    double epsilon = 1e-5;
    // 计算动态容差
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

    // 根据移项后的条件进行带容差的比较
    // 条件: A < B
    // 违反条件: A >= B - tol
    if (A >= B - tol) {
        return 0;
    }

    // 3. 满足条件，返回 1
    return 1;
}
