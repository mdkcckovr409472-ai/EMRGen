TestCase change(TestCase tc) {
    TestCase tf = tc;
    // Allocate new array for the sorted input to avoid modifying original input
    int* new_a = new int[tc.n];
    for (int i = 0; i < tc.n; i++) {
        new_a[i] = tc.a[i];
    }
    // Sort in descending order
    std::sort(new_a, new_a + tc.n, std::greater<int>());
    tf.a = new_a;
    return tf;
}
