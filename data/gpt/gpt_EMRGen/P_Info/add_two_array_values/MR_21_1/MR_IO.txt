TestCase z21Change(TestCase t1) {
    TestCase follow_case = t1;
    // --- Memory handling code ---
    int* follow_case_a = (int*)malloc(t1.n * sizeof(int));
    memcpy(follow_case_a, t1.a, t1.n * sizeof(int));
    follow_case.a = follow_case_a;
    // -------------------------

    // Defensive checks
    if (follow_case.n == 0) {
        return follow_case;
    }
    if (follow_case.i < 0 || follow_case.i >= follow_case.n) {
        return follow_case;
    }

    // Operation: increment element at index i by 10
    follow_case.a[follow_case.i] += 10;

    return follow_case;
}
