TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tc.n > 1 && tc.values != nullptr) {
        int* reversed_values = new int[tc.n];
        for (int i = 0; i < tc.n; ++i) {
            reversed_values[i] = tc.values[tc.n - 1 - i];
        }
        tf.values = reversed_values;
    }
    return tf;
}
