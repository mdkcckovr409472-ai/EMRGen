int check(TestCase tc) {
    double Os = add_two_array_values(tc.a, tc.i, tc.j, tc.n);
    TestCase Tf_case = change(tc);
    double Of = add_two_array_values(Tf_case.a, Tf_case.i, Tf_case.j, Tf_case.n);
    int is_valid = 0;
    if (Os != -100000 && Of != -100000) {
        if (2 * Os == Of) {
            is_valid = 1;
        }
    }
    if (Tf_case.a != tc.a) {
        delete[] Tf_case.a;
    }
    return is_valid;
}
