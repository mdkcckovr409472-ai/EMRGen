TestCase z23Change(TestCase t1) {
    TestCase follow_case = t1;
    // Memory handling for target array x
    double* follow_case_x = (double*)malloc(t1.n * sizeof(double));
    memcpy(follow_case_x, t1.x, t1.n * sizeof(double));
    follow_case.x = follow_case_x;

    // Defensive check: ensure there is at least one element
    if (follow_case.n < 1) {
        return follow_case;
    }

    // Clamp values greater than 50.0 to 50.0
    for (int i = 0; i < follow_case.n; ++i) {
        if (follow_case.x[i] > 50.0) {
            follow_case.x[i] = 50.0;
        }
    }

    return follow_case;
}
