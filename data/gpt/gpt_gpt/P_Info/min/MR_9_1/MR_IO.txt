TestCase change(TestCase tc) {
    TestCase tf = tc;
    double max_val = tf.elements[0];
    for (int i = 1; i < tf.n; i++) {
        if (tf.elements[i] > max_val) {
            max_val = tf.elements[i];
        }
    }
    double* new_elements = new double[tf.n];
    for (int i = 0; i < tf.n; i++) {
        new_elements[i] = tf.elements[i];
    }
    new_elements[0] = max_val;
    tf.elements = new_elements;
    return tf;
}
