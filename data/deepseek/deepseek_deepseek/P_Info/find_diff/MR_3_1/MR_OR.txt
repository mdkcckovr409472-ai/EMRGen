int check(TestCase tc) {
    int* Os = find_diff(tc.a, tc.b, tc.na);
    TestCase Tf_case = change(tc);
    int* Of = find_diff(Tf_case.a, Tf_case.b, Tf_case.na);
    int k = 5;
    int is_valid = 1;
    for (int i = 0; i < tc.na; i++) {
        if (Os[i] != Of[i] + k) {
            is_valid = 0;
            break;
        }
    }
    delete[] Tf_case.b;
    return is_valid;
}
