int check(TestCase tc) {
    int Os = cnt_zeroes(tc.a, tc.n);
    TestCase Tf_case = change(tc);
    int Of = cnt_zeroes(Tf_case.a, Tf_case.n);
    int is_valid = 0;
    if (Of >= Os) {
        is_valid = 1;
    }
    // Clean up dynamically allocated memory
    if (Tf_case.a != tc.a) {
        delete[] Tf_case.a;
    }
    return is_valid;
}
