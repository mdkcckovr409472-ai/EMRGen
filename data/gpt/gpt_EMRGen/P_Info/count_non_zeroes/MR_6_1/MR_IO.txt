TestCase z6Change(TestCase t1) {
    TestCase follow_case = t1;
    // --- Memory handling code ---
    int* follow_case_a = (int*)malloc(t1.n * sizeof(int));
    memcpy(follow_case_a, t1.a, t1.n * sizeof(int));
    follow_case.a = follow_case_a;
    // --------------------------------

    if (follow_case.n == 0) {
        return follow_case;
    }

    // Set last element to INT_MAX if it is less
    if (follow_case.a[follow_case.n - 1] < INT_MAX) {
        follow_case.a[follow_case.n - 1] = INT_MAX;
    }

    return follow_case;
}
