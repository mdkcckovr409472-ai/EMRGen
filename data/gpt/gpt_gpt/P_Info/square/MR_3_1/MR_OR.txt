int check(TestCase tc) {
    double* Os = square(tc.data, tc.n);
    if (Os == NULL) return 0;
    TestCase Tf_case = change(tc);
    double* Of = square(Tf_case.data, Tf_case.n);
    if (Of == NULL) {
        free(Os);
        free(Tf_case.data);
        return 0;
    }
    int is_valid = 1;
    for (int i = 0; i < tc.n; i++) {
        if (fabs(Of[i] - 0.25 * Os[i]) > 1e-9) {
            is_valid = 0;
            break;
        }
    }
    free(Os);
    free(Of);
    free(Tf_case.data);
    return is_valid;
}
