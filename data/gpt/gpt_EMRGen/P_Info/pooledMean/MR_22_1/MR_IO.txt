TestCase z22Change(TestCase t1) {
    TestCase follow_case = t1;
    // Deep copy of data1
    double* follow_case_data1 = (double*)malloc(t1.n1 * sizeof(double));
    memcpy(follow_case_data1, t1.data1, t1.n1 * sizeof(double));
    follow_case.data1 = follow_case_data1;

    // Defensive check for empty array
    if (follow_case.n1 <= 0) {
        return follow_case;
    }

    // Set a random element to 999.0 if it is less than 999.0
    int idx = rand() % follow_case.n1;
    if (follow_case.data1[idx] < 999.0) {
        follow_case.data1[idx] = 999.0;
    }

    return follow_case;
}
