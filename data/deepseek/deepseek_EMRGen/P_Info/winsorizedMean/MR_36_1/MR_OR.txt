#include <math.h>
#include <stddef.h>

int z36check(double result1, double result2) {
    // 1. 防御性编程 (标量情况无需指针和长度检查)
    // 2. 核心判断逻辑 (标量直接比较)
    // 对于浮点数比较，必须进行代数移项和容差处理
    // 条件: result1 >= result2
    // 移项后: A = result1, B = result2, 比较符为 >=

    double A = result1;
    double B = result2;

    // 防御 NaN
    if (isnan(A) || isnan(B)) return 0;

    // 处理无穷大情况
    if (isinf(A) || isinf(B)) {
        // 对于无穷大，退化为精确数学比较
        if (A >= B) return 1;
        else return 0;
    }

    // 定义基础容差
    double epsilon = 1e-5;
    // 计算动态容差
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

    // 带容差的 >= 比较: 若 A < B - tol 则违反条件
    if (A < B - tol) {
        return 0;
    }

    // 3. 满足条件，返回 1
    return 1;
}
