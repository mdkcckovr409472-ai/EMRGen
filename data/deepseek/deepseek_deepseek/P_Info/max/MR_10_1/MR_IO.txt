TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tf.n > 0 && tf.elements != nullptr) {
        double initial_max = max(tf.elements, tf.n);
        for (int i = 0; i < tf.n; ++i) {
            if (tf.elements[i] != initial_max) {
                tf.elements[i] -= 1.0;
            }
        }
    }
    return tf;
}
