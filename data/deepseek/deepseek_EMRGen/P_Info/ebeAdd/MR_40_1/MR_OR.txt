#include <cmath>
#include <cfloat>

int z40check(double* result1, double* result2, int n) {
    // 1. 防御性编程 (空指针与长度检查)
    if (result1 == NULL || result2 == NULL) return 0;
    if (n <= 0) return 1;

    // 2. 核心判断逻辑 (字典序比较)
    for (int i = 0; i < n; i++) {
        double A = result1[i];
        double B = result2[i];

        // 浮点数特殊处理
        // 防御 NaN
        if (std::isnan(A) || std::isnan(B)) return 0;

        // 处理无穷大情况
        if (std::isinf(A) || std::isinf(B)) {
            // 无穷大时直接进行数学比较
            if (A < B) {
                // 第一个不相等元素对中，A < B，不满足 >=，字典序上 result1 < result2
                return 0;
            } else if (A > B) {
                // 第一个不相等元素对中，A > B，满足 >=，字典序上 result1 > result2
                // 可以提前确定满足条件，但需要继续检查后续元素是否完全相等吗？
                // 根据字典序规则：发现第一个不相等元素对即可决定结果，这里 A > B 已经满足 >=
                // 但注意：我们需要的是 result1[] >= result2[] (lexically)
                // 如果 A > B，那么字典序上 result1 已经大于 result2，条件满足，可以立即返回 1
                // 但标准字典序逻辑是：遇到第一个不相等的元素对，根据它决定整个数组的序关系
                // 所以这里应该立即返回 1
                return 1;
            }
            // A == B (都是无穷大且同号)，继续检查下一个元素
            continue;
        }

        // 常规浮点数比较（带容差）
        double epsilon = 1e-5;
        double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

        // 检查是否相等（在容差范围内）
        if (fabs(A - B) <= tol) {
            // 相等，继续检查下一个元素
            continue;
        } else {
            // 不相等，根据这一对元素决定字典序
            if (A < B - tol) {
                // 违反 >= 条件
                return 0;
            } else {
                // A > B + tol，满足 >= 条件
                return 1;
            }
        }
    }

    // 3. 所有元素都相等（在容差范围内），满足 >= 条件
    return 1;
}
