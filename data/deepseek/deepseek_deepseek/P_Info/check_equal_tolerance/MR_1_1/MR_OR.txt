int check(TestCase tc) {
    bool Os = check_equal_tolerance(tc.a, tc.b, tc.tol, tc.n1, tc.n2);
    TestCase Tf_case = change(tc);
    bool Of = check_equal_tolerance(Tf_case.a, Tf_case.b, Tf_case.tol, Tf_case.n1, Tf_case.n2);
    int is_valid = 0;
    if (Os <= Of) {
        is_valid = 1;
    }
    return is_valid;
}
