    TestCase follow_case = t1;
    // Memory handling: deep copy of array1
    int* follow_case_array1 = (int*)malloc(t1.n * sizeof(int));
    memcpy(follow_case_array1, t1.array1, t1.n * sizeof(int));
    follow_case.array1 = follow_case_array1;

    // Defensive check: ensure there is at least one element
    if (follow_case.n <= 0) {
        return follow_case;
    }

    // Increment every third element by 2
    for (int i = 2; i < follow_case.n; i += 3) {
        follow_case.array1[i] += 2;
    }

    return follow_case;
}
