TestCase change(TestCase tc) {
    // Reuse input array by creating a new shifted array
    int* shifted = new int[tc.n];
    if (tc.n > 0) {
        shifted[0] = tc.a[tc.n - 1];
        for (int i = 1; i < tc.n; i++) {
            shifted[i] = tc.a[i - 1];
        }
    }
    TestCase tf = tc;
    tf.a = shifted;
    return tf;
}
