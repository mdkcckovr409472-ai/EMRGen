TestCase change(TestCase tc) {
    TestCase tf = tc;
    tf.a = new double[tc.n];
    for (int i = 0; i < tc.n; ++i) {
        tf.a[i] = tc.a[i];
    }
    for (int i = 0; i < tc.n; ++i) {
        int j = rand() % tc.n;
        double temp = tf.a[i];
        tf.a[i] = tf.a[j];
        tf.a[j] = temp;
    }
    return tf;
}
