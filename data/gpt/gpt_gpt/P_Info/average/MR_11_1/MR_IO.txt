TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tf.n > 0) {
        int mid = tf.n / 2;
        int* new_a = new int[tf.n];
        for (int i = 0; i < tf.n; i++) {
            new_a[i] = tf.a[i];
        }
        new_a[mid] = new_a[mid] - 1;
        tf.a = new_a;
    }
    return tf;
}
