#include <cmath>
#include <cfloat>

int z4check(double* result1, double* result2, int n) {
    // 1. 防御性编程 (空指针与长度检查)
    if (result1 == NULL || result2 == NULL) return 0;
    if (n <= 0) return 1;

    // 2. 核心判断逻辑 (elementwise)
    for (int i = 0; i < n; i++) {
        // 浮点数比较特殊处理
        double epsilon = 1e-5;
        
        // 移项处理：条件 result1[i] >= result2[i] 已经是标准形式
        double A = result1[i];
        double B = result2[i];
        
        // 防御 NaN
        if (std::isnan(A) || std::isnan(B)) return 0;
        
        // 处理无穷大
        if (std::isinf(A) || std::isinf(B)) {
            if (!(A >= B)) return 0;
            continue;
        }
        
        // 计算动态容差
        double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
        
        // 带容差的比较：A >= B 等价于 !(A < B)
        if (A < B - tol) {
            return 0;
        }
    }
    
    // 3. 全部满足，返回 1
    return 1;
}
