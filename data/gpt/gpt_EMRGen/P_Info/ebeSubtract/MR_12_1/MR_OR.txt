int z12check(double* result1, double* result2, int n) {
    if (n <= 0) return 1;
    if (result1 == NULL || result2 == NULL) return 0;

    const double epsilon = 1e-5;
    #include <cmath>

    for (int i = 0; i < n; i++) {
        double A = result1[i];
        double B = result2[i];
        if (std::isnan(A) || std::isnan(B)) return 0;
        if (std::isinf(A) || std::isinf(B)) {
            if (A <= B) continue;
            else return 0;
        }
        double tol = epsilon * std::fmax(1.0, std::fmax(std::fabs(A), std::fabs(B)));

        if (std::fabs(A - B) <= tol) {
            continue;
        } else if (A > B + tol) {
            return 0;
        } else {
            return 1;
        }
    }
    return 1;
}
