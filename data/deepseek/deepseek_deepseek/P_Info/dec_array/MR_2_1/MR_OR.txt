int check(TestCase tc) {
    int* Os = dec_array(tc.a, tc.k, tc.n);
    TestCase Tf_case = change(tc);
    int* Of = dec_array(Tf_case.a, Tf_case.k, Tf_case.n);
    int is_valid = 0;
    bool equal = true;
    for (int i = 0; i < tc.n; ++i) {
        if (Os[i] != Of[i]) {
            equal = false;
            break;
        }
    }
    if (equal) {
        is_valid = 1;
    }
    free(Os);
    free(Of);
    return is_valid;
}
