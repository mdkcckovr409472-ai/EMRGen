#include <cmath>
#include <cstddef>

int z49check(double result1, double result2) {
    // 1. 核心判断逻辑 (标量直接比较)
    // 条件为 result1 < result2
    // 浮点数比较需要特殊处理

    // 第一步：定义基础容差
    const double epsilon = 1e-5;

    // 第二步：提取左右操作数 A 和 B (代数移项)
    // 条件 result1 < result2 已经是标准形式
    double A = result1;
    double B = result2;

    // 第三步：防御 NaN，并对 Inf 进行特殊放行
    if (std::isnan(A) || std::isnan(B)) return 0;
    
    if (std::isinf(A) || std::isinf(B)) {
        // 出现无穷大，退化为数学上的直接比较
        if (A < B) return 1;
        else return 0;
    }

    // 第四步：计算动态宽容度
    double tol = epsilon * std::max(1.0, std::max(std::fabs(A), std::fabs(B)));

    // 第五步：带容差的比较判定
    // 移项后的条件是 A < B
    // 违反条件：A >= B - tol
    if (A >= B - tol) {
        return 0;
    }

    // 2. 满足条件，返回 1
    return 1;
}
