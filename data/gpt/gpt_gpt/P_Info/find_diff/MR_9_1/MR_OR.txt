int check(TestCase tc) {
    int na = tc.na;

    int* Os = find_diff(tc.a, tc.b, na);
    if (!Os) return 0;

    TestCase Tf_case = change(tc);
    int* Of = find_diff(Tf_case.a, Tf_case.b, na);
    if (!Of) {
        delete[] Tf_case.b;
        return 0;
    }

    int valid = 1;
    for (int i = 0; i < na - 1; i++) {
        if (Of[i] != Os[i]) {
            valid = 0;
            break;
        }
    }
    if (valid && (Of[na - 1] != (tc.a[na - 1] - INT_MIN))) {
        valid = 0;
    }

    delete[] Tf_case.b;

    return valid;
}
