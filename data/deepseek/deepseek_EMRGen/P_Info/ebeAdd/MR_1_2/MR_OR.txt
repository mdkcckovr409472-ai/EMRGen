#include <cmath>
#include <cfloat>

int z1check(double* result1, double* result2, int n) {
    // 1. 防御性编程 (空指针与长度检查)
    if (result1 == NULL || result2 == NULL) return 0;
    if (n <= 0) return 1;

    // 2. 核心判断逻辑 (elementwise)
    for (int i = 0; i < n; i++) {
        // 浮点数比较特殊处理
        // 移项后的条件: result1[i] < result2[i]
        double A = result1[i];
        double B = result2[i];
        
        // 防御NaN
        if (std::isnan(A) || std::isnan(B)) return 0;
        
        // 处理无穷大
        if (std::isinf(A) || std::isinf(B)) {
            // 无穷大时退化为数学直接比较
            if (!(A < B)) return 0;
            continue;
        }
        
        // 计算动态容差
        double epsilon = 1e-5;
        double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
        
        // 带容差的严格小于判断: A < B
        // 违反条件: A >= B - tol
        if (A >= B - tol) return 0;
    }

    // 3. 全部满足，返回 1
    return 1;
}
