TestCase z19Change(TestCase t1) {
    TestCase follow_case = t1;
    // --- Memory handling for data1 (deep copy) ---
    double* follow_case_data1 = (double*)malloc(t1.n1 * sizeof(double));
    memcpy(follow_case_data1, t1.data1, t1.n1 * sizeof(double));
    follow_case.data1 = follow_case_data1;
    // ----------------------------------------------

    // Defensive check: ensure there is at least one element to sort
    if (follow_case.n1 == 0) {
        return follow_case;
    }

    // Sort data1 in ascending order (simple bubble sort)
    for (int i = 0; i < follow_case.n1 - 1; ++i) {
        for (int j = 0; j < follow_case.n1 - i - 1; ++j) {
            if (follow_case.data1[j] > follow_case.data1[j + 1]) {
                double temp = follow_case.data1[j];
                follow_case.data1[j] = follow_case.data1[j + 1];
                follow_case.data1[j + 1] = temp;
            }
        }
    }

    return follow_case;
}
