TestCase z22Change(TestCase t1) {
    TestCase follow_case = t1;
    // Memory handling: deep copy of elements array
    double* follow_case_elements = (double*)malloc(t1.n * sizeof(double));
    memcpy(follow_case_elements, t1.elements, t1.n * sizeof(double));
    follow_case.elements = follow_case_elements;
    // Defensive check: ensure there is at least one element
    if (follow_case.n == 0) {
        return follow_case;
    }
    // Decrement a random element by 5.0 if it satisfies the condition
    int idx = rand() % follow_case.n;
    if (follow_case.elements[idx] >= -DBL_MAX + 5.0) {
        follow_case.elements[idx] -= 5.0;
    }
    return follow_case;
}
