int check(TestCase tc) {
    if (tc.n <= 0 || tc.a == nullptr) {
        return 0;
    }
    int Os = find_max(tc.a, tc.n);
    TestCase Tf_case = change(tc);
    int Of = find_max(Tf_case.a, Tf_case.n);
    int d = 10;
    int is_valid = 0;
    if (Os + d == Of) {
        is_valid = 1;
    }
    if (Tf_case.a != tc.a) {
        delete[] Tf_case.a;
    }
    return is_valid;
}
