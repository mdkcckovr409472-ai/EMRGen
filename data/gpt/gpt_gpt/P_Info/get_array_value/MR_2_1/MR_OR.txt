int check(TestCase tc) {
    if (tc.n <= 0 || tc.a == nullptr || tc.k <= 0 || tc.k > tc.n) {
        return 0;
    }
    int Os = get_array_value(tc.a, tc.k, tc.n);
    TestCase Tf_case = change(tc);
    if (Tf_case.a == nullptr) {
        return 0;
    }
    int Of = get_array_value(Tf_case.a, Tf_case.k, Tf_case.n);
    int is_valid = 0;
    if (Os - 1 == Of) {
        is_valid = 1;
    }
    delete[] Tf_case.a;
    return is_valid;
}
