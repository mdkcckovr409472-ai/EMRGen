TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tf.n > 0) {
        double* sorted_elements = new double[tf.n];
        for (int i = 0; i < tf.n; i++) {
            sorted_elements[i] = tc.elements[i];
        }
        std::sort(sorted_elements, sorted_elements + tf.n);
        tf.elements = sorted_elements;
    }
    return tf;
}
