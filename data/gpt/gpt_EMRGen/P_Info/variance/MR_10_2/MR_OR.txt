int z10check(double result1, double result2) {
    // Core evaluation logic (direct scalar comparison)

    // Define tolerance
    double epsilon = 1e-5;

    // Rearrange: return.result1 - return.result2 == 0
    // This is equivalent to: result1 == result2

    double A = result1;
    double B = result2;

    // Check NaN
    if (std::isnan(A) || std::isnan(B)) return 0;

    // Check Inf
    if (std::isinf(A) || std::isinf(B)) {
        if (A == B) return 1;
        else return 0;
    }

    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

    if (fabs(A - B) > tol) return 0;

    return 1;
}
