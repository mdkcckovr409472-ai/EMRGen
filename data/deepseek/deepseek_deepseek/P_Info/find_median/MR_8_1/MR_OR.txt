int check(TestCase tc) {
    int* copy_a = new int[tc.n];
    for (int i = 0; i < tc.n; ++i) {
        copy_a[i] = tc.a[i];
    }
    double Os = find_median(copy_a, tc.n);
    TestCase Tf_case = change(tc);
    double Of = find_median(Tf_case.a, Tf_case.n);
    int is_valid = 0;
    if (Os == Of) {
        is_valid = 1;
    }
    delete[] copy_a;
    return is_valid;
}
