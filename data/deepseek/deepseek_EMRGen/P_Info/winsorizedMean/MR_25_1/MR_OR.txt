#include <math.h>
#include <stddef.h>

int z25check(double result1, double result2) {
    // 浮点数比较：移项后条件为 result1 >= result2
    // 提取 A = result1, B = result2
    double A = result1;
    double B = result2;

    // 防御 NaN
    if (isnan(A) || isnan(B)) return 0;

    // 处理无穷大
    if (isinf(A) || isinf(B)) {
        // 无穷大时直接数学比较
        if (A >= B) return 1;
        else return 0;
    }

    // 动态容差
    double epsilon = 1e-5;
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

    // 带容差的 >= 判断
    if (A < B - tol) return 0;

    // 满足条件
    return 1;
}
