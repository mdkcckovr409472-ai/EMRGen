int check(TestCase tc) {
    double Os = meanDeviation(tc.elements, tc.mean, tc.n);

    TestCase Tf_case = change(tc);
    double Of = meanDeviation(Tf_case.elements, Tf_case.mean, Tf_case.n);

    int is_valid = 0;
    if (Os == Of) {
        is_valid = 1;
    }

    delete[] Tf_case.elements;

    return is_valid;
}
