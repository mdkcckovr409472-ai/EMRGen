#include <math.h>
#include <stddef.h>

int z32check(double result1, double result2) {
    // 1. 核心判断逻辑 (标量直接比较)
    // 条件: return.result1 - 3 * return.result2 == 0
    // 移项恢复量级: result1 == 3 * result2
    
    double A = result1;
    double B = 3.0 * result2;
    
    // 浮点数特殊处理
    double epsilon = 1e-5;
    
    // 防御NaN
    if (isnan(A) || isnan(B)) return 0;
    
    // 处理无穷大情况
    if (isinf(A) || isinf(B)) {
        if (A == B) return 1;
        else return 0;
    }
    
    // 计算动态容差
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    // 带容差的相等比较
    if (fabs(A - B) > tol) {
        return 0;
    }
    
    // 2. 满足条件，返回 1
    return 1;
}
