int check(TestCase tc) {
    // (1) Extract the source input
    double* x_src = tc.x;
    int n = tc.n;

    // (2) Call the function under test to get Os
    double Os = variance(x_src, n);

    // (3) Call change() to get follow-up test case Tf
    TestCase Tf_case = change(tc);

    // (4) Extract follow-up input and get Of
    double Of = variance(Tf_case.x, Tf_case.n);

    // (5) Verify MR: Os == Of (exact equality for doubles here)
    int is_valid = 0;
    if (Os == Of) {
        is_valid = 1;
    }

    // (6) Free dynamic memory allocated in change()
    delete[] Tf_case.x;

    // (7) Return verification result
    return is_valid;
}
