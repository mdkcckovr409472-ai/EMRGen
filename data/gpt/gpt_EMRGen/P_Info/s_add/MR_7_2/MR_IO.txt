TestCase z7Change(TestCase t1) {
    TestCase follow_case = t1;
    // Deep copy of array1
    int* follow_case_array1 = (int*)malloc(t1.n * sizeof(int));
    memcpy(follow_case_array1, t1.array1, t1.n * sizeof(int));
    follow_case.array1 = follow_case_array1;

    // Defensive check: ensure there is at least one element
    if (follow_case.n == 0) {
        return follow_case;
    }

    // Increment element at a random index by 1
    int idx = rand() % follow_case.n;
    follow_case.array1[idx] += 1;

    return follow_case;
}
