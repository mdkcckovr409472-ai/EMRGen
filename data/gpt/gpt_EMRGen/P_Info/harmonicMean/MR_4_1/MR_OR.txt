#include <cmath>

int z4check(double result1, double result2) {
    // Core evaluation logic (direct scalar comparison)
    double epsilon = 1e-5;

    // Rearrange invariant: return.result1 - 2 * return.result2 == 0
    // Rearranged as: result1 == 2 * result2
    // So A = result1, B = 2 * result2

    double A = result1;
    double B = 2 * result2;

    if (std::isnan(A) || std::isnan(B)) return 0;
    if (std::isinf(A) || std::isinf(B)) {
        if (A == B) return 1;
        else return 0;
    }

    double tol = epsilon * std::fmax(1.0, std::fmax(std::fabs(A), std::fabs(B)));
    if (std::fabs(A - B) > tol) return 0;

    return 1;
}
