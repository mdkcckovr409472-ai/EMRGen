TestCase change(TestCase tc) {
    TestCase tf = tc;

    // Allocate new array for sample1 to avoid modifying the original array
    tf.sample1 = new double[tc.n];
    for (int i = 0; i < tc.n; i++) {
        tf.sample1[i] = tc.sample1[i];
    }

    // Add 2 to the first half of the elements of sample1
    int half = tc.n / 2;
    for (int i = 0; i < half; i++) {
        tf.sample1[i] += 2;
    }

    return tf;
}
