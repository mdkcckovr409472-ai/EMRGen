#include <cmath>
#include <cfloat>

int z2check(double* result1, double* result2, int n) {
    // 1. 防御性编程 (空指针与长度检查)
    if (result1 == NULL || result2 == NULL) return 0;
    if (n <= 0) return 1;

    // 2. 核心判断逻辑 (elementwise)
    for (int i = 0; i < n; i++) {
        // 根据条件 2 * result1 - 2 * result2 - 5 == 0 进行移项
        // 移项后得到：2 * result1 == 2 * result2 + 5
        double A = 2.0 * result1[i];
        double B = 2.0 * result2[i] + 5.0;

        // 防御 NaN
        if (std::isnan(A) || std::isnan(B)) return 0;

        // 处理无穷大情况
        if (std::isinf(A) || std::isinf(B)) {
            if (A != B) return 0; // 对于 == 比较，无穷大必须严格相等
            // 如果 A 和 B 都是同号无穷大，则继续循环
            continue;
        }

        // 定义基础容差
        double epsilon = 1e-5;
        // 计算动态宽容度
        double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

        // 带容差的相等判定 (移项后是 ==)
        if (fabs(A - B) > tol) {
            return 0; // 一旦违反条件，立即返回 0
        }
    }

    // 3. 全部满足，返回 1
    return 1;
}
