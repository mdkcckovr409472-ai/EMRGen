int check(TestCase tc) {
    double Os = variance(tc.x, tc.n);
    TestCase Tf_case = change(tc);
    double Of = variance(Tf_case.x, Tf_case.n);
    int is_valid = 0;
    if (Os == Of) {
        is_valid = 1;
    }
    if (Tf_case.x != tc.x) {
        delete[] Tf_case.x;
    }
    return is_valid;
}
