int check(TestCase tc) {
    int Os = hamming_dist(tc.a, tc.b, tc.n);
    TestCase Tf_case = change(tc);
    int Of = hamming_dist(Tf_case.a, Tf_case.b, Tf_case.n);
    int is_valid = 0;
    if (Os - 1 <= Of && Of <= Os + 1) {
        is_valid = 1;
    }
    if (Tf_case.b != tc.b) {
        delete[] Tf_case.b;
    }
    return is_valid;
}
