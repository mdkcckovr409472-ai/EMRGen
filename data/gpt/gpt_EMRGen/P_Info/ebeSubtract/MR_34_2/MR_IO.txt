TestCase z34Change(TestCase t1) {
    TestCase follow_case = t1;

    double* follow_case_b = NULL;
    if (t1.n2 > 0) {
        follow_case_b = (double*)malloc(t1.n2 * sizeof(double));
        if (follow_case_b != NULL) {
            memcpy(follow_case_b, t1.b, t1.n2 * sizeof(double));
        }
    }
    follow_case.b = follow_case_b;

    if (follow_case.n2 <= 0 || follow_case.b == NULL) {
        return follow_case;
    }

    int valid_positions_count = 0;
    for (int i = 0; i < follow_case.n2; i++) {
        if (follow_case.b[i] > -1.0) {
            valid_positions_count++;
        }
    }

    if (valid_positions_count == 0) {
        return follow_case;
    }

    int rand_index = -1;
    int attempted = 0;
    int max_attempts = follow_case.n2 * 2; // to avoid potential infinite loop
    while (attempted < max_attempts) {
        int pos = rand() % follow_case.n2;
        if (follow_case.b[pos] > -1.0) {
            rand_index = pos;
            break;
        }
        attempted++;
    }

    if (rand_index != -1) {
        follow_case.b[rand_index] = -1.0;
    }

    return follow_case;
}
