int check(TestCase tc) {
    double* Os = square(tc.data, tc.n);
    TestCase Tf_case = change(tc);
    double* Of = square(Tf_case.data, Tf_case.n);
    int is_valid = 1;
    for (int i = 0; i < tc.n; i++) {
        if (Os[i] != Of[i]) {
            is_valid = 0;
            break;
        }
    }
    free(Os);
    free(Of);
    free(Tf_case.data);
    return is_valid;
}
