TestCase z16Change(TestCase t1) {
    TestCase follow_case = t1;
    // Memory handling
    int* follow_case_list = (int*)malloc(t1.n * sizeof(int));
    memcpy(follow_case_list, t1.list, t1.n * sizeof(int));
    follow_case.list = follow_case_list;
    // Defensive check
    if (follow_case.n > 0) {
        int first = follow_case.list[0];
        for (int i = 0; i < follow_case.n - 1; ++i) {
            follow_case.list[i] = follow_case.list[i + 1];
        }
        follow_case.list[follow_case.n - 1] = first;
    }
    return follow_case;
}
