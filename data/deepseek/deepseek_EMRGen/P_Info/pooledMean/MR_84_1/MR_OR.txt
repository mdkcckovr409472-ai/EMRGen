#include <math.h>
#include <stddef.h>

int z84check(double result1, double result2) {
    // 1. 核心判断逻辑 (标量直接比较)
    // 移项后的条件: result1 >= result2
    double A = result1;
    double B = result2;
    
    // 防御 NaN
    if (isnan(A) || isnan(B)) return 0;
    
    // 处理无穷大
    if (isinf(A) || isinf(B)) {
        if (A >= B) return 1;
        else return 0;
    }
    
    // 动态容差计算
    double epsilon = 1e-5;
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    // 带容差的 >= 比较
    if (A < B - tol) {
        return 0;
    }
    
    // 2. 满足条件，返回 1
    return 1;
}
