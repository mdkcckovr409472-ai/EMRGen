int z8check(int* result1, int* result2, int n) {
    if (n <= 0) return 1;
    if (result1 == NULL || result2 == NULL) return 0;

    #include <math.h>
    #include <stdbool.h>

    double epsilon = 1e-5;
    for (int i = 0; i < n; i++) {
        double A = (double)result1[i];
        double B = (double)result2[i];

        if (isnan(A) || isnan(B)) return 0;
        if (isinf(A) || isinf(B)) {
            if (A > B) return 1;
            else return 0;
        }

        double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
        double diff = A - B;

        if (fabs(diff) <= tol) continue;
        else if (diff > tol) return 1;
        else return 0;
    }
    return 0;
}
