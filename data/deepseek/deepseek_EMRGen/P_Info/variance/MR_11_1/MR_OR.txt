#include <math.h>
#include <stddef.h>

int z11check(double result1, double result2) {
    // 1. 防御性编程 (针对浮点数的特殊处理)
    // 定义基础容差
    double epsilon = 1e-5;
    
    // 根据条件 "return.result1 >= return.result2" 提取 A 和 B
    // 移项后条件为: result1 >= result2
    double A = result1;
    double B = result2;
    
    // 检查 NaN
    if (isnan(A) || isnan(B)) return 0;
    
    // 处理无穷大情况
    if (isinf(A) || isinf(B)) {
        // 无穷大时退化为数学直接比较
        if (A >= B) return 1;
        else return 0;
    }
    
    // 计算动态容差
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    // 2. 核心判断逻辑 (带容差的浮点数比较)
    // 条件: A >= B  (即 result1 >= result2)
    // 违反条件: A < B - tol
    if (A < B - tol) {
        return 0;
    }
    
    // 3. 满足条件，返回 1
    return 1;
}
