int z23check(double* result1, double* result2, int n) {
    // 1. Defensive programming (Length and null pointer checks)
    if (n <= 0) return 1;
    if (result1 == NULL || result2 == NULL) return 0;

    const double epsilon = 1e-5;

    // 2. Core evaluation logic (elementwise: result1[i] >= result2[i])
    for (int i = 0; i < n; i++) {
        double A = result1[i];
        double B = result2[i];

        // NaN checks
        if (std::isnan(A) || std::isnan(B)) return 0;

        // Inf checks
        if (std::isinf(A) || std::isinf(B)) {
            if (A >= B)
                continue;
            else
                return 0;
        }

        double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

        // Check condition result1[i] >= result2[i] with tolerance
        if (A < B - tol) return 0;
    }

    // 3. Fully satisfied
    return 1;
}
