#include <math.h>
#include <stddef.h>

int z28check(double result1, double result2) {
    // 防御性编程：检查NaN
    if (isnan(result1) || isnan(result2)) return 0;
    
    // 处理无穷大情况
    if (isinf(result1) || isinf(result2)) {
        // 直接数学比较
        if (result1 <= result2) return 1;
        else return 0;
    }
    
    // 定义基础容差
    double epsilon = 1e-5;
    
    // 移项后的操作数：A = result1, B = result2
    double A = result1;
    double B = result2;
    
    // 计算动态容差
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    // 带容差的比较：检查是否违反 result1 <= result2
    // 违反条件：A > B + tol
    if (A > B + tol) {
        return 0;
    }
    
    // 满足条件
    return 1;
}
