int z2check(double result1, double result2) {
    // Core evaluation logic (direct scalar comparison) with floating-point tolerance and NaN/Inf handling
    double epsilon = 1e-5;

    // Rearranged inequality: result1 > result2
    // A = result1, B = result2
    double A = result1;
    double B = result2;

    // NaN check
    if (std::isnan(A) || std::isnan(B))
        return 0;

    // Inf check
    if (std::isinf(A) || std::isinf(B)) {
        if (A > B)
            return 1;
        else
            return 0;
    }

    // Dynamic tolerance
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

    // Check condition result1 > result2 with tolerance:
    // if A <= B + tol, violation
    if (!(A > B + tol))
        return 0;

    return 1;
}
