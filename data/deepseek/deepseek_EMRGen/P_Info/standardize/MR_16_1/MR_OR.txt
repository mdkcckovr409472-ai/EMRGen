#include <cmath>
#include <cfloat>

int z16check(double* result1, double* result2, int n) {
    // 1. 防御性编程 (空指针与长度检查)
    if (result1 == NULL || result2 == NULL) return 0;
    if (n <= 0) return 1;

    // 2. 字典序比较逻辑 (lexically)
    for (int i = 0; i < n; i++) {
        double A = result1[i];
        double B = result2[i];
        
        // 防御 NaN
        if (std::isnan(A) || std::isnan(B)) return 0;
        
        // 处理无穷大
        if (std::isinf(A) || std::isinf(B)) {
            if (A < B) {
                // 在第一个不相等的元素处，result1[i] < result2[i]，字典序条件满足，立即返回1
                return 1;
            } else if (A > B) {
                // 在第一个不相等的元素处，result1[i] > result2[i]，字典序条件不满足，立即返回0
                return 0;
            }
            // 如果相等（都是正无穷或都是负无穷），继续检查下一个元素
            continue;
        }
        
        // 浮点数容差比较
        double epsilon = 1e-5;
        double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
        
        // 检查是否不相等（考虑容差）
        if (fabs(A - B) > tol) {
            // 找到第一个不相等的元素对
            if (A < B - tol) {
                // result1[i] < result2[i]，字典序条件满足，立即返回1
                return 1;
            } else {
                // result1[i] > result2[i]，字典序条件不满足，立即返回0
                return 0;
            }
        }
        // 如果相等（在容差范围内），继续检查下一个元素
    }
    
    // 3. 所有元素都相等（在容差范围内），满足 <= 条件，返回1
    return 1;
}
