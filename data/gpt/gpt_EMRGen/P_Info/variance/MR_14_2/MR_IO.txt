TestCase z14Change(TestCase t1) {
    TestCase follow_case = t1;
    // Memory handling: deep copy of array x
    double* follow_case_x = (double*)malloc(t1.n * sizeof(double));
    memcpy(follow_case_x, t1.x, t1.n * sizeof(double));
    follow_case.x = follow_case_x;
    // Defensive check: ensure there is at least one element
    if (follow_case.n > 0) {
        // Sort array x in descending order
        for (int i = 0; i < follow_case.n - 1; ++i) {
            int maxIdx = i;
            for (int j = i + 1; j < follow_case.n; ++j) {
                if (follow_case.x[j] > follow_case.x[maxIdx]) {
                    maxIdx = j;
                }
            }
            if (maxIdx != i) {
                double temp = follow_case.x[i];
                follow_case.x[i] = follow_case.x[maxIdx];
                follow_case.x[maxIdx] = temp;
            }
        }
    }
    return follow_case;
}
