int check(TestCase tc) {
    int Os = sum(tc.values, tc.n);
    TestCase Tf_case = change(tc);
    int Of = sum(Tf_case.values, Tf_case.n);
    int is_valid = 0;
    if (2 * Os == Of) {
        is_valid = 1;
    }
    if (Tf_case.values != tc.values) {
        delete[] Tf_case.values;
    }
    return is_valid;
}
