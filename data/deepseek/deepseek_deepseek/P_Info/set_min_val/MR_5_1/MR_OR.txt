int check(TestCase tc) {
    int* Os = set_min_val(tc.a, tc.k, tc.n);
    TestCase Tf_case = change(tc);
    int* Of = set_min_val(Tf_case.a, Tf_case.k, Tf_case.n);
    int is_valid = 0;
    int all_less_equal = 1;
    for (int i = 0; i < tc.n; i++) {
        if (Os[i] > Of[i]) {
            all_less_equal = 0;
            break;
        }
    }
    if (all_less_equal) {
        is_valid = 1;
    }
    free(Os);
    free(Of);
    return is_valid;
}
