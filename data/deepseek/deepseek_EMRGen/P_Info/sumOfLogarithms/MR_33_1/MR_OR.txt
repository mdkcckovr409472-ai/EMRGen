#include <cmath>
#include <cstddef>

int z33check(double result1, double result2) {
    // 1. 核心判断逻辑 (标量直接比较)
    // 条件: result1 == result2
    // 浮点数比较需特殊处理

    // 第一步：定义基础容差
    const double epsilon = 1e-5;

    // 第二步：提取左右操作数（移项后）
    // 原条件: result1 == result2
    // 无需移项，直接使用
    double A = result1;
    double B = result2;

    // 第三步：防御NaN，处理Inf
    if (std::isnan(A) || std::isnan(B)) return 0;
    if (std::isinf(A) || std::isinf(B)) {
        // 无穷大时退化为直接比较
        if (A == B) return 1;
        else return 0;
    }

    // 第四步：计算动态宽容度
    double tol = epsilon * std::fmax(1.0, std::fmax(std::fabs(A), std::fabs(B)));

    // 第五步：带容差的相等判定
    if (std::fabs(A - B) > tol) {
        return 0;
    }

    // 2. 满足条件，返回 1
    return 1;
}
