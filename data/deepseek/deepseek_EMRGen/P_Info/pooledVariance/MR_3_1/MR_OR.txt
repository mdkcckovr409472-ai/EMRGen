#include <cmath>
#include <cstddef>

int z3check(double result1, double result2) {
    // 1. 防御性编程 (针对浮点数的特殊处理)
    // 移项后的条件: result1 <= result2
    double A = result1;
    double B = result2;

    // 检查 NaN
    if (std::isnan(A) || std::isnan(B)) return 0;

    // 处理无穷大情况
    if (std::isinf(A) || std::isinf(B)) {
        // 直接使用数学比较
        if (A <= B) return 1;
        else return 0;
    }

    // 2. 动态容差计算
    double epsilon = 1e-5;
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

    // 3. 核心判断逻辑 (带容差的 <= 比较)
    if (A > B + tol) {
        return 0;
    }

    // 4. 满足条件，返回 1
    return 1;
}
