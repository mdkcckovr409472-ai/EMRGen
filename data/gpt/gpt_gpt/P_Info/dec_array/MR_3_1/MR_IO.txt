TestCase change(TestCase tc) {
    // Reuse original test case but reverse the array 'a' in place using allocated memory
    int* reversed = (int*)malloc(tc.n * sizeof(int));
    for (int i = 0; i < tc.n; i++) {
        reversed[i] = tc.a[tc.n - 1 - i];
    }
    TestCase tf = tc;
    tf.a = reversed;
    return tf;
}
