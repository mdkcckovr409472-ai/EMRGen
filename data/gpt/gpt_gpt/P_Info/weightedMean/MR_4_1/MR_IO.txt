TestCase change(TestCase tc) {
    TestCase tf = tc;
    // Allocate new array for elements to hold divided-by-2 values
    double* newElements = new double[tf.n];
    for (int i = 0; i < tf.n; i++) {
        newElements[i] = tf.elements[i] / 2.0;
    }
    tf.elements = newElements;
    return tf;
}
