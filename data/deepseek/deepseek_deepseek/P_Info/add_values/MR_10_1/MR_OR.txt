int check(TestCase tc) {
    int Os = add_values(tc.a, tc.n);
    TestCase Tf_case = change(tc);
    int Of = add_values(Tf_case.a, Tf_case.n);
    int k = 5;
    int n = tc.n;
    int expected_diff = static_cast<int>(floor(n / 2.0)) * k;
    int is_valid = 0;
    if (Os - expected_diff == Of) {
        is_valid = 1;
    }
    delete[] Tf_case.a;
    return is_valid;
}
