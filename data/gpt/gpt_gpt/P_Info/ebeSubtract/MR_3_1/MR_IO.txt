TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tf.a != NULL && tf.n1 > 0) {
        double* a_new = (double*)malloc(tf.n1 * sizeof(double));
        if (a_new != NULL) {
            for (int i = 0; i < tf.n1; i++) {
                a_new[i] = tf.a[i];
            }
            a_new[0] = DBL_MAX;
            tf.a = a_new;
        }
    }
    return tf;
}
