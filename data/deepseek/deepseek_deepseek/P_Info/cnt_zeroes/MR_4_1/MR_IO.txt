TestCase change(TestCase tc) {
    TestCase tf = tc;
    // 为衍生用例分配新的数组内存，复制原始数据
    tf.a = new int[tc.n];
    for (int i = 0; i < tc.n; i++) {
        tf.a[i] = tc.a[i];
    }
    // 对数组进行降序排序
    for (int i = 0; i < tf.n - 1; i++) {
        for (int j = 0; j < tf.n - i - 1; j++) {
            if (tf.a[j] < tf.a[j + 1]) {
                int temp = tf.a[j];
                tf.a[j] = tf.a[j + 1];
                tf.a[j + 1] = temp;
            }
        }
    }
    return tf;
}
