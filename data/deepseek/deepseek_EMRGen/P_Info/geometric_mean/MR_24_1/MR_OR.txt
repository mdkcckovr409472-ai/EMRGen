#include <cmath>
#include <cstddef>

int z24check(double result1, double result2) {
    // 1. 防御 NaN
    if (std::isnan(result1) || std::isnan(result2)) {
        return 0;
    }

    // 2. 处理无穷大情况 (动态容差对Inf无效)
    if (std::isinf(result1) || std::isinf(result2)) {
        // 直接进行数学比较
        if (result1 <= result2) {
            return 1;
        } else {
            return 0;
        }
    }

    // 3. 定义基础容差并计算动态宽容度
    double epsilon = 1e-5;
    // 根据条件 return.result1 <= return.result2，移项后为 result1 <= result2
    // 因此 A = result1, B = result2
    double A = result1;
    double B = result2;
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

    // 4. 带容差的比较判定 (条件为 A <= B)
    if (A > B + tol) {
        return 0; // 违反条件
    }

    // 5. 满足条件
    return 1;
}
