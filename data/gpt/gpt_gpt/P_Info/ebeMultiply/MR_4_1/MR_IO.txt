TestCase change(TestCase tc) {
    TestCase tf = tc;
    // Allocate new memory for tf.a since we must divide all elements by 2 without altering original tc.a
    if (tc.n1 > 0 && tc.a != NULL) {
        tf.a = (double*)malloc(tc.n1 * sizeof(double));
        if (tf.a != NULL) {
            for (int i = 0; i < tc.n1; ++i) {
                tf.a[i] = tc.a[i] / 2.0;
            }
        }
        else {
            tf.a = NULL; // Allocation failed, set tf.a to NULL to indicate
        }
    }
    else {
        tf.a = NULL;
    }
    // Reuse all other fields unchanged (including tf.b, n1, n2)
    return tf;
}
