TestCase z15Change(TestCase t1) {
    TestCase follow_case = t1;
    // Memory handling code
    int* follow_case_a = (int*)malloc(t1.n * sizeof(int));
    memcpy(follow_case_a, t1.a, t1.n * sizeof(int));
    follow_case.a = follow_case_a;
    // Defensive check for empty array
    if (follow_case.n <= 0) {
        return follow_case;
    }
    // Decrement element at random index by 5 if result stays ≥ INT_MIN
    int idx = rand() % follow_case.n;
    if (follow_case.a[idx] >= INT_MIN + 5) {
        follow_case.a[idx] -= 5;
    }
    return follow_case;
}
