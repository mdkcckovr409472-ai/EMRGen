int check(TestCase tc) {
    int Os = find_max2(tc.a, tc.n);

    TestCase Tf_case = change(tc);
    int Of = find_max2(Tf_case.a, Tf_case.n);

    int is_valid = (Os == Of) ? 1 : 0;

    // Free dynamically allocated memory in Tf_case only if it is different from original
    if (Tf_case.a != tc.a) {
        delete[] Tf_case.a;
    }

    return is_valid;
}
