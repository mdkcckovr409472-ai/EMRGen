#include <math.h>
#include <stddef.h>

int z58check(double result1, double result2) {
    // 1. 防御性编程 (标量版本无需指针和长度检查)
    // 2. 核心判断逻辑 (标量直接比较)
    // 对于浮点数，必须使用带容差的比较
    double epsilon = 1e-5;
    
    // 提取移项后的操作数 A 和 B
    // 条件: return.result1 >= return.result2
    // 移项后: result1 >= result2
    double A = result1;
    double B = result2;
    
    // 防御 NaN
    if (isnan(A) || isnan(B)) return 0;
    
    // 处理无穷大情况
    if (isinf(A) || isinf(B)) {
        // 对于无穷大，退化为数学直接比较
        if (A >= B) return 1;
        else return 0;
    }
    
    // 计算动态容差
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    // 带容差的 >= 判断
    // 如果 A < B - tol，则违反条件
    if (A < B - tol) {
        return 0;
    }
    
    // 3. 满足条件，返回 1
    return 1;
}
