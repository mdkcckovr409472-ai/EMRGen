int check(TestCase tc) {
    int n = tc.n;
    int* Os = s_add(tc.array1, tc.array2, n);
    if (Os == NULL) {
        return 0;
    }
    TestCase Tf_case = change(tc);
    int* Of = s_add(Tf_case.array1, Tf_case.array2, n);
    if (Of == NULL) {
        free(Os);
        free(Tf_case.array1);
        return 0;
    }

    int valid = 1;
    for(int i = 0; i < n; i++) {
        if (i < n / 2) {
            if (Of[i] != Os[i] + 1) {
                valid = 0;
                break;
            }
        } else {
            if (Of[i] != Os[i]) {
                valid = 0;
                break;
            }
        }
    }

    free(Os);
    free(Of);
    free(Tf_case.array1);

    return valid;
}
