TestCase change(TestCase tc) {
    TestCase tf = tc;
    double c = 2.0; // 非零常数，示例使用2.0
    int size = tf.n;
    double* new_elements = new double[size];
    for (int i = 0; i < size; ++i) {
        new_elements[i] = tf.elements[i] * c;
    }
    tf.elements = new_elements;
    tf.mean = tf.mean * c;
    return tf;
}
