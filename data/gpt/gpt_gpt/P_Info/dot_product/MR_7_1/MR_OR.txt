int check(TestCase tc) {
    int Os = dot_product(tc.a, tc.b, tc.n);
    TestCase Tf_case = change(tc);
    int Of = dot_product(Tf_case.a, Tf_case.b, Tf_case.n);

    int sum_a = 0;
    for (int i = 0; i < tc.n; i++) {
        sum_a += tc.a[i];
    }

    int is_valid = (Of == Os - sum_a) ? 1 : 0;

    delete[] Tf_case.b;

    return is_valid;
}
