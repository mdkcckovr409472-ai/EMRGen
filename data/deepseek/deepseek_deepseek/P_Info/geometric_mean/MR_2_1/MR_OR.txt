int check(TestCase tc) {
    if (tc.n <= 0 || tc.a == nullptr) {
        return 1;
    }
    
    double Os = geometric_mean(tc.a, tc.n);
    
    TestCase Tf_case = change(tc);
    
    double Of = geometric_mean(Tf_case.a, Tf_case.n);
    
    int is_valid = 0;
    const double epsilon = 1e-9;
    if (std::abs(Os - Of) < epsilon) {
        is_valid = 1;
    }
    
    return is_valid;
}
