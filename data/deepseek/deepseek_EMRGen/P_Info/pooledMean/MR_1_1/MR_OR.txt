#include <math.h>
#include <stddef.h>

int z1check(double result1, double result2) {
    // 浮点数比较特殊处理
    double epsilon = 1e-5;
    
    // 提取左右操作数（移项后：result1 < result2）
    double A = result1;
    double B = result2;
    
    // 防御NaN
    if (isnan(A) || isnan(B)) return 0;
    
    // 处理无穷大情况
    if (isinf(A) || isinf(B)) {
        if (A < B) return 1;
        else return 0;
    }
    
    // 计算动态容差
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    // 带容差的严格小于比较
    // 对于 A < B，违反条件是 A >= B - tol
    if (A >= B - tol) {
        return 0;
    }
    
    // 满足条件
    return 1;
}
