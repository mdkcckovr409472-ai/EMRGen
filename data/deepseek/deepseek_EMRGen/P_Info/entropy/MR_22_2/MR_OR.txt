#include <cmath>
#include <cstddef>

int z22check(double result1, double result2) {
    // 核心判断逻辑 (标量直接比较)
    // 条件: return.result1 - return.result2 == 0
    // 移项后: result1 == result2

    // 提取移项后的左右操作数
    double A = result1;
    double B = result2;

    // 防御 NaN
    if (std::isnan(A) || std::isnan(B)) return 0;

    // 处理无穷大 (动态容差对无穷大无效)
    if (std::isinf(A) || std::isinf(B)) {
        // 退化为数学上的直接比较
        if (A == B) return 1;
        else return 0;
    }

    // 定义基础容差
    double epsilon = 1e-5;
    // 计算动态宽容度
    double tol = epsilon * std::fmax(1.0, std::fmax(std::fabs(A), std::fabs(B)));

    // 带容差的相等判定 (移项后为 ==)
    if (std::fabs(A - B) > tol) {
        return 0;
    }

    // 满足条件，返回 1
    return 1;
}
