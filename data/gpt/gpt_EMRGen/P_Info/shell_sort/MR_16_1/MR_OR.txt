#include <cmath>
int z16check(int* result1, int* result2, int n) {
    // 4a) Length validity check
    if (n <= 0) return 1;
    // 4b) Null pointer check
    if (result1 == NULL || result2 == NULL) return 0;

    // Lexicographical comparison: return.result1[] < return.result2[] (lexically)
    // int arrays, no floating tolerance needed
    for (int i = 0; i < n; i++) {
        if (result1[i] < result2[i]) {
            return 1; // first unequal pair satisfies <
        } else if (result1[i] > result2[i]) {
            return 0; // first unequal pair violates <
        }
        // if equal, continue to next element
    }
    // arrays are equal, strictly less means return 0
    return 0;
}
