#include <math.h>
#include <stddef.h>

int z52check(double result1, double result2) {
    // 1. 核心判断逻辑 (标量直接比较)
    // 移项处理: return.result1 - return.result2 == 0 等价于 result1 == result2
    double A = result1;
    double B = result2;

    // 防御 NaN
    if (isnan(A) || isnan(B)) return 0;

    // 处理无穷大
    if (isinf(A) || isinf(B)) {
        // 无穷大时直接进行数学比较
        if (A == B) return 1;
        else return 0;
    }

    // 定义基础容差
    double epsilon = 1e-5;
    // 计算动态宽容度
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

    // 带容差的相等判断
    if (fabs(A - B) > tol) {
        return 0;
    }

    // 2. 满足条件，返回 1
    return 1;
}
