TestCase change(TestCase tc) {
    TestCase tf = tc;
    int* new_a = (int*)malloc(tf.n * sizeof(int));
    memcpy(new_a, tf.a, tf.n * sizeof(int));
    for (int i = 0; i < tf.n; i++) {
        if (new_a[i] < tf.k) {
            new_a[i] = tf.k - 1;
        }
    }
    tf.a = new_a;
    return tf;
}
