#include <cmath>
#include <cfloat>

int z6check(double result1, double result2) {
    // 1. 防御 NaN
    if (std::isnan(result1) || std::isnan(result2)) {
        return 0;
    }

    // 2. 处理无穷大 (Inf - Inf = NaN，动态容差失效，需直接比较)
    if (std::isinf(result1) || std::isinf(result2)) {
        // 根据移项后的条件 result1 <= result2 直接比较
        if (result1 <= result2) {
            return 1;
        } else {
            return 0;
        }
    }

    // 3. 定义基础容差
    const double epsilon = 1e-5;

    // 4. 移项后条件为 result1 <= result2
    // 提取 A = result1, B = result2
    double A = result1;
    double B = result2;

    // 5. 计算动态宽容度
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

    // 6. 带容差的比较判定 (若违反则 return 0)
    // 条件 A <= B 等价于 A > B + tol 时违反
    if (A > B + tol) {
        return 0;
    }

    // 7. 满足条件，返回 1
    return 1;
}
