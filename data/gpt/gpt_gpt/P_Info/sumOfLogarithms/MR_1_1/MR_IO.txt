TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tc.n > 0) {
        double* new_elements = new double[tc.n];
        for (int i = 0; i < tc.n; ++i) {
            new_elements[i] = tc.elements[i] + 1.0;
        }
        tf.elements = new_elements;
    } else {
        tf.elements = nullptr;
    }
    return tf;
}
