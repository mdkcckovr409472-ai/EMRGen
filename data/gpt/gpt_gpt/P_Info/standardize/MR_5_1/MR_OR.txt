int check(TestCase tc) {
    double* Os = standardize(tc.data, tc.n);
    TestCase Tf_case = change(tc);
    double* Of = standardize(Tf_case.data, Tf_case.n);
    int is_valid = 0;
    if (Os != nullptr && Of != nullptr) {
        // Check MR: Os != Of
        bool different = false;
        for (int i = 0; i < tc.n; i++) {
            if (Os[i] != Of[i]) {
                different = true;
                break;
            }
        }
        if (different) {
            is_valid = 1;
        }
    }
    free(Os);
    free(Of);
    return is_valid;
}
