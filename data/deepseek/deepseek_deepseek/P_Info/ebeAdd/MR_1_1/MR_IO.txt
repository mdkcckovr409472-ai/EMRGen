TestCase change(TestCase tc) {
    TestCase tf = tc;
    double k = 5.0; // 任意常数，例如 5.0
    int n = tf.n1;
    if (tf.a != NULL) {
        double* new_a = (double*)malloc(n * sizeof(double));
        if (new_a != NULL) {
            for (int i = 0; i < n; i++) {
                new_a[i] = tf.a[i] + k;
            }
            tf.a = new_a;
        }
    }
    return tf;
}
