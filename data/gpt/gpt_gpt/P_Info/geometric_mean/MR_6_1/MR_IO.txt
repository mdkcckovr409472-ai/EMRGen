TestCase change(TestCase tc) {
    TestCase tf = tc;
    // Allocate new array to avoid modifying original input and to follow reuse principle by creating a new pointer
    tf.a = new int[tc.n];
    // Copy original array
    for (int i = 0; i < tc.n; i++) {
        tf.a[i] = tc.a[i];
    }
    // Sort in descending order as per MR rule
    std::sort(tf.a, tf.a + tf.n, std::greater<int>());
    return tf;
}
