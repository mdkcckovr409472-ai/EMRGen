TestCase z17Change(TestCase t1) {
    TestCase follow_case = t1;
    // --- Memory handling code ---
    double* follow_case_k = (double*)malloc(t1.n * sizeof(double));
    memcpy(follow_case_k, t1.k, t1.n * sizeof(double));
    follow_case.k = follow_case_k;
    // ---------------------------

    // Defensive check: ensure there is at least one element
    if (follow_case.n == 0) {
        return follow_case;
    }

    // Negate all elements
    for (int i = 0; i < follow_case.n; ++i) {
        follow_case.k[i] = -follow_case.k[i];
    }

    return follow_case;
}
