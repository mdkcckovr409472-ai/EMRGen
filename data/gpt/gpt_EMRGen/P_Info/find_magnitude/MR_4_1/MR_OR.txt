int z4check(double result1, double result2) {
    // Core evaluation logic (direct scalar comparison)

    // Step 1: Define base tolerance
    double epsilon = 1e-5;

    // Step 2: Rearrange inequality result1 >= result2 as A >= B with A = result1, B = result2
    double A = result1;
    double B = result2;

    // Step 3: Defend against NaN and Inf
    if (std::isnan(A) || std::isnan(B)) return 0;
    if (std::isinf(A) || std::isinf(B)) {
        if (A >= B) return 1;
        else return 0;
    }

    // Step 4: Calculate dynamic tolerance
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

    // Step 5: Tolerance-based comparison evaluation for >=
    if (A < B - tol) return 0;

    // Satisfied
    return 1;
}
