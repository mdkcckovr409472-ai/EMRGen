int check(TestCase tc) {
    int* Os = dec_array(tc.a, tc.k, tc.n);
    if (!Os) return 0;

    TestCase Tf_case = change(tc);
    if (!Tf_case.a) {
        free(Os);
        return 0;
    }
    int* Of = dec_array(Tf_case.a, Tf_case.k, Tf_case.n);
    if (!Of) {
        free(Os);
        free(Tf_case.a);
        return 0;
    }

    int is_valid = 1;
    for (int i = 0; i < tc.n; i++) {
        if (Os[i] != Of[i]) {
            is_valid = 0;
            break;
        }
    }

    free(Os);
    free(Of);
    free(Tf_case.a);

    return is_valid;
}
