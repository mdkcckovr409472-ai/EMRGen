int check(TestCase tc) {
    // Get source output
    int* Os = dec_array(tc.a, tc.k, tc.n);

    // Construct follow-up test case
    TestCase Tf_case = change(tc);

    // Get follow-up output
    int* Of = dec_array(Tf_case.a, Tf_case.k, Tf_case.n);

    // Verify if Os and Of satisfy the MR: Os == Of (element-wise equality)
    int is_valid = 1;
    for (int i = 0; i < tc.n; i++) {
        if (Os[i] != Of[i]) {
            is_valid = 0;
            break;
        }
    }

    // Free allocated memory
    free(Os);
    free(Of);
    free(Tf_case.a);

    return is_valid;
}
