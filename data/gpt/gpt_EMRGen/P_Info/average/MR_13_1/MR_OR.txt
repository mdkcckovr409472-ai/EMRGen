int z13check(double result1, double result2) {
    // Core evaluation logic (direct scalar comparison)

    // Step 1: Define epsilon
    double epsilon = 1e-5;
    // Step 2: Rearrangement for ==
    // For return.result1 == return.result2  => compare result1 and result2 directly

    double A = result1;
    double B = result2;

    // Step 3: NaN and Inf handling
    #include <math.h>
    if (isnan(A) || isnan(B)) return 0;
    if (isinf(A) || isinf(B)) {
        if (A == B) return 1;
        else return 0;
    }

    // Step 4: Calculate dynamic tolerance
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

    // Step 5: Tolerance-based comparison for equality
    if (fabs(A - B) > tol) return 0;

    // Satisfied
    return 1;
}
