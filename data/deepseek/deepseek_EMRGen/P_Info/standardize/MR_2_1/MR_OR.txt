#include <cmath>
#include <algorithm>

int z2check(double* result1, double* result2, int n) {
    // 1. 防御性编程 (空指针与长度检查)
    if (result1 == NULL || result2 == NULL) return 0;
    if (n <= 0) return 1;

    // 2. 核心判断逻辑 (elementwise)
    for (int i = 0; i < n; i++) {
        // 浮点数比较特殊处理
        double A = result1[i];
        double B = result2[i];
        
        // 防御 NaN
        if (std::isnan(A) || std::isnan(B)) return 0;
        
        // 处理无穷大情况
        if (std::isinf(A) || std::isinf(B)) {
            if (!(A == B)) return 0;
            continue;
        }
        
        // 动态容差计算
        double epsilon = 1e-5;
        double tol = epsilon * std::max(1.0, std::max(std::fabs(A), std::fabs(B)));
        
        // 带容差的相等判断 (移项后为 A == B)
        if (std::fabs(A - B) > tol) {
            return 0;
        }
    }

    // 3. 全部满足，返回 1
    return 1;
}
