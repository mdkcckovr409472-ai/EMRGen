TestCase z2Change(TestCase t1) {
    TestCase follow_case = t1;
    // Memory handling for deep copy of sample1
    double* follow_case_sample1 = (double*)malloc(t1.n * sizeof(double));
    memcpy(follow_case_sample1, t1.sample1, t1.n * sizeof(double));
    follow_case.sample1 = follow_case_sample1;
    // Defensive check for empty array
    if (follow_case.n <= 0) {
        return follow_case;
    }
    // Subtract 2.5 from all elements of sample1
    for (int i = 0; i < follow_case.n; ++i) {
        follow_case.sample1[i] -= 2.5;
    }
    return follow_case;
}
