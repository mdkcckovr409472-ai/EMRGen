TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tf.n > 1 && tf.elements != nullptr) {
        double first_element = tf.elements[0];
        for (int i = 0; i < tf.n - 1; ++i) {
            tf.elements[i] = tf.elements[i + 1];
        }
        tf.elements[tf.n - 1] = first_element;
    }
    return tf;
}
