TestCase change(TestCase tc) {
    TestCase tf = tc;
    // Allocate new array for modification to satisfy test case reuse principle
    int* new_a = (int*)malloc(tf.n * sizeof(int));
    if (!new_a) return tf; // fallback if malloc fails, though unlikely in test framework
    memcpy(new_a, tf.a, tf.n * sizeof(int));

    int half = tf.n / 2;
    for (int i = 0; i < half; ++i) {
        new_a[i] += 1;
    }
    // Overwrite a pointer in tf with newly allocated modified array
    tf.a = new_a;
    return tf;
}
