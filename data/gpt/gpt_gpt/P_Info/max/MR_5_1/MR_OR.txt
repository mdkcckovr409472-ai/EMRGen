int check(TestCase tc) {
    // (1) Extract the source input
    double* Ts = tc.elements;
    int n = tc.n;
    
    // (2) Get source output Os
    double Os = max(Ts, n);
    
    // (3) Construct follow-up test case Tf_case
    TestCase Tf_case = change(tc);
    
    // (4) Extract follow-up input and get output Of
    double* Tf = Tf_case.elements;
    double Of = max(Tf, Tf_case.n);
    
    // (5) Verify Os == Of
    int is_valid = (Os == Of) ? 1 : 0;
    
    // (6) Free dynamic memory allocated in change()
    if (Tf != Ts) {
        delete[] Tf;
    }
    
    // (7) Return verification result
    return is_valid;
}
