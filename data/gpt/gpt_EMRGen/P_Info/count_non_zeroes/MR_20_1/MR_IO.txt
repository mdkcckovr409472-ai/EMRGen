TestCase z20Change(TestCase t1) {
    TestCase follow_case = t1;
    // --- Memory handling code ---
    int* follow_case_a = (int*)malloc(t1.n * sizeof(int));
    memcpy(follow_case_a, t1.a, t1.n * sizeof(int));
    follow_case.a = follow_case_a;
    // --------------------------------

    // Defensive check for empty array
    if (follow_case.n == 0) {
        return follow_case;
    }

    // Set elements at even indices to absolute value
    for (int i = 0; i < follow_case.n; ++i) {
        if (i % 2 == 0) {
            int val = follow_case.a[i];
            follow_case.a[i] = (val < 0) ? -val : val;
        }
    }

    return follow_case;
}
