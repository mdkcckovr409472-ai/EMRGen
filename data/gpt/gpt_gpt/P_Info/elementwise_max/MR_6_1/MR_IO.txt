TestCase change(TestCase tc) {
    TestCase tf = tc;

    if (tf.b != NULL && tf.n > 0) {
        int* new_b = (int*)malloc(tf.n * sizeof(int));
        if (new_b != NULL) {
            for (int i = 0; i < tf.n; i++) {
                new_b[i] = tf.b[i];
            }
            new_b[0] = 2147483647;
            tf.b = new_b;
        }
    }

    return tf;
}
