TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tc.a != nullptr && tc.n > 1) {
        int* sorted_arr = new int[tc.n];
        for (int i = 0; i < tc.n; i++) {
            sorted_arr[i] = tc.a[i];
        }
        std::sort(sorted_arr, sorted_arr + tc.n);
        tf.a = sorted_arr;
    }
    return tf;
}
