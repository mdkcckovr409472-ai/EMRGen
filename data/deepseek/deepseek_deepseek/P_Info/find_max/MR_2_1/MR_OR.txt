int check(TestCase tc) {
    int Os = find_max(tc.a, tc.n);
    TestCase Tf_case = change(tc);
    int Of = find_max(Tf_case.a, Tf_case.n);
    int k = 5; // 必须与 change 函数中使用的 k 值一致
    int is_valid = 0;
    if (Os - k == Of) {
        is_valid = 1;
    }
    delete[] Tf_case.a;
    return is_valid;
}
