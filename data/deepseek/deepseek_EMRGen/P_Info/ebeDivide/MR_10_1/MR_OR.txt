#include <cmath>
#include <cfloat>

int z10check(double* result1, double* result2, int n) {
    // 1. 防御性编程 (空指针与长度检查)
    if (result1 == NULL || result2 == NULL) return 0;
    if (n <= 0) return 1;

    // 2. 核心判断逻辑 (elementwise)
    for (int i = 0; i < n; i++) {
        // 根据条件 2 * result1[i] - result2[i] == 0 进行代数移项
        // 移项后为: 2 * result1[i] == result2[i]
        double A = 2.0 * result1[i];
        double B = result2[i];

        // 防御 NaN
        if (std::isnan(A) || std::isnan(B)) return 0;

        // 处理无穷大情况
        if (std::isinf(A) || std::isinf(B)) {
            if (A != B) return 0; // 移项后为相等比较，直接使用 !=
            // 如果相等，继续循环（即不返回0）
            continue;
        }

        // 定义基础容差并计算动态宽容度
        double epsilon = 1e-5;
        double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

        // 带容差的相等判定 (移项后为 ==)
        if (fabs(A - B) > tol) {
            return 0; // 一旦违反条件，立即返回 0
        }
    }

    // 3. 全部满足，返回 1
    return 1;
}
