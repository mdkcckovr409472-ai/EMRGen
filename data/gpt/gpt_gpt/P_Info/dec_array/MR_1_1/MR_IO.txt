TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tf.a != nullptr && tf.n > 0) {
        int* new_a = (int*)malloc(tf.n * sizeof(int));
        if (new_a == nullptr) {
            tf.a = nullptr;
            tf.n = 0;
            return tf;
        }
        for (int i = 0; i < tf.n; i++) {
            new_a[i] = tf.a[i] + 1;
        }
        tf.a = new_a;
    }
    return tf;
}
