TestCase change(TestCase tc) {
    TestCase tf = tc; 
    if (tf.n2 > 0) {
        double max_val = tf.data2[0];
        for (int i = 1; i < tf.n2; i++) {
            if (tf.data2[i] > max_val) {
                max_val = tf.data2[i];
            }
        }
        tf.data2[tf.n2 - 1] = max_val;
    }
    return tf;
}
