TestCase z15Change(TestCase t1) {
    TestCase follow_case = t1;
    // Memory handling: deep copy of array
    int* follow_case_array = (int*)malloc(t1.n * sizeof(int));
    memcpy(follow_case_array, t1.array, t1.n * sizeof(int));
    follow_case.array = follow_case_array;
    // Defensive check for empty array
    if (follow_case.n <= 0) {
        return follow_case;
    }
    // Right circular shift: move last element to front
    int last = follow_case.array[follow_case.n - 1];
    for (int i = follow_case.n - 1; i > 0; --i) {
        follow_case.array[i] = follow_case.array[i - 1];
    }
    follow_case.array[0] = last;
    return follow_case;
}
