int check(TestCase tc) {
    int* Os = insertion_sort(tc.array, tc.n);
    TestCase Tf_case = change(tc);
    int* Of = insertion_sort(Tf_case.array, Tf_case.n);
    int is_valid = 1;
    for(int i = 0; i < tc.n; i++){
        if(Os[i] < Of[i]){
            is_valid = 0;
            break;
        }
    }
    free(Os);
    free(Of);
    return is_valid;
}
