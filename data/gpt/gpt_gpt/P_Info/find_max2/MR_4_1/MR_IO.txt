TestCase change(TestCase tc) {
    TestCase tf = tc;
    // Reverse the entire array in place by creating a new array, to avoid modifying original array
    int* reversed = new int[tf.n];
    for (int i = 0; i < tf.n; i++) {
        reversed[i] = tf.a[tf.n - 1 - i];
    }
    tf.a = reversed;
    return tf;
}
