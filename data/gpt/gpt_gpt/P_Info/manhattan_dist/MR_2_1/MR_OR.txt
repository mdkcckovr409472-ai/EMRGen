int check(TestCase tc) {
    double Os = manhattan_dist(tc.a, tc.b, tc.n);
    TestCase Tf_case = change(tc);
    double Of = manhattan_dist(Tf_case.a, Tf_case.b, Tf_case.n);
    int is_valid = 0;
    if (Os >= 0 && Of >= 0) {
        is_valid = 1;
    }
    if (Tf_case.a != tc.a) {
        delete[] Tf_case.a;
    }
    return is_valid;
}
