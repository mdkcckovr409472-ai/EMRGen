#include <cmath>
#include <cfloat>

int z3check(double result1, double result2) {
    // 1. 防御性编程：检查 NaN
    if (std::isnan(result1) || std::isnan(result2)) return 0;
    
    // 2. 处理无穷大情况
    if (std::isinf(result1) || std::isinf(result2)) {
        // 直接数学比较：result1 < result2
        if (result1 < result2) return 1;
        else return 0;
    }
    
    // 3. 浮点数比较的特殊处理
    double epsilon = 1e-5;
    
    // 移项：条件为 result1 < result2
    // 直接提取：A = result1, B = result2
    double A = result1;
    double B = result2;
    
    // 计算动态容差
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    // 带容差的比较：检查是否违反 result1 < result2
    // 违反条件：A >= B - tol
    if (A >= B - tol) {
        return 0;
    }
    
    // 4. 满足条件，返回 1
    return 1;
}
