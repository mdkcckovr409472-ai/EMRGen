TestCase change(TestCase tc) {
    TestCase tf = tc;
    tf.data = new double[tc.n];
    for (int i = 0; i < tc.n; ++i) {
        tf.data[i] = tc.data[i];
    }
    for (int i = 0; i < tc.n - 1; ++i) {
        int j = i + rand() % (tc.n - i);
        double temp = tf.data[i];
        tf.data[i] = tf.data[j];
        tf.data[j] = temp;
    }
    return tf;
}
