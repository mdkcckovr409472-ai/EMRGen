TestCase z22Change(TestCase t1) {
    TestCase follow_case = t1;
    // --- Memory handling code ---
    int* follow_case_a = (int*)malloc(t1.n * sizeof(int));
    memcpy(follow_case_a, t1.a, t1.n * sizeof(int));
    follow_case.a = follow_case_a;
    // ---------------------------

    // Defensive check for array length
    if (follow_case.n <= 0) {
        return follow_case;
    }

    // Replace each element with its integer square root (when element ≥ 0)
    for (int i = 0; i < follow_case.n; ++i) {
        if (follow_case.a[i] >= 0) {
            follow_case.a[i] = (int)sqrt((double)follow_case.a[i]);
        }
    }

    return follow_case;
}
