TestCase z19Change(TestCase t1) {
    TestCase follow_case = t1;
    // Memory handling code for target parameter x
    double* follow_case_x = (double*)malloc(t1.n * sizeof(double));
    memcpy(follow_case_x, t1.x, t1.n * sizeof(double));
    follow_case.x = follow_case_x;

    // Defensive programming: ensure there is at least one element
    if (follow_case.n <= 0) {
        return follow_case;
    }

    // Apply natural logarithm to each positive element of x
    for (int i = 0; i < follow_case.n; ++i) {
        if (follow_case.x[i] > 0) {
            follow_case.x[i] = log(follow_case.x[i]);
        }
    }

    return follow_case;
}
