int check(TestCase tc) {
    bool Os = checkPositive(tc.a, tc.n);
    TestCase Tf_case = change(tc);
    bool Of = checkPositive(Tf_case.a, Tf_case.n);
    int is_valid = 0;
    if (Os == Of) {
        is_valid = 1;
    }
    if (Tf_case.n > 0 && Tf_case.a != tc.a) {
        delete[] Tf_case.a;
    }
    return is_valid;
}
