TestCase change(TestCase tc) {
    // Reuse the original test case and modify elements in place
    TestCase tf = tc;
    if (tf.n > 0 && tf.elements != nullptr) {
        double maxVal = tf.elements[0];
        for (int i = 1; i < tf.n; ++i) {
            if (tf.elements[i] > maxVal) {
                maxVal = tf.elements[i];
            }
        }
        tf.elements[0] = maxVal;
    }
    return tf;
}
