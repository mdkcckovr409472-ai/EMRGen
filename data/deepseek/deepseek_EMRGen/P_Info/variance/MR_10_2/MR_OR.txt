#include <math.h>
#include <stddef.h>

int z10check(double result1, double result2) {
    // 1. 防御性编程 (标量情况无需空指针和长度检查)

    // 2. 核心判断逻辑 (标量直接比较)
    // 条件: result1 <= result2
    // 移项后: A = result1, B = result2, 比较符为 <=

    // 浮点数特殊处理
    double epsilon = 1e-5;
    
    double A = result1;
    double B = result2;
    
    // 防御NaN
    if (isnan(A) || isnan(B)) return 0;
    
    // 处理无穷大情况
    if (isinf(A) || isinf(B)) {
        // 无穷大时退化为数学直接比较
        if (A <= B) return 1;
        else return 0;
    }
    
    // 计算动态容差
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    // 带容差的比较判定
    // 条件: A <= B
    // 违反条件: A > B + tol
    if (A > B + tol) {
        return 0;
    }
    
    // 3. 满足条件，返回 1
    return 1;
}
