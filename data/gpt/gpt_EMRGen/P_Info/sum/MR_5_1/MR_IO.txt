TestCase z5Change(TestCase t1) {
    TestCase follow_case = t1;
    // Memory handling: deep copy of the array
    int* follow_case_values = (int*)malloc(t1.n * sizeof(int));
    memcpy(follow_case_values, t1.values, t1.n * sizeof(int));
    follow_case.values = follow_case_values;

    // Defensive check: ensure there is at least one element
    if (follow_case.n == 0) {
        return follow_case;
    }

    // Operation: set first element to 0 when it is positive
    if (follow_case.values[0] > 0) {
        follow_case.values[0] = 0;
    }

    return follow_case;
}
