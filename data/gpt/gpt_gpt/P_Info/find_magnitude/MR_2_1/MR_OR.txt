int check(TestCase tc) {
    if (tc.a == nullptr || tc.n <= 0) {
        // Undefined or trivial case, but still run and verify
        double Os = find_magnitude(tc.a, tc.n);
        TestCase Tf_case = change(tc);
        double Of = find_magnitude(Tf_case.a, Tf_case.n);

        if (Tf_case.a != tc.a) {
            delete[] Tf_case.a;
        }

        return (Os == Of) ? 1 : 0;
    }

    double Os = find_magnitude(tc.a, tc.n);

    TestCase Tf_case = change(tc);
    double Of = find_magnitude(Tf_case.a, Tf_case.n);

    int is_valid = (Os == Of) ? 1 : 0;

    if (Tf_case.a != tc.a) {
        delete[] Tf_case.a;
    }

    return is_valid;
}
