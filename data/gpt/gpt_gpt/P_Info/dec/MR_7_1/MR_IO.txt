TestCase change(TestCase tc) {
    TestCase tf = tc;

    // Allocate new array2 for follow-up input and add 1 to each element
    int* new_array2 = (int*)malloc(tc.n * sizeof(int));
    for(int i = 0; i < tc.n; i++) {
        new_array2[i] = tc.array2[i] + 1;
    }
    tf.array2 = new_array2;

    return tf;
}
