TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tf.n > 0) {
        tf.elements = new double[tf.n];
        for (int i = 0; i < tf.n; ++i) {
            tf.elements[i] = tc.elements[i];
        }
        int index_to_modify = 0;
        tf.elements[index_to_modify] *= 0.5;
    }
    return tf;
}
