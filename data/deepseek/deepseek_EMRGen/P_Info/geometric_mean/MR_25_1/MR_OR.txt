#include <math.h>
#include <stddef.h>

int z25check(double result1, double result2) {
    // 1. 防御性编程 (针对浮点数)
    // 检查 NaN
    if (isnan(result1) || isnan(result2)) return 0;
    
    // 2. 处理无穷大 (动态容差对无穷大无效)
    if (isinf(result1) || isinf(result2)) {
        // 直接进行数学比较
        if (result1 >= result2) return 1;
        else return 0;
    }
    
    // 3. 浮点数比较的特殊处理
    double epsilon = 1e-5;
    
    // 移项后的条件: result1 >= result2
    double A = result1;
    double B = result2;
    
    // 计算动态容差
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    // 4. 核心判断逻辑 (带容差的 >= 比较)
    // 条件: A >= B
    // 违反条件: A < B - tol
    if (A < B - tol) {
        return 0;
    }
    
    // 5. 满足条件，返回 1
    return 1;
}
