int check(TestCase tc) {
    // 提取初始输入并计算初始输出
    double Os = harmonicMean(tc.data, tc.n);
    
    // 构造衍生用例
    TestCase Tf_case = change(tc);
    
    // 计算衍生输出
    double Of = harmonicMean(Tf_case.data, Tf_case.n);
    
    // 验证蜕变关系 Os == Of
    int is_valid = 0;
    const double epsilon = 1e-12;
    if (std::abs(Os - Of) < epsilon) {
        is_valid = 1;
    }
    
    // 释放 change 函数中分配的动态内存
    delete[] Tf_case.data;
    
    return is_valid;
}
