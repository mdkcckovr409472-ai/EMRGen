TestCase change(TestCase tc) {
    TestCase tf = tc;
    double m = 0.0;
    if (tf.n2 > 0 && tf.data2 != nullptr) {
        for (int i = 0; i < tf.n2; i++) {
            m += tf.data2[i];
        }
        m = m / tf.n2;
        for (int i = 0; i < tf.n2; i++) {
            tf.data2[i] = m;
        }
    }
    return tf;
}
