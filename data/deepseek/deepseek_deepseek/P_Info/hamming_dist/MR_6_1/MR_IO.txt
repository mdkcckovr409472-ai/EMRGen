TestCase change(TestCase tc) {
    TestCase tf = tc;
    tf.b = new int[tc.n];
    for (int i = 0; i < tc.n; i++) {
        tf.b[i] = tc.b[i];
    }
    for (int i = 0; i < tc.n; i++) {
        if (tc.a[i] == tc.b[i]) {
            tf.b[i] = tc.a[i] + 1;
            break;
        }
    }
    return tf;
}
