int check(TestCase tc) {
    // Source input
    int* Ts = tc.a;
    int n = tc.n;

    // Obtain source output
    double Os = geometric_mean(Ts, n);

    // Construct follow-up test case
    TestCase Tf_case = change(tc);

    // Obtain follow-up output
    double Of = geometric_mean(Tf_case.a, Tf_case.n);

    // Verify metamorphic relation Os > Of
    int is_valid = (Os > Of) ? 1 : 0;

    // Clean up dynamically allocated memory in follow-up test case
    if (Tf_case.a != tc.a) {
        delete[] Tf_case.a;
    }

    return is_valid;
}
