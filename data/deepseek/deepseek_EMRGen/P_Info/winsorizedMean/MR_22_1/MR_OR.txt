#include <cmath>
#include <cfloat>

int z22check(double result1, double result2) {
    // 1. 防御NaN
    if (std::isnan(result1) || std::isnan(result2)) return 0;
    
    // 2. 处理无穷大（直接比较）
    if (std::isinf(result1) || std::isinf(result2)) {
        if (result1 == result2) return 1;
        else return 0;
    }
    
    // 3. 动态容差比较（移项后条件：result1 == result2）
    double epsilon = 1e-5;
    double A = result1;
    double B = result2;
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    if (fabs(A - B) > tol) return 0;
    
    // 4. 满足条件
    return 1;
}
