#include <cmath>
#include <algorithm>

int z4check(double* result1, double* result2, int n) {
    // 1. 防御性编程 (空指针与长度检查)
    if (result1 == NULL || result2 == NULL) return 0;
    if (n <= 0) return 1;

    // 2. 核心判断逻辑 (elementwise)
    for (int i = 0; i < n; i++) {
        // 提取移项后的操作数
        double A = result1[i];
        double B = result2[i];

        // 防御 NaN
        if (std::isnan(A) || std::isnan(B)) return 0;

        // 处理无穷大
        if (std::isinf(A) || std::isinf(B)) {
            // 无穷大时直接数学比较
            if (!(A >= B)) return 0;
            continue; // 此对元素检查通过，继续下一对
        }

        // 定义基础容差
        double epsilon = 1e-5;
        // 计算动态宽容度
        double tol = epsilon * std::max(1.0, std::max(std::fabs(A), std::fabs(B)));

        // 根据移项后的条件 (A >= B) 进行带容差判定
        // 若违反 A >= B，则返回 0
        if (A < B - tol) {
            return 0;
        }
        // 若满足条件，继续循环检查下一对元素
    }

    // 3. 全部元素对满足条件，返回 1
    return 1;
}
