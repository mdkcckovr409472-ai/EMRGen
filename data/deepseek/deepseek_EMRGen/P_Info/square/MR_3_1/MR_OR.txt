#include <cmath>
#include <cfloat>

int z3check(double* result1, double* result2, int n) {
    // 1. 防御性编程 (空指针与长度检查)
    if (result1 == NULL || result2 == NULL) return 0;
    if (n <= 0) return 1;

    // 2. 核心判断逻辑 (elementwise)
    for (int i = 0; i < n; i++) {
        // 根据规则3，对浮点数条件进行移项和容差比较
        // 原始条件: 9 * result1[i] - result2[i] == 0
        // 移项后: 9 * result1[i] == result2[i]
        double A = 9.0 * result1[i];
        double B = result2[i];

        // 防御 NaN
        if (std::isnan(A) || std::isnan(B)) return 0;

        // 处理无穷大
        if (std::isinf(A) || std::isinf(B)) {
            if (!(A == B)) return 0; // 无穷大直接比较
            continue; // 相等则继续检查下一个元素
        }

        // 动态容差比较
        double epsilon = 1e-5;
        double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
        if (fabs(A - B) > tol) return 0; // 违反相等条件
    }

    // 3. 全部满足，返回 1
    return 1;
}
