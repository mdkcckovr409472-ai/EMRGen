int check(TestCase tc) {
    double Os = kurtosis(tc.data, tc.n);
    TestCase Tf_case = change(tc);
    double Of = kurtosis(Tf_case.data, Tf_case.n);
    int is_valid = 0;
    if (fabs(Os - Of) < 1e-9) {
        is_valid = 1;
    }
    if (Tf_case.data != tc.data && Tf_case.data != nullptr) {
        delete[] Tf_case.data;
    }
    return is_valid;
}
