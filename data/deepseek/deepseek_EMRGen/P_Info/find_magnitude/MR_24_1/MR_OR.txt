#include <math.h>
#include <stddef.h>

int z24check(double result1, double result2) {
    // 1. 防御性编程 (针对标量，无需空指针和长度检查)
    // 2. 核心判断逻辑 (标量直接比较)
    // 浮点数比较特殊处理
    double epsilon = 1e-5;
    
    // 提取左右操作数 A 和 B (代数移项)
    // 条件: return.result1 == return.result2
    // 移项后: result1 == result2
    double A = result1;
    double B = result2;
    
    // 防御 NaN
    if (isnan(A) || isnan(B)) return 0;
    
    // 处理无穷大 (Inf)
    if (isinf(A) || isinf(B)) {
        // 对于无穷大，退化为数学上的直接比较
        if (A == B) return 1;
        else return 0;
    }
    
    // 计算动态宽容度
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    // 带容差的相等判定 (若违反则 return 0)
    if (fabs(A - B) > tol) {
        return 0;
    }
    
    // 3. 满足条件，返回 1
    return 1;
}
