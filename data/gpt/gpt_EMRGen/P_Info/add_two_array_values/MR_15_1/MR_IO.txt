TestCase z15Change(TestCase t1) {
    TestCase follow_case = t1;
    // Deep copy of array a
    int* follow_case_a = (int*)malloc(t1.n * sizeof(int));
    memcpy(follow_case_a, t1.a, t1.n * sizeof(int));
    follow_case.a = follow_case_a;
    // Defensive check: need at least one element
    if (follow_case.n == 0) {
        return follow_case;
    }
    // Set a random position to 999 if the element is less than 999
    int idx = rand() % follow_case.n;
    if (follow_case.a[idx] < 999) {
        follow_case.a[idx] = 999;
    }
    return follow_case;
}
