TestCase z4Change(TestCase t1) {
    TestCase follow_case = t1;
    // Memory handling for deep copy of array 'v'
    double* follow_case_v = (double*)malloc(t1.n * sizeof(double));
    memcpy(follow_case_v, t1.v, t1.n * sizeof(double));
    follow_case.v = follow_case_v;

    // Defensive check: ensure there is at least one element
    if (follow_case.n > 0) {
        for (int i = 0; i < follow_case.n; ++i) {
            follow_case.v[i] /= 3.0;
        }
    }
    return follow_case;
}
