TestCase z7Change(TestCase t1) {
    TestCase follow_case = t1;
    // --- Memory handling code ---
    double* follow_case_data1 = (double*)malloc(t1.n1 * sizeof(double));
    memcpy(follow_case_data1, t1.data1, t1.n1 * sizeof(double));
    follow_case.data1 = follow_case_data1;
    // -------------------------

    // Defensive check: ensure there is at least one element
    if (follow_case.n1 <= 0) {
        return follow_case;
    }

    // Increment a random element by 1.0 if it is less than the maximum double value
    int idx = rand() % follow_case.n1;
    if (follow_case.data1[idx] < std::numeric_limits<double>::max()) {
        follow_case.data1[idx] += 1.0;
    }

    return follow_case;
}
