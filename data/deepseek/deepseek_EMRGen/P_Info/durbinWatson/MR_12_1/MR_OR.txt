#include <cmath>
#include <cfloat>

int z12check(double result1, double result2) {
    // 1. 核心判断逻辑 (标量直接比较)
    // 移项处理：5.39729777351867E14 * result1 - 2.06158430208E14 * result2 == 0
    // 等价于：5.39729777351867E14 * result1 == 2.06158430208E14 * result2
    double A = 5.39729777351867E14 * result1;
    double B = 2.06158430208E14 * result2;

    // 防御 NaN
    if (std::isnan(A) || std::isnan(B)) return 0;

    // 处理无穷大情况
    if (std::isinf(A) || std::isinf(B)) {
        if (A == B) return 1;
        else return 0;
    }

    // 定义基础容差并计算动态容差
    double epsilon = 1e-5;
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

    // 带容差的相等判断
    if (fabs(A - B) > tol) {
        return 0;
    }

    // 2. 满足条件，返回 1
    return 1;
}
