TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tf.n1 > 0) {
        double* new_data1 = new double[tf.n1];
        new_data1[0] = tc.data1[tc.n1 - 1];
        for (int i = 1; i < tf.n1; i++) {
            new_data1[i] = tc.data1[i - 1];
        }
        tf.data1 = new_data1;
    }
    return tf;
}
