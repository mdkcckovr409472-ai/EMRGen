#include <cmath>
#include <cfloat>

int z17check(double* result1, double* result2, int n) {
    // 1. 防御性编程 (空指针与长度检查)
    if (result1 == NULL || result2 == NULL) return 0;
    if (n <= 0) return 1;

    // 2. 字典序比较逻辑
    const double epsilon = 1e-5;
    
    for (int i = 0; i < n; i++) {
        double A = result1[i];
        double B = result2[i];
        
        // 防御 NaN
        if (std::isnan(A) || std::isnan(B)) return 0;
        
        // 处理无穷大
        if (std::isinf(A) || std::isinf(B)) {
            if (A < B) {
                // 在第一个不相等的元素处，result1[i] < result2[i]，字典序成立
                return 1;
            } else if (A > B) {
                // 在第一个不相等的元素处，result1[i] > result2[i]，字典序不成立
                return 0;
            }
            // 相等则继续循环
            continue;
        }
        
        // 计算动态容差
        double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
        
        // 检查第一个不相等的元素对
        if (fabs(A - B) > tol) {
            // 发现不相等的元素
            if (A < B - tol) {
                // result1[i] < result2[i]，字典序成立
                return 1;
            } else {
                // result1[i] > result2[i]，字典序不成立
                return 0;
            }
        }
        // 相等则继续循环
    }
    
    // 3. 循环结束，所有元素在容差内相等
    // 对于严格小于 (<) 的字典序，完全相等不满足条件
    return 0;
}
