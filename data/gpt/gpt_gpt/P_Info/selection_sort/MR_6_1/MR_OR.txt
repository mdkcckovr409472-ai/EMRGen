int check(TestCase tc) {
    // Get source output Os
    int* Os = selection_sort(tc.list, tc.n);

    // Construct follow-up test case Tf
    TestCase Tf_case = change(tc);

    // Get follow-up output Of
    int* Of = selection_sort(Tf_case.list, Tf_case.n);

    int is_valid = 1;
    // Check Os == Of, check element-wise equality
    for (int i = 0; i < tc.n; i++) {
        if (Os[i] != Of[i]) {
            is_valid = 0;
            break;
        }
    }

    // Free allocated memories
    free(Os);
    free(Of);
    free(Tf_case.list);

    return is_valid;
}
