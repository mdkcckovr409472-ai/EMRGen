#include <math.h>
#include <stddef.h>

int z60check(double result1, double result2) {
    // 1. 防御性编程 (标量情况无需空指针和长度检查)

    // 2. 核心判断逻辑 (标量直接比较)
    // 条件为 return.result1 == return.result2，移项后为 result1 == result2
    // 提取左右操作数
    double A = result1;
    double B = result2;

    // 3. 浮点数比较的特殊处理
    // 第一步：定义基础容差
    double epsilon = 1e-5;

    // 第二步：防御 NaN
    if (isnan(A) || isnan(B)) return 0;

    // 第三步：处理无穷大
    if (isinf(A) || isinf(B)) {
        // 无穷大时退化为数学上的直接比较
        if (A == B) return 1;
        else return 0;
    }

    // 第四步：计算动态宽容度
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

    // 第五步：带容差的相等判定
    // 移项后的条件是相等 (==)
    if (fabs(A - B) > tol) {
        return 0;
    }

    // 3. 满足条件，返回 1
    return 1;
}
