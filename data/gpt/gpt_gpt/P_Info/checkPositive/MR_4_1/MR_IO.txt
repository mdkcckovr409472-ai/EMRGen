TestCase change(TestCase tc) {
    TestCase tf = tc;
    // Create a new array to store the multiplied elements
    double* new_a = new double[tf.n];
    for (int i = 0; i < tf.n; i++) {
        new_a[i] = tf.a[i] * 0.5;
    }
    // Assign new_a to tf.a to represent the follow-up input
    tf.a = new_a;
    return tf;
}
