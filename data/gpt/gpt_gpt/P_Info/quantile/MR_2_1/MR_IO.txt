TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tc.n > 0 && tc.sortedElements != nullptr) {
        double* newArr = new double[tc.n];
        for (int i = 0; i < tc.n; i++) {
            newArr[i] = tc.sortedElements[i] - 1.0;
        }
        tf.sortedElements = newArr;
    }
    return tf;
}
