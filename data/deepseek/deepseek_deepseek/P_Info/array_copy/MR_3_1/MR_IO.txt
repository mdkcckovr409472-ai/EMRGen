TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tf.a != NULL && tf.n > 0) {
        int* reversed = (int*)malloc(tf.n * sizeof(int));
        if (reversed != NULL) {
            for (int i = 0; i < tf.n; i++) {
                reversed[i] = tf.a[tf.n - 1 - i];
            }
            tf.a = reversed;
        } else {
            tf.a = NULL;
        }
    }
    return tf;
}
