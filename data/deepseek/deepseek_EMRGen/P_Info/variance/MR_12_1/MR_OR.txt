#include <math.h>
#include <stddef.h>

int z12check(double result1, double result2) {
    // 1. 核心判断逻辑 (标量直接比较)
    // 条件: result1 >= result2
    // 移项后: result1 >= result2 (无需移项)

    // 防御 NaN
    if (isnan(result1) || isnan(result2)) return 0;

    // 处理无穷大
    if (isinf(result1) || isinf(result2)) {
        // 对于无穷大，直接使用数学比较
        if (result1 >= result2) return 1;
        else return 0;
    }

    // 定义基础容差
    double epsilon = 1e-5;
    
    // 提取左右操作数
    double A = result1;
    double B = result2;
    
    // 计算动态宽容度
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    // 带容差的比较判定 (>=)
    // 若 A < B - tol，则违反条件
    if (A < B - tol) {
        return 0;
    }
    
    // 2. 满足条件，返回 1
    return 1;
}
