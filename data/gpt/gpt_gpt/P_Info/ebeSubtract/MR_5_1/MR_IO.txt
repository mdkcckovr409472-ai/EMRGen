TestCase change(TestCase tc) {
    TestCase tf = tc; 
    if (tf.b != NULL && tf.n2 > 0) {
        double* new_b = (double*)malloc(tf.n2 * sizeof(double));
        if (new_b != NULL) {
            for (int i = 0; i < tf.n2; i++) {
                new_b[i] = tf.b[i] + 1.0;
            }
            tf.b = new_b;
        }
    }
    return tf;
}
