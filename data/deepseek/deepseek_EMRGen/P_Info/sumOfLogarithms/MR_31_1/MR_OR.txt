#include <math.h>
#include <stddef.h>

int z31check(double result1, double result2) {
    // 1. 防御性编程 (针对浮点数的特殊检查)
    // 检查 NaN
    if (isnan(result1) || isnan(result2)) return 0;
    
    // 2. 核心判断逻辑 (标量直接比较)
    // 条件为 result1 == result2
    // 移项后: A = result1, B = result2, 比较符为 ==
    double A = result1;
    double B = result2;
    
    // 处理无穷大情况
    if (isinf(A) || isinf(B)) {
        // 无穷大时退化为数学直接比较
        if (A == B) return 1;
        else return 0;
    }
    
    // 计算动态容差
    double epsilon = 1e-5;
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    // 带容差的相等比较
    if (fabs(A - B) > tol) {
        return 0;
    }
    
    // 3. 满足条件，返回 1
    return 1;
}
