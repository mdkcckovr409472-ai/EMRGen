TestCase change(TestCase tc) {
    // Reuse the original TestCase struct, performing circular right shift on the array a
    int* shifted = (int*)malloc(tc.n * sizeof(int));
    if (tc.n > 0) {
        shifted[0] = tc.a[tc.n - 1];
        for (int i = 1; i < tc.n; i++) {
            shifted[i] = tc.a[i - 1];
        }
    }
    TestCase tf = tc;
    tf.a = shifted;
    return tf;
}
