#include <math.h>
#include <stddef.h>

int z15check(double result1, double result2) {
    // 1. 防御 NaN
    if (isnan(result1) || isnan(result2)) return 0;

    // 2. 处理无穷大情况（动态容差对Inf无效）
    if (isinf(result1) || isinf(result2)) {
        // 直接使用数学上的严格小于比较
        if (result1 < result2) return 1;
        else return 0;
    }

    // 3. 定义基础容差
    double epsilon = 1e-5;
    
    // 4. 代数移项：条件 "result1 < result2" 已是最简形式
    //    A = result1, B = result2
    double A = result1;
    double B = result2;
    
    // 5. 计算动态宽容度
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    // 6. 带容差的严格小于判断
    //    对于 A < B：若 A >= B - tol，则违反条件
    if (A >= B - tol) return 0;
    
    // 7. 满足条件
    return 1;
}
