TestCase change(TestCase tc) {
    TestCase tf = tc;
    tf.x = new double[tc.n];
    for (int i = 0; i < tc.n; i++) {
        tf.x[i] = tc.x[i];
    }
    for (int i = 0; i < tf.n - 1; i++) {
        int j = i + rand() % (tf.n - i);
        double temp = tf.x[i];
        tf.x[i] = tf.x[j];
        tf.x[j] = temp;
    }
    return tf;
}
