TestCase z25Change(TestCase t1) {
    TestCase follow_case = t1;
    // Memory handling: deep copy of the array
    double* follow_case_elements = (double*)malloc(t1.n * sizeof(double));
    memcpy(follow_case_elements, t1.elements, t1.n * sizeof(double));
    follow_case.elements = follow_case_elements;
    // Defensive check: need at least 2 elements to reverse a subarray
    if (follow_case.n < 2) {
        return follow_case;
    }
    // Reverse subarray between two random indices
    int idx1 = rand() % follow_case.n;
    int idx2 = rand() % follow_case.n;
    if (idx1 > idx2) {
        int tmp = idx1;
        idx1 = idx2;
        idx2 = tmp;
    }
    while (idx1 < idx2) {
        double tmp = follow_case.elements[idx1];
        follow_case.elements[idx1] = follow_case.elements[idx2];
        follow_case.elements[idx2] = tmp;
        idx1++;
        idx2--;
    }
    return follow_case;
}
