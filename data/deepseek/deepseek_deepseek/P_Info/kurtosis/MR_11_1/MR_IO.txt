TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tf.n > 1 && tf.data != nullptr) {
        int p = 1; // 循环左移1个位置
        double* new_data = new double[tf.n];
        for (int i = 0; i < tf.n; i++) {
            new_data[i] = tf.data[(i + p) % tf.n];
        }
        tf.data = new_data;
    }
    return tf;
}
