TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tc.n > 0) {
        int* shifted = (int*)malloc(tc.n * sizeof(int));
        shifted[0] = tc.a[tc.n - 1];
        for (int i = 1; i < tc.n; i++) {
            shifted[i] = tc.a[i - 1];
        }
        tf.a = shifted;
    }
    return tf;
}
