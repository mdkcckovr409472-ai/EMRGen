TestCase z22Change(TestCase t1) {
    TestCase follow_case = t1;
    // Memory handling
    int* follow_case_a = (int*)malloc(t1.n * sizeof(int));
    memcpy(follow_case_a, t1.a, t1.n * sizeof(int));
    follow_case.a = follow_case_a;
    // Defensive check
    if (follow_case.n == 0) {
        return follow_case;
    }
    // Sort array ascending
    for (int i = 0; i < follow_case.n - 1; ++i) {
        int min_idx = i;
        for (int j = i + 1; j < follow_case.n; ++j) {
            if (follow_case.a[j] < follow_case.a[min_idx]) {
                min_idx = j;
            }
        }
        if (min_idx != i) {
            int temp = follow_case.a[i];
            follow_case.a[i] = follow_case.a[min_idx];
            follow_case.a[min_idx] = temp;
        }
    }
    return follow_case;
}
