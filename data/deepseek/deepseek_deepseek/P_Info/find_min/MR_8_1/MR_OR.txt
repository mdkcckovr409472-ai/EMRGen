int check(TestCase tc) {
    if (tc.a == nullptr || tc.n <= 0) {
        return 0;
    }
    int Os = find_min(tc.a, tc.n);
    TestCase Tf_case = change(tc);
    int Of = find_min(Tf_case.a, Tf_case.n);
    int is_valid = 0;
    if (Os == Of) {
        is_valid = 1;
    }
    if (Tf_case.a != tc.a && Tf_case.a != nullptr) {
        delete[] Tf_case.a;
    }
    return is_valid;
}
