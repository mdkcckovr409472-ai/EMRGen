TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tf.n > 0) {
        double* reversed_elements = new double[tf.n];
        for (int i = 0; i < tf.n; ++i) {
            reversed_elements[i] = tc.elements[tf.n - 1 - i];
        }
        tf.elements = reversed_elements;
    }
    return tf;
}
