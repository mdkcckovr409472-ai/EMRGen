#include <math.h>
#include <stddef.h>

int z40check(double* result1, double* result2, int n) {
    // 1. 防御性编程 (空指针与长度检查)
    if (result1 == NULL || result2 == NULL) return 0;
    if (n <= 0) return 1;

    // 2. 核心判断逻辑 (elementwise)
    for (int i = 0; i < n; i++) {
        // 提取移项后的操作数 A 和 B
        double A = result1[i];
        double B = result2[i];

        // 防御 NaN
        if (isnan(A) || isnan(B)) return 0;

        // 处理无穷大情况
        if (isinf(A) || isinf(B)) {
            // 移项后的条件为 A <= B
            if (!(A <= B)) return 0;
            continue; // 此对元素检查通过，继续下一对
        }

        // 定义基础容差并计算动态宽容度
        double epsilon = 1e-5;
        double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

        // 根据移项后的条件 A <= B 进行带容差的判定
        // 违反条件为 A > B + tol
        if (A > B + tol) {
            return 0;
        }
    }

    // 3. 全部满足，返回 1
    return 1;
}
