int check(TestCase tc) {
    if (tc.a == NULL || tc.b == NULL || tc.n1 != tc.n2 || tc.n1 <= 0) {
        return 0;
    }

    double* Os = ebeSubtract(tc.a, tc.b, tc.n1, tc.n2);
    if (Os == NULL) {
        return 0;
    }

    TestCase Tf_case = change(tc);
    double* Of = NULL;
    if (Tf_case.a != NULL) {
        Of = ebeSubtract(Tf_case.a, Tf_case.b, Tf_case.n1, Tf_case.n2);
    }
    // Free the dynamically allocated follow-up input array if changed
    if (Tf_case.a != tc.a) {
        free(Tf_case.a);
    }

    if (Of == NULL) {
        free(Os);
        return 0;
    }

    int is_valid = 1;
    for (int i = 0; i < tc.n1; ++i) {
        // Check the metamorphic relation: Of[i] == Os[i] + 1
        if (Of[i] != Os[i] + 1.0) {
            is_valid = 0;
            break;
        }
    }

    free(Os);
    free(Of);
    return is_valid;
}
