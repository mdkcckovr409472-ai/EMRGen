TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tc.n > 1) {
        int* new_a = new int[tc.n];
        for (int i = 0; i < tc.n - 1; i++) {
            new_a[i] = tc.a[i + 1];
        }
        new_a[tc.n - 1] = tc.a[0];
        tf.a = new_a;
    } else {
        // For n <= 1, array remains the same, no allocation needed
        tf.a = tc.a;
    }
    return tf;
}
