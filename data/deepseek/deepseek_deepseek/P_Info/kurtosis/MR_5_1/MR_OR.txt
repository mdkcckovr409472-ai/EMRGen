int check(TestCase tc) {
    if (tc.n <= 0 || tc.data == nullptr) {
        return 0;
    }
    double* original_data_copy = new double[tc.n];
    std::memcpy(original_data_copy, tc.data, tc.n * sizeof(double));
    double Os = kurtosis(original_data_copy, tc.n);
    TestCase Tf_case = change(tc);
    double Of = kurtosis(Tf_case.data, Tf_case.n);
    int is_valid = 0;
    if (std::abs(Os - Of) < 1e-9) {
        is_valid = 1;
    }
    delete[] original_data_copy;
    return is_valid;
}
