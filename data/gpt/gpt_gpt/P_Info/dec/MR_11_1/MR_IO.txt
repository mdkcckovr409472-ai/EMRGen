TestCase change(TestCase tc) {
    TestCase tf = tc;
    // To avoid modifying original array2, allocate new sorted array2
    int* sorted_array2 = (int*)malloc(tc.n * sizeof(int));
    memcpy(sorted_array2, tc.array2, tc.n * sizeof(int));
    std::sort(sorted_array2, sorted_array2 + tc.n);
    tf.array2 = sorted_array2;
    return tf;
}
