TestCase z6Change(TestCase t1) {
    TestCase follow_case = t1;
    // Deep copy of array a
    int* follow_case_a = (int*)malloc(t1.n * sizeof(int));
    memcpy(follow_case_a, t1.a, t1.n * sizeof(int));
    follow_case.a = follow_case_a;
    // Defensive check: need at least 2 elements to swap
    if (follow_case.n < 2) {
        return follow_case;
    }
    // Swap two random distinct elements
    int idx1 = rand() % follow_case.n;
    int idx2 = rand() % follow_case.n;
    while (idx2 == idx1) {
        idx2 = rand() % follow_case.n;
    }
    int temp = follow_case.a[idx1];
    follow_case.a[idx1] = follow_case.a[idx2];
    follow_case.a[idx2] = temp;
    return follow_case;
}
