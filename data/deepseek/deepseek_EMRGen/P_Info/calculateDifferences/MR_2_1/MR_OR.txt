#include <cmath>
#include <cfloat>

int z2check(double* result1, double* result2, int n) {
    // 1. 防御性编程 (空指针与长度检查)
    if (result1 == NULL || result2 == NULL) return 0;
    if (n <= 0) return 1;

    // 2. 核心判断逻辑 (elementwise)
    for (int i = 0; i < n; i++) {
        // 根据条件 2 * result1[i] - 2 * result2[i] + 5 == 0 进行移项
        // 移项后: 2 * result1[i] + 5 == 2 * result2[i]
        double A = 2.0 * result1[i] + 5.0;
        double B = 2.0 * result2[i];

        // 防御 NaN
        if (std::isnan(A) || std::isnan(B)) return 0;

        // 处理无穷大情况
        if (std::isinf(A) || std::isinf(B)) {
            if (A != B) return 0; // 移项后条件为相等，若不等则违反
            // 如果相等，则继续检查下一个元素
            continue;
        }

        // 动态容差比较
        double epsilon = 1e-5;
        double tol = epsilon * std::max(1.0, std::max(std::fabs(A), std::fabs(B)));

        // 移项后条件为 A == B，使用容差判断
        if (std::fabs(A - B) > tol) {
            return 0; // 一旦违反条件，立即返回 0
        }
    }

    // 3. 全部满足，返回 1
    return 1;
}
