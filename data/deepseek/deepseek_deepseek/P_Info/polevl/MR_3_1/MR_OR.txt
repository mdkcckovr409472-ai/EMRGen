int check(TestCase tc) {
    double Os = polevl(tc.x, tc.coef, tc.N);
    TestCase Tf_case = change(tc);
    double Of = polevl(Tf_case.x, Tf_case.coef, Tf_case.N);
    int is_valid = 0;
    if (tc.x * Os == Of) {
        is_valid = 1;
    }
    if (Tf_case.coef != tc.coef) {
        delete[] Tf_case.coef;
    }
    return is_valid;
}
