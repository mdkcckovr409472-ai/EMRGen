#include <math.h>
#include <stddef.h>

int z14check(double result1, double result2) {
    // 1. 核心判断逻辑 (标量直接比较)
    // 条件: return.result1 == return.result2
    // 移项后: result1 == result2
    
    // 防御 NaN
    if (isnan(result1) || isnan(result2)) return 0;
    
    // 处理无穷大情况
    if (isinf(result1) || isinf(result2)) {
        if (result1 == result2) return 1;
        else return 0;
    }
    
    // 定义基础容差
    double epsilon = 1e-5;
    
    // 提取移项后的操作数
    double A = result1;
    double B = result2;
    
    // 计算动态宽容度
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    // 带容差的相等比较
    if (fabs(A - B) > tol) {
        return 0;
    }
    
    // 2. 满足条件，返回 1
    return 1;
}
