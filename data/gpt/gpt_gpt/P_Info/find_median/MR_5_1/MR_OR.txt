int check(TestCase tc) {
    // Source output Os
    double Os = find_median(tc.a, tc.n);

    // Follow-up input and output Of
    TestCase Tf_case = change(tc);
    double Of = find_median(Tf_case.a, Tf_case.n);

    // MR verification
    int is_valid = 0;
    if (Os == Of) {
        is_valid = 1;
    }

    // Free dynamically allocated memory in Tf_case
    delete[] Tf_case.a;

    return is_valid;
}
