TestCase change(TestCase tc) {
    // Reuse original TestCase struct, only overwrite values pointer to new sorted array
    TestCase tf = tc;
    int* sorted_values = new int[tc.n];
    for (int i = 0; i < tc.n; i++) {
        sorted_values[i] = tc.values[i];
    }
    // Sort in descending order
    std::sort(sorted_values, sorted_values + tc.n, std::greater<int>());
    tf.values = sorted_values;
    return tf;
}
