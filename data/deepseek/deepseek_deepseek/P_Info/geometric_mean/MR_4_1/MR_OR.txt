int check(TestCase tc) {
    double Os = geometric_mean(tc.a, tc.n);
    TestCase Tf_case = change(tc);
    double Of = geometric_mean(Tf_case.a, Tf_case.n);
    int is_valid = 0;
    if (fabs(Os - Of) < 1e-9) {
        is_valid = 1;
    }
    if (Tf_case.a != tc.a && Tf_case.a != nullptr) {
        delete[] Tf_case.a;
    }
    return is_valid;
}
