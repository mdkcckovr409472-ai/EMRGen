#include <cmath>
#include <cstddef>

int z1check(double result1, double result2) {
    // 1. 核心判断逻辑 (标量直接比较)
    // 条件: return.result1 - return.result2 + 1 == 0
    // 移项: result1 - result2 + 1 == 0  ->  result1 + 1 == result2

    // 提取移项后的左右操作数，恢复真实量级
    double A = result1 + 1.0;
    double B = result2;

    // 防御 NaN
    if (std::isnan(A) || std::isnan(B)) return 0;

    // 处理无穷大 (动态容差对 Inf 无效)
    if (std::isinf(A) || std::isinf(B)) {
        // 根据移项后的等式 (==) 直接比较
        if (A == B) return 1;
        else return 0;
    }

    // 定义基础容差
    const double epsilon = 1e-5;
    // 计算动态宽容度
    double tol = epsilon * std::fmax(1.0, std::fmax(std::fabs(A), std::fabs(B)));

    // 带容差的相等判定 (==)
    // 若违反条件 (差值大于容差)，返回 0
    if (std::fabs(A - B) > tol) {
        return 0;
    }

    // 2. 满足条件，返回 1
    return 1;
}
