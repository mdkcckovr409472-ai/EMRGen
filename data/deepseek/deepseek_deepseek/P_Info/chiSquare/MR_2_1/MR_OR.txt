int check(TestCase tc) {
    double Os = chiSquate(tc.expected, tc.observed, tc.n);
    TestCase Tf_case = change(tc);
    double Of = chiSquate(Tf_case.expected, Tf_case.observed, Tf_case.n);
    int is_valid = 0;
    if (abs(Os - Of) < 10E-6) {
        is_valid = 1;
    }
    delete[] Tf_case.expected;
    return is_valid;
}
