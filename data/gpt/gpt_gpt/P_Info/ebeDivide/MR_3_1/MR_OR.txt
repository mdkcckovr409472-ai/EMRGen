int check(TestCase tc) {
    double* Os = ebeDivide(tc.a, tc.b, tc.n1, tc.n2);
    if (Os == NULL) {
        return 0;
    }

    TestCase Tf_case = change(tc);
    double* Of = NULL;
    if (Tf_case.a != tc.a) {
        Of = ebeDivide(Tf_case.a, Tf_case.b, Tf_case.n1, Tf_case.n2);
        if (Of == NULL) {
            free(Os);
            free(Tf_case.a);
            return 0;
        }
    } else {
        // If change() failed malloc and didn't change a pointer:
        Of = NULL;
        free(Os);
        return 0;
    }

    int is_valid = 1;
    for (int i = 0; i < tc.n1; ++i) {
        if (tc.b[i] == 0.0) {
            if (Of[i] != INFINITY) {
                is_valid = 0;
                break;
            }
        } else {
            if (Of[i] != 2.0 * Os[i]) {
                is_valid = 0;
                break;
            }
        }
    }

    free(Os);
    free(Of);
    free(Tf_case.a);
    return is_valid;
}
