int check(TestCase tc) {
    int* Os = clip(tc.a, tc.lowerLim, tc.upperLim, tc.n);
    TestCase Tf_case = change(tc);
    int* Of = clip(Tf_case.a, Tf_case.lowerLim, Tf_case.upperLim, Tf_case.n);
    int is_valid = 1;
    for (int i = 0; i < tc.n; i++) {
        if (Os[i] > Of[i]) {
            is_valid = 0;
            break;
        }
    }
    if (Tf_case.a != tc.a) {
        delete[] Tf_case.a;
    }
    return is_valid;
}
