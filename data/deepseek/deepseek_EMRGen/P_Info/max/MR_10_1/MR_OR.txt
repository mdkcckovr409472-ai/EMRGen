#include <cmath>
#include <cfloat>

int z10check(double result1, double result2) {
    // 防御性编程：检查NaN
    if (std::isnan(result1) || std::isnan(result2)) return 0;
    
    // 处理无穷大情况（动态容差对Inf无效）
    if (std::isinf(result1) || std::isinf(result2)) {
        // 直接数学比较：result1 >= result2
        if (result1 >= result2) return 1;
        else return 0;
    }
    
    // 浮点数比较：result1 >= result2
    // 移项后：A = result1, B = result2, 条件为 A >= B
    double A = result1;
    double B = result2;
    double epsilon = 1e-5;
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    // 检查是否违反 A >= B
    if (A < B - tol) return 0;
    
    // 满足条件
    return 1;
}
