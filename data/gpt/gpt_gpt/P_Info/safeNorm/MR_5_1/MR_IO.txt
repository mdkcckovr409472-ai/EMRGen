TestCase change(TestCase tc) {
    TestCase tf = tc;
    double* sorted_v = new double[tf.n];
    for (int i = 0; i < tf.n; ++i) {
        sorted_v[i] = tf.v[i];
    }
    for (int i = 0; i < tf.n - 1; ++i) {
        for (int j = 0; j < tf.n - i - 1; ++j) {
            if (sorted_v[j] < sorted_v[j + 1]) {
                double tmp = sorted_v[j];
                sorted_v[j] = sorted_v[j + 1];
                sorted_v[j + 1] = tmp;
            }
        }
    }
    tf.v = sorted_v;
    return tf;
}
