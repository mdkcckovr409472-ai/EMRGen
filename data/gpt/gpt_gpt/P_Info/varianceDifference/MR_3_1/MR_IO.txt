TestCase change(TestCase tc) {
    TestCase tf = tc;
    // Reuse the original sample1 array by shallow copy, but create a copy for safe modification
    double* new_sample1 = new double[tf.n];
    for (int i = 0; i < tf.n; i++) {
        new_sample1[i] = tf.sample1[i];
    }
    new_sample1[0] = DBL_MAX;
    tf.sample1 = new_sample1;
    return tf;
}
