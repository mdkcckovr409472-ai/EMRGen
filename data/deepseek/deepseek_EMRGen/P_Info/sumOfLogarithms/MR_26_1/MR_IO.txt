TestCase z26Change(TestCase t1) {
    TestCase follow_case = t1;
    // --- 这是根据规则4生成的内存处理代码 ---
    double* follow_case_elements = (double*)malloc(t1.n * sizeof(double));
    memcpy(follow_case_elements, t1.elements, t1.n * sizeof(double));
    follow_case.elements = follow_case_elements;
    // ------------------------------------

    // 这是根据规则6a生成的防御性代码，防止n<=0时出错
    if (follow_case.n <= 0) {
        // 数组为空或长度为负，无法进行操作，直接返回副本
        return follow_case;
    }

    // 这是根据规则5生成的具体操作代码
    // 1. 寻找数组中的最大值
    double max_val = follow_case.elements[0];
    for (int i = 1; i < follow_case.n; i++) {
        if (follow_case.elements[i] > max_val) {
            max_val = follow_case.elements[i];
        }
    }
    
    // 2. 生成一个随机位置索引
    // 使用简单的伪随机方法（基于当前时间微秒部分），实际应用中应使用更健壮的随机数生成器
    #include <ctime>
    srand(static_cast<unsigned int>(time(nullptr) * 1000));
    int random_index = rand() % follow_case.n;
    
    // 3. 将随机位置的元素置为最大值
    follow_case.elements[random_index] = max_val;

    return follow_case;
}
