#include <cmath>
#include <cfloat>

int z31check(double result1, double result2) {
    // 1. 防御性编程 (标量情况无需指针和长度检查)
    
    // 2. 核心判断逻辑 (标量直接比较)
    // 条件: result1 >= result2
    
    // 浮点数特殊处理
    // 提取移项后的操作数: A = result1, B = result2 (条件已为 result1 >= result2)
    double A = result1;
    double B = result2;
    
    // 检查NaN
    if (std::isnan(A) || std::isnan(B)) return 0;
    
    // 处理无穷大情况
    if (std::isinf(A) || std::isinf(B)) {
        // 无穷大时直接数学比较
        if (A >= B) return 1;
        else return 0;
    }
    
    // 定义基础容差
    double epsilon = 1e-5;
    
    // 计算动态宽容度
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    // 带容差的比较判定 (条件: A >= B)
    // 违反条件: A < B - tol
    if (A < B - tol) {
        return 0;
    }
    
    // 3. 满足条件，返回 1
    return 1;
}
