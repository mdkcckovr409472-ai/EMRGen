int z48check(double result1, double result2) {
    // Core evaluation logic (direct scalar comparison) with floating-point tolerance
    double epsilon = 1e-5;

    double A = result1;
    double B = result2;

    // Defend against NaN
    if (std::isnan(A) || std::isnan(B)) return 0;

    // Defend against Inf
    if (std::isinf(A) || std::isinf(B)) {
        if (A <= B) return 1;
        else return 0;
    }

    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

    // For operator <= : if A > B + tol, violation
    if (A > B + tol) return 0;

    // Satisfied
    return 1;
}
