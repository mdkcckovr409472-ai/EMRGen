int z6check(int* result1, int* result2, int n) {
    // 1. Defensive programming (Length and Null pointer checks)
    if (n <= 0) return 1;
    if (result1 == NULL || result2 == NULL) return 0;

    // 2. Core evaluation logic (elementwise): check result1[i] <= result2[i]
    for (int i = 0; i < n; i++) {
        if (!(result1[i] <= result2[i])) {
            return 0; // violation found
        }
    }
    // 3. Fully satisfied
    return 1;
}
