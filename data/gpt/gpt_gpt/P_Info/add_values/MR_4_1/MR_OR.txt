int check(TestCase tc) {
    int Os = add_values(tc.a, tc.n);
    TestCase Tf_case = change(tc);
    int Of = add_values(Tf_case.a, Tf_case.n);
    int is_valid = (Os == Of) ? 1 : 0;
    // Free dynamically allocated memory in Tf_case
    if (Tf_case.a != tc.a) {
        delete[] Tf_case.a;
    }
    return is_valid;
}
