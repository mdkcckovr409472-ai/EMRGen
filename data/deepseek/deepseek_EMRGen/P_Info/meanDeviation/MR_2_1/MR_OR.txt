#include <math.h>
#include <stddef.h>

int z2check(double result1, double result2) {
    // 1. 核心判断逻辑 (标量直接比较)
    // 浮点数比较特殊处理
    // 提取移项后的操作数 A 和 B
    double A = result1;
    double B = result2;

    // 防御 NaN
    if (isnan(A) || isnan(B)) return 0;

    // 处理无穷大
    if (isinf(A) || isinf(B)) {
        // 根据移项后的条件 result1 <= result2 直接比较
        if (A <= B) return 1;
        else return 0;
    }

    // 定义基础容差
    double epsilon = 1e-5;
    // 计算动态宽容度
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

    // 根据移项后的条件 result1 <= result2 进行带容差的比较
    // 条件：A <= B
    if (A > B + tol) {
        return 0;
    }

    // 2. 满足条件，返回 1
    return 1;
}
