TestCase z13Change(TestCase t1) {
    TestCase follow_case = t1;
    // Memory handling: deep copy of the array
    int* follow_case_values = (int*)malloc(t1.n * sizeof(int));
    memcpy(follow_case_values, t1.values, t1.n * sizeof(int));
    follow_case.values = follow_case_values;
    // Defensive check: ensure there is at least one element
    if (follow_case.n == 0) {
        return follow_case;
    }
    // Rotate right by 1 (move last element to front)
    int last = follow_case.values[follow_case.n - 1];
    for (int i = follow_case.n - 1; i > 0; --i) {
        follow_case.values[i] = follow_case.values[i - 1];
    }
    follow_case.values[0] = last;
    return follow_case;
}
