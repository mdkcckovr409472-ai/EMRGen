#include <cmath>
#include <cfloat>

int z1check(double result1, double result2) {
    // 浮点数比较需要特殊处理
    double epsilon = 1e-5;
    
    // 第一步：代数移项，恢复数据量级
    // 原条件: result1 - result2 + 1 == 0
    // 移项后: result1 + 1 == result2
    double A = result1 + 1.0;
    double B = result2;
    
    // 第二步：防御NaN
    if (std::isnan(A) || std::isnan(B)) return 0;
    
    // 第三步：处理无穷大（动态容差对Inf无效）
    if (std::isinf(A) || std::isinf(B)) {
        // 直接数学比较
        if (A == B) return 1;
        else return 0;
    }
    
    // 第四步：计算动态容差
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    // 第五步：带容差的相等比较
    if (fabs(A - B) > tol) return 0;
    
    // 满足条件
    return 1;
}
