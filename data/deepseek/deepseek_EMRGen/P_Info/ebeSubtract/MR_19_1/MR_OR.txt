#include <cmath>
#include <cfloat>

int z19check(double* result1, double* result2, int n) {
    // 1. 防御性编程 (空指针与长度检查)
    if (result1 == NULL || result2 == NULL) return 0;
    if (n <= 0) return 1;

    // 2. 字典序比较逻辑 (lexically)
    for (int i = 0; i < n; i++) {
        double A = result1[i];
        double B = result2[i];

        // 防御 NaN
        if (std::isnan(A) || std::isnan(B)) return 0;

        // 处理无穷大
        if (std::isinf(A) || std::isinf(B)) {
            if (A < B) {
                // 第一个不相等的元素满足 result1[i] < result2[i]，字典序成立，立即返回1
                return 1;
            } else if (A > B) {
                // 第一个不相等的元素不满足 result1[i] < result2[i]，字典序不成立，立即返回0
                return 0;
            }
            // 如果相等，继续检查下一个元素
            continue;
        }

        // 浮点数动态容差比较
        double epsilon = 1e-5;
        double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

        // 条件：A < B (移项后 result1[i] < result2[i])
        if (A >= B - tol) {
            // 不满足严格小于，需要进一步判断
            if (A > B + tol) {
                // 明确大于，字典序不成立
                return 0;
            }
            // 在容差范围内视为相等，继续检查下一个元素
            continue;
        } else {
            // 明确小于，字典序成立
            return 1;
        }
    }

    // 3. 循环结束，所有元素在容差内都相等，严格小于条件不满足
    return 0;
}
