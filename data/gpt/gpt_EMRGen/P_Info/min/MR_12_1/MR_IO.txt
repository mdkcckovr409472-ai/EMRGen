TestCase z12Change(TestCase t1) {
    TestCase follow_case = t1;
    // Memory handling: deep copy of the array
    double* follow_case_elements = (double*)malloc(t1.n * sizeof(double));
    memcpy(follow_case_elements, t1.elements, t1.n * sizeof(double));
    follow_case.elements = follow_case_elements;
    // Defensive check: need at least 2 elements to perform a meaningful sort
    if (follow_case.n < 2) {
        return follow_case;
    }
    // Sort array ascending (simple selection sort)
    for (int i = 0; i < follow_case.n - 1; ++i) {
        for (int j = i + 1; j < follow_case.n; ++j) {
            if (follow_case.elements[i] > follow_case.elements[j]) {
                double temp = follow_case.elements[i];
                follow_case.elements[i] = follow_case.elements[j];
                follow_case.elements[j] = temp;
            }
        }
    }
    return follow_case;
}
