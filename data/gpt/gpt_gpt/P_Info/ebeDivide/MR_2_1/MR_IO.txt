TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tf.n1 > 0) {
        double* new_a = (double*)malloc(tf.n1 * sizeof(double));
        if (new_a == NULL) {
            tf.a = NULL; 
            return tf;
        }
        for (int i = 0; i < tf.n1; i++) {
            new_a[i] = tf.a[i] - 1.0;
        }
        tf.a = new_a;
    }
    return tf;
}
