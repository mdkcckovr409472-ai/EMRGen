TestCase change(TestCase tc) {
    TestCase tf = tc;
    // Allocate new array for the follow-up input to avoid modifying original data
    tf.a = new int[tc.n];
    for (int i = 0; i < tc.n; i++) {
        // Assume no element is zero or odd as per MR description
        tf.a[i] = tc.a[i] / 2;
    }
    return tf;
}
