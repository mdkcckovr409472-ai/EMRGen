#include <math.h>
#include <stddef.h>

int z52check(double* result1, double* result2, int n) {
    // 1. 防御性编程 (空指针与长度检查)
    if (result1 == NULL || result2 == NULL) return 0;
    if (n <= 0) return 1;

    // 2. 核心判断逻辑 (字典序比较)
    for (int i = 0; i < n; i++) {
        double A = result1[i];
        double B = result2[i];

        // 浮点数特殊处理
        // 防御 NaN
        if (isnan(A) || isnan(B)) return 0;

        // 处理无穷大
        if (isinf(A) || isinf(B)) {
            if (A < B) {
                // 在第一个不相等的元素处，result1[i] < result2[i]，字典序满足 <=，立即返回成功
                return 1;
            } else if (A > B) {
                // 在第一个不相等的元素处，result1[i] > result2[i]，字典序不满足 <=，立即返回失败
                return 0;
            }
            // A == B (包括 +Inf == +Inf, -Inf == -Inf)，继续循环
            continue;
        }

        // 动态容差比较（用于判断是否相等）
        double epsilon = 1e-5;
        double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

        if (fabs(A - B) > tol) {
            // 找到第一个不相等的元素对
            if (A < B) {
                // result1[i] < result2[i]，字典序满足 <=，立即返回成功
                return 1;
            } else {
                // result1[i] > result2[i]，字典序不满足 <=，立即返回失败
                return 0;
            }
        }
        // 当前元素在容差范围内视为相等，继续检查下一个元素
    }

    // 3. 循环结束，所有元素在容差范围内都相等，满足 <= 条件
    return 1;
}
