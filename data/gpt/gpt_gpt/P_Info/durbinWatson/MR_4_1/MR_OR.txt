int check(TestCase tc) {
    double Os = durbinWatson(tc.elements, tc.n);
    TestCase Tf_case = change(tc);
    double Of = durbinWatson(Tf_case.elements, Tf_case.n);

    int is_valid = 0;
    // Check if 0.25 * Os == Of within a small tolerance due to floating point arithmetic
    const double epsilon = 1e-9;
    if (fabs(0.25 * Os - Of) < epsilon) {
        is_valid = 1;
    }

    delete[] Tf_case.elements;
    return is_valid;
}
