int check(TestCase tc) {
    double Os = sampleKurtosis(tc.moment4, tc.sampleVariance, tc.size);
    TestCase Tf_case = change(tc);
    double Of = sampleKurtosis(Tf_case.moment4, Tf_case.sampleVariance, Tf_case.size);
    int is_valid = 0;
    if (Of > Os) {
        is_valid = 1;
    }
    return is_valid;
}
