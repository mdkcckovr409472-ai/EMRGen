#include <cmath>
#include <cfloat>

int z31check(double* result1, double* result2, int n) {
    // 1. 防御性编程 (空指针与长度检查)
    if (result1 == NULL || result2 == NULL) return 0;
    if (n <= 0) return 1;

    // 2. 核心判断逻辑 (elementwise)
    for (int i = 0; i < n; i++) {
        // 条件: 10 * return.result1[] - return.result2[] == 0
        // 移项后: 10 * result1[i] == result2[i]
        double A = 10.0 * result1[i];
        double B = result2[i];

        // 防御 NaN
        if (std::isnan(A) || std::isnan(B)) return 0;

        // 处理无穷大
        if (std::isinf(A) || std::isinf(B)) {
            if (A != B) return 0; // 移项后是 ==，所以不相等就返回0
            // 如果相等，继续循环（或进入下一步，但下一步容差计算对Inf无效，所以这里直接处理完）
            continue;
        }

        // 动态容差比较
        double epsilon = 1e-5;
        double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
        if (fabs(A - B) > tol) return 0;
    }

    // 3. 全部满足，返回 1
    return 1;
}
