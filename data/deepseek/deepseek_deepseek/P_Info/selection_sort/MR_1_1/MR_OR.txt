int check(TestCase tc) {
    int* Os = selection_sort(tc.list, tc.n);
    TestCase Tf_case = change(tc);
    int* Of = selection_sort(Tf_case.list, Tf_case.n);
    int is_valid = 1;
    int k = 5;
    for (int i = 0; i < tc.n; i++) {
        if (Os[i] + k != Of[i]) {
            is_valid = 0;
            break;
        }
    }
    free(Os);
    free(Of);
    free(Tf_case.list);
    return is_valid;
}
