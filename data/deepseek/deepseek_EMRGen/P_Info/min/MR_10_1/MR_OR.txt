#include <math.h>
#include <stddef.h>

int z10check(double result1, double result2) {
    // 1. 核心判断逻辑 (标量直接比较)
    // 条件: 8.84279719003555E14 * result1 - 2.81474976710656E14 * result2 == 0
    // 移项恢复量级: 8.84279719003555E14 * result1 == 2.81474976710656E14 * result2

    double A = 8.84279719003555E14 * result1;
    double B = 2.81474976710656E14 * result2;

    // 防御 NaN
    if (isnan(A) || isnan(B)) return 0;

    // 处理无穷大 (动态容差对Inf无效)
    if (isinf(A) || isinf(B)) {
        // 退化为精确数学比较
        if (A == B) return 1;
        else return 0;
    }

    // 定义基础容差并计算动态容差
    double epsilon = 1e-5;
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

    // 带容差的相等比较 (移项后为 ==)
    if (fabs(A - B) > tol) {
        return 0;
    }

    // 2. 满足条件，返回 1
    return 1;
}
