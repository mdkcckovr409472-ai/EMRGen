TestCase change(TestCase tc) {
    TestCase tf = tc;

    tf.a = (double*)malloc(tc.n1 * sizeof(double));
    if (tf.a == NULL) {
        // If malloc fails, fallback to original to avoid crash
        tf.a = tc.a;
        return tf;
    }

    for (int i = 0; i < tc.n1; i++) {
        tf.a[i] = tc.a[i] / 2.0;
    }

    return tf;
}
