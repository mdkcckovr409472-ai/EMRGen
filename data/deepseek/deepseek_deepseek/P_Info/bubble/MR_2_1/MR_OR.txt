int check(TestCase tc) {
    int* Os = bubble(tc.a, tc.n);
    TestCase Tf_case = change(tc);
    int* Of = bubble(Tf_case.a, Tf_case.n);
    int is_valid = 1;
    for (int i = 0; i < tc.n; i++) {
        if (Of[i] != -Os[tc.n - 1 - i]) {
            is_valid = 0;
            break;
        }
    }
    free(Os);
    free(Tf_case.a);
    free(Of);
    return is_valid;
}
