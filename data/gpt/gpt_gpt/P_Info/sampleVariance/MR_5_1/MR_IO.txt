TestCase change(TestCase tc) {
    TestCase tf = tc;
    // Allocate new array to avoid modifying original elements
    double* new_elements = new double[tf.n];
    for (int i = 0; i < tf.n; ++i) {
        new_elements[i] = tf.elements[i];
    }
    // Set the last element to minimum double value
    if (tf.n > 0) {
        new_elements[tf.n - 1] = -std::numeric_limits<double>::max();
    }
    tf.elements = new_elements;
    return tf;
}
