TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tf.n > 0) {
        double max_val = tf.observed[0];
        for (int i = 1; i < tf.n; i++) {
            if (tf.observed[i] > max_val) {
                max_val = tf.observed[i];
            }
        }
        tf.observed[0] = max_val;
    }
    return tf;
}
