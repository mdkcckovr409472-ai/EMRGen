int z12check(double* result1, double* result2, int n) {
    if (n <= 0) return 1;
    if (result1 == NULL || result2 == NULL) return 0;

    double epsilon = 1e-5;
    for (int i = 0; i < n; i++) {
        double A = result1[i];
        double B = result2[i];

        if (std::isnan(A) || std::isnan(B)) return 0;
        if (std::isinf(A) || std::isinf(B)) {
            if (A < B) return 1;
            else if (A > B) return 0;
            else return 0; // equal infinities violate strict <
        }

        double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
        double diff = A - B;

        if (fabs(diff) <= tol) {
            // Equal within tolerance, continue to next element
            continue;
        } else if (diff > tol) {
            // A > B, so result1[i] >= result2[i], violates result1[i] < result2[i]
            return 0;
        } else {
            // A < B, result1[i] < result2[i], lex comparison returns satisfied at first unequal
            return 1;
        }
    }
    // All elements equal within tolerance, strict less than returns 0
    return 0;
}
