TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tf.n <= 0) {
        return tf;
    }
    double* original_elements = tf.elements;
    double Os = max(original_elements, tf.n);
    double* new_elements = new double[tf.n];
    for (int i = 0; i < tf.n; ++i) {
        if (original_elements[i] < Os) {
            double new_value = original_elements[i] + (Os - original_elements[i]) / 2.0;
            new_elements[i] = new_value;
        } else {
            new_elements[i] = original_elements[i];
        }
    }
    tf.elements = new_elements;
    return tf;
}
