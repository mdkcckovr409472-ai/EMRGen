#include <math.h>
#include <stddef.h>

int z14check(double result1, double result2) {
    // 1. 防御 NaN
    if (isnan(result1) || isnan(result2)) return 0;
    
    // 2. 处理无穷大 (直接数学比较)
    if (isinf(result1) || isinf(result2)) {
        if (result1 <= result2) return 1;
        else return 0;
    }
    
    // 3. 定义基础容差
    double epsilon = 1e-5;
    
    // 4. 提取移项后的操作数 (result1 <= result2)
    double A = result1;
    double B = result2;
    
    // 5. 计算动态容差
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    // 6. 带容差的 <= 比较 (若违反则返回0)
    if (A > B + tol) return 0;
    
    // 7. 满足条件
    return 1;
}
