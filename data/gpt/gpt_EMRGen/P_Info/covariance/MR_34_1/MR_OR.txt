int z34check(double result1, double result2) {
    // 1. Core evaluation logic (direct scalar comparison)
    double epsilon = 1e-5;
    double A = result1; // Algebraic rearrangement: result1 + result2 == 0 -> result1 == -result2
    double B = -result2;

    if (std::isnan(A) || std::isnan(B)) return 0;
    if (std::isinf(A) || std::isinf(B)) {
        if (fabs(A - B) < 1e-15) return 1;
        else return 0;
    }

    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    if (fabs(A - B) > tol) return 0;

    return 1;
}
