int check(TestCase tc) {
    double Os = entropy(tc.k, tc.n);
    TestCase Tf_case = change(tc);
    double Of = entropy(Tf_case.k, Tf_case.n);
    int is_valid = 0;
    if (fabs(Os - Of) < 1e-12) {
        is_valid = 1;
    }
    if (Tf_case.k != tc.k) {
        delete[] Tf_case.k;
    }
    return is_valid;
}
