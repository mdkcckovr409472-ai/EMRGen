#include <cmath>
#include <cfloat>

int z14check(double* result1, double* result2, int n) {
    // 1. 防御性编程 (空指针与长度检查)
    if (result1 == NULL || result2 == NULL) return 0;
    if (n <= 0) return 1;

    // 2. 核心判断逻辑 (字典序比较)
    for (int i = 0; i < n; i++) {
        double A = result1[i];
        double B = result2[i];
        
        // 防御 NaN
        if (std::isnan(A) || std::isnan(B)) return 0;
        
        // 处理无穷大
        if (std::isinf(A) || std::isinf(B)) {
            if (A < B) {
                // 第一个不相等元素对满足 result1[i] < result2[i]，字典序成立
                return 1;
            } else if (A > B) {
                // 第一个不相等元素对不满足 result1[i] < result2[i]，字典序不成立
                return 0;
            }
            // A == B (包括 +Inf == +Inf, -Inf == -Inf)，继续检查下一个元素
            continue;
        }
        
        // 浮点数比较的特殊处理
        double epsilon = 1e-5;
        double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
        
        // 字典序比较：寻找第一个不相等的元素对
        if (fabs(A - B) > tol) {
            // 找到第一个不相等的元素对
            if (A < B - tol) {
                // result1[i] < result2[i]，字典序成立
                return 1;
            } else {
                // result1[i] > result2[i] 或 A >= B，字典序不成立
                return 0;
            }
        }
        // A ≈ B，继续检查下一个元素
    }
    
    // 3. 所有元素都相等（在容差范围内），严格小于不成立
    return 0;
}
