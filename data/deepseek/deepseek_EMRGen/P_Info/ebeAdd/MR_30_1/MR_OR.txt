#include <cmath>
#include <cfloat>

int z30check(double* result1, double* result2, int n) {
    // 1. 防御性编程 (空指针与长度检查)
    if (result1 == NULL || result2 == NULL) return 0;
    if (n <= 0) return 1;

    // 2. 核心判断逻辑 (elementwise)
    for (int i = 0; i < n; i++) {
        // 提取移项后的操作数：2 * result1[i] == 2 * result2[i] + 3
        double A = 2.0 * result1[i];
        double B = 2.0 * result2[i] + 3.0;

        // 防御 NaN
        if (std::isnan(A) || std::isnan(B)) return 0;

        // 处理无穷大
        if (std::isinf(A) || std::isinf(B)) {
            if (A != B) return 0; // 移项后是 == 比较
            continue; // 相等则继续检查下一个元素
        }

        // 动态容差比较
        double epsilon = 1e-5;
        double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
        if (fabs(A - B) > tol) return 0;
    }

    // 3. 全部满足，返回 1
    return 1;
}
