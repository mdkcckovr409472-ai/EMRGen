TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tc.expected != nullptr && tc.n > 0) {
        double* newExpected = new double[tc.n];
        for (int i = 0; i < tc.n; i++) {
            newExpected[i] = tc.expected[i] - 1.0;
        }
        tf.expected = newExpected;
    }
    return tf;
}
