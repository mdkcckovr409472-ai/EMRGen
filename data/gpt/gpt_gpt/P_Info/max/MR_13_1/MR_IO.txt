TestCase change(TestCase tc) {
    TestCase tf = tc;
    // Allocate new array for follow-up input because we must add 1 to max element without modifying original
    double* new_elements = new double[tc.n];
    for (int i = 0; i < tc.n; ++i) {
        new_elements[i] = tc.elements[i];
    }
    // Find the max element index
    int maxIndex = 0;
    for (int i = 1; i < tc.n; ++i) {
        if (new_elements[i] > new_elements[maxIndex]) {
            maxIndex = i;
        }
    }
    new_elements[maxIndex] += 1.0;
    tf.elements = new_elements;
    return tf;
}
