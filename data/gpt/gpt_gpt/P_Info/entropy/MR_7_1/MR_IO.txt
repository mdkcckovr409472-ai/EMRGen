TestCase change(TestCase tc) {
    TestCase tf = tc;
    double* sorted_k = new double[tf.n];
    for (int i = 0; i < tf.n; i++) {
        sorted_k[i] = tf.k[i];
    }
    // Sort in descending order
    std::sort(sorted_k, sorted_k + tf.n, [](double a, double b) { return a > b; });
    tf.k = sorted_k;
    return tf;
}
