#include <math.h>
#include <stddef.h>

int z28check(double result1, double result2) {
    // 1. 防御性编程 (标量版本无需指针和长度检查)
    // 2. 核心判断逻辑 (标量直接比较)
    // 对于浮点数，必须进行容差比较
    double epsilon = 1e-5;
    
    // 移项：条件为 result1 == result2
    // 直接提取 A = result1, B = result2
    double A = result1;
    double B = result2;
    
    // 检查NaN
    if (isnan(A) || isnan(B)) return 0;
    
    // 处理无穷大情况
    if (isinf(A) || isinf(B)) {
        // 无穷大时直接比较
        if (A == B) return 1;
        else return 0;
    }
    
    // 计算动态容差
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    // 带容差的相等比较
    if (fabs(A - B) > tol) {
        return 0;
    }
    
    // 3. 满足条件，返回 1
    return 1;
}
