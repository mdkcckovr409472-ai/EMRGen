#include <cmath>
#include <cfloat>

int z27check(double result1, double result2) {
    // 1. 防御性编程 (标量版本无需空指针和长度检查)
    
    // 2. 核心判断逻辑 (标量直接比较)
    // 条件: result1 <= result2
    // 移项后: result1 <= result2 (无需移项)
    
    // 第三步: 防御NaN和对Inf的特殊处理
    if (std::isnan(result1) || std::isnan(result2)) return 0;
    
    if (std::isinf(result1) || std::isinf(result2)) {
        // 无穷大时退化为数学直接比较
        if (result1 <= result2) return 1;
        else return 0;
    }
    
    // 第四步: 计算动态宽容度
    double epsilon = 1e-5;
    double A = result1;
    double B = result2;
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    // 第五步: 带容差的比较判定 (<=)
    // 若 A > B + tol，则违反条件
    if (A > B + tol) {
        return 0;
    }
    
    // 3. 满足条件，返回 1
    return 1;
}
