#include <cmath>
#include <cstddef>

int z3check(double result1, double result2) {
    // 防御性编程：检查 NaN
    if (std::isnan(result1) || std::isnan(result2)) return 0;

    // 移项：将条件 2 * result1 - result2 == 0 转化为 2 * result1 == result2
    double A = 2.0 * result1;
    double B = result2;

    // 处理无穷大：动态容差对无穷大无效，需直接比较
    if (std::isinf(A) || std::isinf(B)) {
        if (A == B) return 1;
        else return 0;
    }

    // 定义基础容差并计算动态容差
    double epsilon = 1e-5;
    double tol = epsilon * std::fmax(1.0, std::fmax(std::fabs(A), std::fabs(B)));

    // 带容差的相等性判断
    if (std::fabs(A - B) > tol) return 0;

    // 满足条件
    return 1;
}
