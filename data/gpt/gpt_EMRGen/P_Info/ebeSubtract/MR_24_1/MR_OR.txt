#include <cmath>
int z24check(double* result1, double* result2, int n) {
    // Defensive programming: length check first
    if (n <= 0) return 1;
    // Null pointer check second
    if (result1 == NULL || result2 == NULL) return 0;

    const double epsilon = 1e-5;
    for (int i = 0; i < n; i++) {
        double A = result1[i];
        double B = result2[i];

        if (std::isnan(A) || std::isnan(B)) return 0;
        if (std::isinf(A) || std::isinf(B)) {
            if (A > B) return 1;
            else return 0;
        }

        double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
        double diff = A - B;

        if (fabs(diff) <= tol) {
            // Elements equal within tolerance, continue to next element
            continue;
        } else if (diff > tol) {
            // result1[i] > result2[i], lex condition satisfied
            return 1;
        } else {
            // result1[i] < result2[i], lex condition not satisfied
            return 0;
        }
    }
    // All elements equal within tolerance, no element greater, condition strictly requires '>' lexically
    return 0;
}
