int z2check(double result1, double result2) {
    // Core evaluation logic (direct scalar comparison) with floating-point tolerance handling
    double epsilon = 1e-5;
    double A = result1;
    double B = result2;

    // Defend against NaN
    if (std::isnan(A) || std::isnan(B)) return 0;

    // Handle Inf cases
    if (std::isinf(A) || std::isinf(B)) {
        if (A == B) return 1;
        else return 0;
    }

    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

    // Check equality with tolerance
    if (fabs(A - B) > tol) return 0;

    return 1;
}
