TestCase z12Change(TestCase t1) {
    TestCase follow_case = t1;
    double* follow_case_v = (double*)malloc(t1.n * sizeof(double));
    memcpy(follow_case_v, t1.v, t1.n * sizeof(double));
    follow_case.v = follow_case_v;
    if (follow_case.n <= 0) {
        return follow_case;
    }
    for (int i = 0; i < follow_case.n - 1; ++i) {
        for (int j = i + 1; j < follow_case.n; ++j) {
            if (follow_case.v[i] < follow_case.v[j]) {
                double temp = follow_case.v[i];
                follow_case.v[i] = follow_case.v[j];
                follow_case.v[j] = temp;
            }
        }
    }
    return follow_case;
}
