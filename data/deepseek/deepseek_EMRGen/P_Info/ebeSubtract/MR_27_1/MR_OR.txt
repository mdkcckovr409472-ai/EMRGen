#include <cmath>
#include <cfloat>

int z27check(double* result1, double* result2, int n) {
    // 1. 防御性编程 (空指针与长度检查)
    if (result1 == NULL || result2 == NULL) return 0;
    if (n <= 0) return 1;

    // 2. 字典序比较逻辑
    for (int i = 0; i < n; i++) {
        double A = result1[i];
        double B = result2[i];
        
        // 防御 NaN
        if (std::isnan(A) || std::isnan(B)) return 0;
        
        // 处理无穷大（动态容差对Inf无效）
        if (std::isinf(A) || std::isinf(B)) {
            if (A < B) return 0;  // 发现第一个不满足 >= 的元素对，立即返回0
            if (A > B) break;     // 发现第一个满足 > 的元素对，可以提前结束比较
            // A == B (Inf == Inf) 继续检查下一个元素
            continue;
        }
        
        // 动态容差比较（针对浮点数）
        double epsilon = 1e-5;
        double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
        
        // 检查是否满足 A >= B
        if (A < B - tol) {
            // 发现第一个不满足 >= 的元素对，立即返回0
            return 0;
        } else if (A > B + tol) {
            // 发现第一个满足 > 的元素对，可以提前结束比较
            break;
        }
        // A ≈ B (在容差范围内视为相等)，继续检查下一个元素
    }
    
    // 3. 全部满足，返回1
    return 1;
}
