int check(TestCase tc) {
    // Calculate initial output Os
    double Os = meanDifference(tc.sample1, tc.sample2, tc.n);

    // Generate follow-up test case
    TestCase Tf_case = change(tc);

    // Calculate follow-up output Of
    double Of = meanDifference(Tf_case.sample1, Tf_case.sample2, Tf_case.n);

    // Check for equality of Os and Of (with a small tolerance for floating point comparison)
    int is_valid = 0;
    const double epsilon = 1e-9;
    if ( (Of - Os > -epsilon) && (Of - Os < epsilon) ) {
        is_valid = 1;
    }

    // Free dynamically allocated memory in Tf_case
    delete[] Tf_case.sample2;

    return is_valid;
}
