int check(TestCase tc) {
    int Os = find_min(tc.a, tc.n);
    TestCase Tf_case = change(tc);
    int Of = find_min(Tf_case.a, Tf_case.n);

    int is_valid = 0;
    if (Os - 1 == Of) {
        is_valid = 1;
    }

    if (Tf_case.a != tc.a) {
        delete[] Tf_case.a;
        Tf_case.a = nullptr;
    }

    return is_valid;
}
