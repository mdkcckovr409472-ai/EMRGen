#include <math.h>
#include <stddef.h>

int z49check(double* result1, double* result2, int n) {
    // 1. 防御性编程 (空指针与长度检查)
    if (result1 == NULL || result2 == NULL) return 0;
    if (n <= 0) return 1;

    // 2. 核心判断逻辑 (字典序比较)
    double epsilon = 1e-5;
    for (int i = 0; i < n; i++) {
        double A = result1[i];
        double B = result2[i];

        // 防御 NaN
        if (isnan(A) || isnan(B)) return 0;

        // 处理无穷大
        if (isinf(A) || isinf(B)) {
            if (A < B) {
                // 在第一个不相等的元素处，result1[i] < result2[i]，满足字典序 <=，立即返回成功
                return 1;
            } else if (A > B) {
                // 在第一个不相等的元素处，result1[i] > result2[i]，违反字典序 <=，立即返回失败
                return 0;
            }
            // A 和 B 都是无穷大且相等，继续检查下一个元素
            continue;
        }

        // 计算动态容差
        double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

        // 检查第一个不相等的元素对
        if (fabs(A - B) > tol) {
            // 元素不相等，根据它们的关系决定字典序
            if (A < B - tol) {
                // result1[i] < result2[i]，满足字典序 <=，立即返回成功
                return 1;
            } else {
                // result1[i] > result2[i]，违反字典序 <=，立即返回失败
                return 0;
            }
        }
        // 元素在容差内相等，继续循环
    }

    // 3. 循环结束，所有元素在容差内相等，满足字典序 <=
    return 1;
}
