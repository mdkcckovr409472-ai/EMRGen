TestCase change(TestCase tc) {
    TestCase tf = tc;
    int* sorted_a = (int*)malloc(tc.n * sizeof(int));
    memcpy(sorted_a, tc.a, tc.n * sizeof(int));
    int t;
    for (int i = 0; i < tc.n - 1; i++) {
        for (int j = 0; j < tc.n - 1 - i; j++) {
            if (sorted_a[j] > sorted_a[j + 1]) {
                t = sorted_a[j];
                sorted_a[j] = sorted_a[j + 1];
                sorted_a[j + 1] = t;
            }
        }
    }
    tf.a = sorted_a;
    return tf;
}
