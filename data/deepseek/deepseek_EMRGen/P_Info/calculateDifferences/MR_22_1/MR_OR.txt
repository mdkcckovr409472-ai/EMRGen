#include <cmath>
#include <cfloat>

int z22check(double* result1, double* result2, int n) {
    // 1. 防御性编程 (空指针与长度检查)
    if (result1 == NULL || result2 == NULL) return 0;
    if (n <= 0) return 1;

    // 2. 核心判断逻辑 (字典序比较)
    for (int i = 0; i < n; i++) {
        double A = result1[i];
        double B = result2[i];

        // 防御 NaN
        if (std::isnan(A) || std::isnan(B)) return 0;

        // 处理无穷大
        if (std::isinf(A) || std::isinf(B)) {
            // 对于字典序，遇到第一个不相等的元素对就决定结果
            if (A > B) {
                // result1[i] > result2[i]，满足 >= 条件，整个字典序比较在此处已满足，返回1
                return 1;
            } else if (A < B) {
                // result1[i] < result2[i]，违反 >= 条件，返回0
                return 0;
            }
            // 如果相等，继续检查下一个元素
            continue;
        }

        // 常规浮点数比较（带容差）
        double epsilon = 1e-5;
        double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

        // 判断是否相等（在容差内）
        if (fabs(A - B) > tol) {
            // 第一个不相等的元素对
            if (A > B) {
                // result1[i] > result2[i]，满足 >= 条件
                return 1;
            } else {
                // result1[i] < result2[i]，违反 >= 条件
                return 0;
            }
        }
        // 如果相等（在容差内），继续检查下一个元素
    }

    // 3. 循环结束，所有元素都相等（在容差内），满足 >= 条件
    return 1;
}
