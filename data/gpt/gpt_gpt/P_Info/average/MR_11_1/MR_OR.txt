int check(TestCase tc) {
    double Os = average(tc.a, tc.n);
    TestCase Tf_case = change(tc);
    double Of = average(Tf_case.a, Tf_case.n);
    int is_valid = 0;
    if (tc.n > 0) {
        double diff = Os - Of;
        double expected = 1.0 / tc.n;
        double epsilon = 1e-9;
        if (diff > expected - epsilon && diff < expected + epsilon) {
            is_valid = 1;
        }
    } else {
        // For n == 0, no elements, output likely NaN or undefined, we treat as valid (no MR)
        is_valid = 1;
    }
    if (Tf_case.a != tc.a) {
        delete[] Tf_case.a;
    }
    return is_valid;
}
