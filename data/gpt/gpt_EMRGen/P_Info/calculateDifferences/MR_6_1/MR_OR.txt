#include <cmath>
#include <cstddef>

int z6check(double* result1, double* result2, int n) {
    // 4.a Defensive programming: length check first
    if (n <= 0) return 1;

    // 4.b Defensive programming: null pointer check second
    if (result1 == NULL || result2 == NULL) return 0;

    const double epsilon = 1e-5;

    for (int i = 0; i < n; i++) {
        double A = result1[i];
        double B = result2[i];

        // Step 3: NaN and inf handling
        if (std::isnan(A) || std::isnan(B)) return 0;

        if (std::isinf(A) || std::isinf(B)) {
            if (A > B) return 1;
            else return 0;
        }

        double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

        // Lexicographical comparison for return.result1[] > return.result2[] (lexically)
        // Using tolerance for floating point
        if (fabs(A - B) <= tol) {
            // elements equal within tolerance, continue checking next element
            continue;
        } else if (A > B + tol) {
            return 1;
        } else {
            return 0;
        }
    }

    // All elements equal (within tolerance), condition is strictly greater, so return 0
    return 0;
}
