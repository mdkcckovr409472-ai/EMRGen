int check(TestCase tc) {
    double* Os = square(tc.data, tc.n);
    if (Os == NULL) {
        return 0;
    }
    TestCase Tf_case = change(tc);
    double* Of = square(Tf_case.data, Tf_case.n);
    if (Of == NULL) {
        free(Os);
        return 0;
    }
    int is_valid = 1;
    for (int i = 0; i < tc.n; i++) {
        if (Of[i] != 4.0 * Os[i]) {
            is_valid = 0;
            break;
        }
    }
    free(Os);
    free(Of);
    return is_valid;
}
