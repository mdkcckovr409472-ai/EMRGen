#include <math.h>
#include <stddef.h>

int z4check(double result1, double result2) {
    // 浮点数比较处理
    // 1. 移项恢复量级：原条件 result1 - 4 * result2 == 0
    //    移项后：result1 == 4 * result2
    double A = result1;
    double B = 4.0 * result2;
    
    // 2. 防御NaN
    if (isnan(A) || isnan(B)) return 0;
    
    // 3. 处理无穷大
    if (isinf(A) || isinf(B)) {
        // 无穷大时直接数学比较
        if (A == B) return 1;
        else return 0;
    }
    
    // 4. 动态容差计算
    double epsilon = 1e-5;
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    // 5. 带容差的相等判断
    if (fabs(A - B) > tol) return 0;
    
    // 满足条件
    return 1;
}
