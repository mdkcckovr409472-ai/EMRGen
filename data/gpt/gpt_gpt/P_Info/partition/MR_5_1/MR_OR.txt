int check(TestCase tc) {
    // Call the function under test on the original input to get Os
    int Os = partition(tc.work, tc.begin, tc.end, tc.pivot);
    // Construct the follow-up test case
    TestCase Tf_case = change(tc);
    // Call the function under test on the follow-up input to get Of
    int Of = partition(Tf_case.work, Tf_case.begin, Tf_case.end, Tf_case.pivot);
    // Verify metamorphic relation Os <= Of
    int is_valid = (Os <= Of) ? 1 : 0;
    return is_valid;
}
