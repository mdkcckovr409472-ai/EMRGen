#include <cmath>
#include <cfloat>

int z28check(double* result1, double* result2, int n) {
    // 1. 防御性编程 (空指针与长度检查)
    if (result1 == NULL || result2 == NULL) return 0;
    if (n <= 0) return 1;

    // 2. 核心判断逻辑 (字典序比较)
    for (int i = 0; i < n; i++) {
        double A = result1[i];
        double B = result2[i];

        // 3. 浮点数特殊处理
        // 防御 NaN
        if (std::isnan(A) || std::isnan(B)) return 0;

        // 处理无穷大
        if (std::isinf(A) || std::isinf(B)) {
            // 无穷大时直接数学比较
            if (A < B) {
                // 第一个不相等的元素满足 result1[i] < result2[i]，字典序成立
                return 1;
            } else if (A > B) {
                // 第一个不相等的元素违反 result1[i] < result2[i]，字典序不成立
                return 0;
            }
            // A == B (包括 +Inf == +Inf, -Inf == -Inf)，继续检查下一个元素
            continue;
        }

        // 动态容差比较
        double epsilon = 1e-5;
        double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

        // 条件：A < B (移项后)
        if (A >= B - tol) {
            // 不满足 A < B，需要进一步判断
            if (A > B + tol) {
                // 明确违反 A < B
                return 0;
            }
            // 在容差范围内视为相等，继续检查下一个元素
        } else {
            // 满足 A < B，字典序成立
            return 1;
        }
    }

    // 循环结束，所有元素在容差范围内相等，不满足严格小于 (<)
    return 0;
}
