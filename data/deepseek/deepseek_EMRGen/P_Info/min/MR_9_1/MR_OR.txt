#include <math.h>
#include <stddef.h>

int z9check(double result1, double result2) {
    // 1. 核心判断逻辑 (标量直接比较)
    // 原条件: 2.81474976710656E14 * result1 - 2.81474976710656E14 * result2 + 8.84279719003555E14 == 0
    // 移项恢复量级: 2.81474976710656E14 * result1 + 8.84279719003555E14 == 2.81474976710656E14 * result2
    // 提取 A 和 B
    double A = 2.81474976710656E14 * result1 + 8.84279719003555E14;
    double B = 2.81474976710656E14 * result2;

    // 2. 浮点数特殊处理
    // 防御 NaN
    if (isnan(A) || isnan(B)) return 0;
    
    // 处理无穷大情况
    if (isinf(A) || isinf(B)) {
        // 无穷大时退化为精确数学比较
        if (A == B) return 1;
        else return 0;
    }
    
    // 3. 动态容差比较
    double epsilon = 1e-5;
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    // 移项后是相等比较 (==)
    if (fabs(A - B) > tol) return 0;
    
    // 4. 满足条件，返回 1
    return 1;
}
