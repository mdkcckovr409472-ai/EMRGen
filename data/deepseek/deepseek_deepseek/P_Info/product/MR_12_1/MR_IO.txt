TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tf.n > 0) {
        int index_to_negate = 0; // 选择第一个元素作为被替换的元素
        tf.elements = new double[tf.n];
        for (int i = 0; i < tf.n; ++i) {
            tf.elements[i] = tc.elements[i];
        }
        tf.elements[index_to_negate] = -tf.elements[index_to_negate];
    }
    return tf;
}
