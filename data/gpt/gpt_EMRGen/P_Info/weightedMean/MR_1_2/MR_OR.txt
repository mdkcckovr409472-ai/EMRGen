int z1check(double result1, double result2) {
    // Core evaluation logic (direct scalar comparison) with floating-point tolerance and special cases
    double epsilon = 1e-5;

    double A = result1;
    double B = result2;

    // Defend against NaN
    if (std::isnan(A) || std::isnan(B)) return 0;

    // Handle infinity cases
    if (std::isinf(A) || std::isinf(B)) {
        if (A < B) return 1; else return 0;
    }

    // Calculate dynamic tolerance
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

    // Evaluate condition: result1 < result2
    if (A >= B - tol) {
        return 0;
    }

    return 1;
}
