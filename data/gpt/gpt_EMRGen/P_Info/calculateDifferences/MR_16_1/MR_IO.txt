TestCase z16Change(TestCase t1) {
    TestCase follow_case = t1;
    // Deep copy of x array
    double* follow_case_x = (double*)malloc(t1.n * sizeof(double));
    memcpy(follow_case_x, t1.x, t1.n * sizeof(double));
    follow_case.x = follow_case_x;

    // Defensive check for empty array
    if (follow_case.n <= 0) {
        return follow_case;
    }

    // Replace each element with its absolute value
    for (int i = 0; i < follow_case.n; ++i) {
        follow_case.x[i] = fabs(follow_case.x[i]);
    }

    return follow_case;
}
