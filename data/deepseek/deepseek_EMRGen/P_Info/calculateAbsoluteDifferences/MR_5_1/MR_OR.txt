#include <math.h>
#include <stddef.h>

int z5check(double* result1, double* result2, int n) {
    // 1. 防御性编程 (空指针与长度检查)
    if (result1 == NULL || result2 == NULL) return 0;
    if (n <= 0) return 1;

    // 2. 核心判断逻辑 (elementwise)
    // 条件: result1[] is the reverse of result2[]
    // 这意味着对于所有 i: result1[i] == result2[n-1-i]
    for (int i = 0; i < n; i++) {
        // 提取移项后的操作数 A = result1[i], B = result2[n-1-i]
        double A = result1[i];
        double B = result2[n-1-i];
        
        // 防御 NaN
        if (isnan(A) || isnan(B)) return 0;
        
        // 处理无穷大
        if (isinf(A) || isinf(B)) {
            // 移项后的条件是 A == B
            if (A != B) return 0;
            // 如果相等，继续检查下一个元素
            continue;
        }
        
        // 定义基础容差
        double epsilon = 1e-5;
        // 计算动态宽容度
        double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
        
        // 移项后的条件是相等 (==)
        // 若 fabs(A - B) > tol，则违反条件
        if (fabs(A - B) > tol) {
            return 0;
        }
    }

    // 3. 全部满足，返回 1
    return 1;
}
