TestCase z29Change(TestCase t1) {
    TestCase follow_case = t1;
    // Memory handling code
    double* follow_case_elements = (double*)malloc(t1.n * sizeof(double));
    memcpy(follow_case_elements, t1.elements, t1.n * sizeof(double));
    follow_case.elements = follow_case_elements;
    // Defensive check for empty array
    if (follow_case.n <= 0) {
        return follow_case;
    }
    // Add a random constant between -0.5 and 0.5 to each element
    double rand_const = ((double)rand() / RAND_MAX) - 0.5;
    for (int i = 0; i < follow_case.n; ++i) {
        follow_case.elements[i] += rand_const;
    }
    return follow_case;
}
