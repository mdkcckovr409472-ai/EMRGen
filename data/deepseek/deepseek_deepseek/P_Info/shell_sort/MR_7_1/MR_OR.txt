int check(TestCase tc) {
    int* Os = shell_sort(tc.a, tc.n);
    TestCase Tf_case = change(tc);
    int* Of = shell_sort(Tf_case.a, Tf_case.n);
    int is_valid = 0;
    int i;
    int arrays_equal = 1;
    for (i = 0; i < tc.n; i++) {
        if (Os[i] != Of[i]) {
            arrays_equal = 0;
            break;
        }
    }
    if (arrays_equal) {
        is_valid = 1;
    }
    free(Os);
    free(Of);
    return is_valid;
}
