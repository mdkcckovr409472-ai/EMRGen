int check(TestCase tc) {
    int Os = dot_product(tc.a, tc.b, tc.n);
    TestCase Tf_case = change(tc);
    int Of = dot_product(Tf_case.a, Tf_case.b, Tf_case.n);
    int64_t expected_Of = (int64_t)Os - (int64_t)tc.a[0] * (int64_t)tc.b[0] + (int64_t)INT_MAX * (int64_t)tc.b[0];
    int is_valid = (Of == expected_Of) ? 1 : 0;
    if (Tf_case.n > 0) {
        delete[] Tf_case.a;
    }
    return is_valid;
}
