#include <cmath>
#include <cfloat>

int z2check(double result1, double result2) {
    // 1. 核心判断逻辑 (标量直接比较)
    // 条件: return.result1 - return.result2 - 1 == 0
    // 移项恢复量级: result1 - result2 == 1  →  result1 == result2 + 1
    // 提取A和B: A = result1, B = result2 + 1
    
    double A = result1;
    double B = result2 + 1.0;
    
    // 防御NaN
    if (std::isnan(A) || std::isnan(B)) return 0;
    
    // 处理无穷大
    if (std::isinf(A) || std::isinf(B)) {
        // 无穷大时退化为精确比较
        if (A == B) return 1;
        else return 0;
    }
    
    // 动态容差比较
    double epsilon = 1e-5;
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    // 检查相等性
    if (fabs(A - B) > tol) return 0;
    
    // 2. 满足条件，返回 1
    return 1;
}
