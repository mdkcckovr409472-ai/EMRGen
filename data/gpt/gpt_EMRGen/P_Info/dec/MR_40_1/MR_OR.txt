#include <cmath>
#include <algorithm>

int z40check(int* result1, int* result2, int n) {
    // 4.a Defensive length check
    if (n <= 0) return 1;
    // 4.b Defensive null pointer check
    if (result1 == NULL || result2 == NULL) return 0;

    // 3. Lexicographical comparison with tolerance (for floats/doubles, here int so tolerance not needed)
    // But since data_type is int*, and no floating-point, simple lexical comparison
    for (int i = 0; i < n; i++) {
        if (result1[i] > result2[i]) return 1;
        if (result1[i] < result2[i]) return 0;
        // if equal, continue to next element
    }
    // All elements equal, strictly greater lexically means return 0
    return 0;
}
