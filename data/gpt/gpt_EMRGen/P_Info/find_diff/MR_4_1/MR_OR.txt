#include <cmath>
#include <algorithm>

int z4check(int* result1, int* result2, int n) {
    // 1. Defensive programming (length check first)
    if (n <= 0) return 1;
    // 2. Defensive programming (null pointer check)
    if (result1 == NULL || result2 == NULL) return 0;

    for (int i = 0; i < n; i++) {
        int A = result1[i];
        int B = result2[i];
        if (A > B) return 1;
        else if (A < B) return 0;
        // else equal, continue checking next element
    }
    // all equal elements
    return 1;
}
