#include <math.h>
#include <stddef.h>

int z4check(double result1, double result2) {
    // 浮点数比较：result1 >= result2
    
    // 提取移项后的操作数
    double A = result1;
    double B = result2;
    
    // 防御NaN
    if (isnan(A) || isnan(B)) return 0;
    
    // 处理无穷大情况
    if (isinf(A) || isinf(B)) {
        // 直接数学比较
        if (A >= B) return 1;
        else return 0;
    }
    
    // 定义基础容差
    double epsilon = 1e-5;
    
    // 计算动态容差
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    // 带容差的比较：A >= B
    // 违反条件：A < B - tol
    if (A < B - tol) {
        return 0;
    }
    
    // 满足条件
    return 1;
}
