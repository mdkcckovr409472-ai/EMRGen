int check(TestCase tc) {
    int* Os = selection_sort(tc.list, tc.n);
    if (Os == NULL) {
        return 0;
    }
    TestCase Tf_case = change(tc);
    int* Of = selection_sort(Tf_case.list, Tf_case.n);
    if (Of == NULL) {
        free(Os);
        if (Tf_case.list != tc.list) {
            free(Tf_case.list);
        }
        return 0;
    }
    int is_valid = 1;
    for (int i = 0; i < tc.n; i++) {
        if (Os[i] != Of[i]) {
            is_valid = 0;
            break;
        }
    }
    free(Os);
    free(Of);
    if (Tf_case.list != tc.list) {
        free(Tf_case.list);
    }
    return is_valid;
}
