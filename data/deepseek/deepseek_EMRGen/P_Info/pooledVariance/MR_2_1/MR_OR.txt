#include <math.h>
#include <stddef.h>

int z2check(double result1, double result2) {
    // 浮点数比较：移项后条件为 result1 == result2
    double A = result1;
    double B = result2;
    
    // 防御 NaN
    if (isnan(A) || isnan(B)) return 0;
    
    // 处理无穷大（动态容差对无穷大无效）
    if (isinf(A) || isinf(B)) {
        if (A == B) return 1;
        else return 0;
    }
    
    // 计算动态容差
    double epsilon = 1e-5;
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    // 带容差的相等判断
    if (fabs(A - B) > tol) return 0;
    
    return 1;
}
