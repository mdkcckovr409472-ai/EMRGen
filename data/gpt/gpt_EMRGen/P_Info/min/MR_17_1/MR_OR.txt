int z17check(double result1, double result2) {
    // Step 1: Define the base tolerance
    const double epsilon = 1e-5;

    // Step 2: Algebraic rearrangement for return.result1 <= return.result2
    // Inequality: result1 <= result2
    // Rearranged: A = result1, B = result2

    double A = result1;
    double B = result2;

    // Step 3: Defend against NaN, handle Inf
    #include <math.h>
    if (isnan(A) || isnan(B)) return 0;
    if (isinf(A) || isinf(B)) {
        if (A <= B) return 1;
        else return 0;
    }

    // Step 4: Calculate dynamic tolerance
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

    // Step 5: Tolerance-based comparison evaluation for <=
    if (A > B + tol) return 0;

    return 1;
}
