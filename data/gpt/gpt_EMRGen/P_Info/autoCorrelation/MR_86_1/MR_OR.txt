int z86check(double result1, double result2) {
    // Step 1: Define tolerance epsilon
    double epsilon = 1e-5;

    // Step 2: Algebraic rearrangement of return.result1 + return.result2 == 0
    // The condition is result1 + result2 == 0  
    // Rearranged as: result1 == -result2
    double A = result1;
    double B = -result2;

    // Step 3: Check NaN and Inf
    if (std::isnan(A) || std::isnan(B)) return 0;
    if (std::isinf(A) || std::isinf(B)) {
        if (A == B) return 1; // Equal infinities satisfy ==
        else return 0;
    }

    // Step 4: Calculate dynamic tolerance
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

    // Step 5: Tolerance-based comparison for equality
    if (fabs(A - B) > tol) return 0;

    return 1;
}
