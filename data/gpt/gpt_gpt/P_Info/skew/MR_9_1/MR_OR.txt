int check(TestCase tc) {
    double Os = skew(tc.data, tc.n);
    TestCase Tf_case = change(tc);
    double Of = skew(Tf_case.data, Tf_case.n);
    int is_valid = 0;
    const double epsilon = 1e-9;
    if (fabs(Os - Of) < epsilon) {
        is_valid = 1;
    }
    if (Tf_case.data != tc.data) {
        delete[] Tf_case.data;
    }
    return is_valid;
}
