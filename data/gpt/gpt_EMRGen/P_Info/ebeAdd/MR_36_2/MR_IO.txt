    double* follow_case_b = (double*)malloc(t1.n2 * sizeof(double));
    memcpy(follow_case_b, t1.b, t1.n2 * sizeof(double));
    follow_case.b = follow_case_b;
    
    if (follow_case.n2 <= 0) {
        return follow_case;

}
