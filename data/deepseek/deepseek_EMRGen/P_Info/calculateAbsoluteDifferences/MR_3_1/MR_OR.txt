#include <cmath>
#include <cfloat>

int z3check(double* result1, double* result2, int n) {
    // 1. 防御性编程 (空指针与长度检查)
    if (result1 == NULL || result2 == NULL) return 0;
    if (n <= 0) return 1;

    // 2. 核心判断逻辑 (elementwise)
    for (int i = 0; i < n; i++) {
        // 提取移项后的操作数：条件 3 * result1[i] - result2[i] == 0 移项为 3.0 * result1[i] == result2[i]
        double A = 3.0 * result1[i];
        double B = result2[i];

        // 第三步：防御 NaN 和 Inf
        if (std::isnan(A) || std::isnan(B)) return 0;
        if (std::isinf(A) || std::isinf(B)) {
            // 对于无穷大，退化为精确数学比较
            if (!(A == B)) return 0;
            // 如果相等，继续检查下一个元素
            continue;
        }

        // 第四步：计算动态宽容度
        double epsilon = 1e-5;
        double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

        // 第五步：带容差的相等判定 (移项后为 ==)
        if (fabs(A - B) > tol) {
            return 0; // 一旦违反条件，立即返回 0
        }
    }

    // 3. 全部满足，返回 1
    return 1;
}
