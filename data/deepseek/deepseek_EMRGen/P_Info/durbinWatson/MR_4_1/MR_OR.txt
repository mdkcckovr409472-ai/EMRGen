#include <cmath>
#include <cfloat>

int z4check(double result1, double result2) {
    // 1. 核心判断逻辑 (标量直接比较)
    // 移项处理：return.result1 - 16 * return.result2 == 0
    // 等价于：result1 == 16 * result2
    double A = result1;
    double B = 16.0 * result2;

    // 浮点数特殊处理
    double epsilon = 1e-5;
    
    // 防御 NaN
    if (std::isnan(A) || std::isnan(B)) return 0;
    
    // 处理无穷大情况
    if (std::isinf(A) || std::isinf(B)) {
        if (A == B) return 1;
        else return 0;
    }
    
    // 计算动态容差
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    // 带容差的相等比较
    if (fabs(A - B) > tol) {
        return 0;
    }
    
    // 2. 满足条件，返回 1
    return 1;
}
