TestCase z40Change(TestCase t1) {
    TestCase follow_case = t1;
    // Deep copy of target array b
    int* follow_case_b = (int*)malloc(t1.na * sizeof(int));
    memcpy(follow_case_b, t1.b, t1.na * sizeof(int));
    follow_case.b = follow_case_b;

    // Defensive check: ensure there is at least one element
    if (follow_case.na == 0) {
        return follow_case;
    }

    // Increment each element of b by 1
    for (int i = 0; i < follow_case.na; ++i) {
        follow_case.b[i] += 1;
    }

    return follow_case;
}
