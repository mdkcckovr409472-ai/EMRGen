int z5check(int* result1, int* result2, int n) {
    if (n <= 0) return 1;
    if (result1 == NULL || result2 == NULL) return 0;

    const double epsilon = 1e-5;
    for (int i = 0; i < n; i++) {
        int A = result1[i];
        int B = result2[i];

        if (A == B) {
            continue;
        } else if (A > B) {
            return 0;
        } else { // A < B
            return 1;
        }
    }
    return 1;
}
