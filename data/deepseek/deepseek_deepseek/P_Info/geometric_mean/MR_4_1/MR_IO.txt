TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tf.n > 1 && tf.a != nullptr) {
        int p = 1; // 循环左移1个位置
        int* new_array = new int[tf.n];
        for (int i = 0; i < tf.n; i++) {
            new_array[i] = tf.a[(i + p) % tf.n];
        }
        tf.a = new_array;
    }
    return tf;
}
