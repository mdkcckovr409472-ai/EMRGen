#include <cmath>
#include <cfloat>

int z29check(double* result1, double* result2, int n) {
    // 1. 防御性编程 (空指针与长度检查)
    if (result1 == NULL || result2 == NULL) return 0;
    if (n <= 0) return 1;

    // 2. 核心判断逻辑 (elementwise)
    for (int i = 0; i < n; i++) {
        // 条件: 2 * result1[i] - 2 * result2[i] + 1 == 0
        // 移项: 2 * result1[i] + 1 == 2 * result2[i]
        double A = 2.0 * result1[i] + 1.0;
        double B = 2.0 * result2[i];

        // 防御 NaN
        if (std::isnan(A) || std::isnan(B)) return 0;

        // 处理无穷大
        if (std::isinf(A) || std::isinf(B)) {
            if (A != B) return 0; // 对于等式，无穷大必须相等
            // 如果 A 和 B 都是同号无穷大，则继续循环
            continue;
        }

        // 动态容差比较
        double epsilon = 1e-5;
        double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
        if (fabs(A - B) > tol) {
            return 0; // 一旦违反条件，立即返回 0
        }
    }

    // 3. 全部满足，返回 1
    return 1;
}
