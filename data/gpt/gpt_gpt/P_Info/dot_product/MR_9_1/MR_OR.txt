int check(TestCase tc) {
    int Os = dot_product(tc.a, tc.b, tc.n);
    
    TestCase Tf_case = change(tc);
    int Of = dot_product(Tf_case.a, Tf_case.b, Tf_case.n);
    
    int last_b = tc.b[tc.n - 1];
    int last_a = tc.a[tc.n - 1];
    
    int expected_Of = Os - last_b * last_a + INT_MIN * last_a;
    
    int is_valid = (Of == expected_Of) ? 1 : 0;
    
    delete[] Tf_case.b;
    
    return is_valid;
}
