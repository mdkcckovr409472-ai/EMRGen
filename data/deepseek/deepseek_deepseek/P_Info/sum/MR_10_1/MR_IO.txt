TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tf.n > 1) {
        int* new_values = new int[tf.n];
        for (int i = 0; i < tf.n; i++) {
            new_values[i] = tf.values[i];
        }
        int idx1 = 0;
        int idx2 = tf.n - 1;
        int temp = new_values[idx1];
        new_values[idx1] = new_values[idx2];
        new_values[idx2] = temp;
        tf.values = new_values;
    }
    return tf;
}
