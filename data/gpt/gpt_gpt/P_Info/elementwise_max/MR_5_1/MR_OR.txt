int check(TestCase tc) {
    if (tc.a == NULL || tc.b == NULL || tc.n <= 0) {
        return 0;
    }

    int* Os = elementwise_max(tc.a, tc.b, tc.n);
    if (Os == NULL) {
        return 0;
    }

    TestCase Tf_case = change(tc);
    if (Tf_case.b == NULL || Tf_case.a == NULL || Tf_case.n <= 0) {
        free(Os);
        if (Tf_case.b != tc.b) {
            free(Tf_case.b);
        }
        return 0;
    }

    int* Of = elementwise_max(Tf_case.a, Tf_case.b, Tf_case.n);

    int is_valid = 1;
    if (Of == NULL) {
        is_valid = 0;
    } else {
        for (int i = 0; i < tc.n; i++) {
            if (Os[i] < Of[i]) {
                is_valid = 0;
                break;
            }
        }
    }

    free(Os);
    if (Tf_case.b != tc.b) {
        free(Tf_case.b);
    }
    if (Of != NULL) {
        free(Of);
    }

    return is_valid;
}
