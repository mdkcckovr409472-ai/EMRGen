TestCase z12Change(TestCase t1) {
    TestCase follow_case = t1;
    // Memory handling for array1 (deep copy)
    int* follow_case_array1 = (int*)malloc(t1.n * sizeof(int));
    memcpy(follow_case_array1, t1.array1, t1.n * sizeof(int));
    follow_case.array1 = follow_case_array1;

    // Defensive check: ensure there is at least one element
    if (follow_case.n <= 0) {
        return follow_case;
    }

    // Take absolute value of each element in array1
    for (int i = 0; i < follow_case.n; ++i) {
        if (follow_case.array1[i] < 0) {
            follow_case.array1[i] = -follow_case.array1[i];
        }
    }

    return follow_case;
}
