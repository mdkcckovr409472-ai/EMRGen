TestCase z11Change(TestCase t1) {
    TestCase follow_case = t1;
    // --- Memory handling code ---
    double* follow_case_elements = (double*)malloc(t1.n * sizeof(double));
    memcpy(follow_case_elements, t1.elements, t1.n * sizeof(double));
    follow_case.elements = follow_case_elements;
    // ---------------------------

    // Defensive check: operation requires at least 1 element
    if (follow_case.n > 0) {
        for (int i = 0; i < follow_case.n / 2; ++i) {
            double temp = follow_case.elements[i];
            follow_case.elements[i] = follow_case.elements[follow_case.n - 1 - i];
            follow_case.elements[follow_case.n - 1 - i] = temp;
        }
    }

    return follow_case;
}
