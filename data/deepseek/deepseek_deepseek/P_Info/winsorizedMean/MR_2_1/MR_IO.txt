TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tf.n > 0 && tf.elements != nullptr) {
        int half = tf.n / 2;
        for (int i = 0; i < half; ++i) {
            double temp = tf.elements[i];
            tf.elements[i] = tf.elements[tf.n - 1 - i];
            tf.elements[tf.n - 1 - i] = temp;
        }
    }
    return tf;
}
