#include <cmath>
#include <cfloat>

int z15check(double result1, double result2) {
    // 1. 防御性编程：检查 NaN
    if (std::isnan(result1) || std::isnan(result2)) return 0;
    
    // 2. 处理无穷大情况
    if (std::isinf(result1) || std::isinf(result2)) {
        // 移项后的条件为 result1 <= result2
        if (result1 <= result2) return 1;
        else return 0;
    }
    
    // 3. 浮点数比较的特殊处理
    double epsilon = 1e-5;
    
    // 移项后的条件为 result1 <= result2
    // 所以 A = result1, B = result2
    double A = result1;
    double B = result2;
    
    // 计算动态容差
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    // 根据移项后的条件 result1 <= result2 进行判定
    // 条件为 A <= B，违反条件是 A > B
    // 带容差的判定：若 A > B + tol，则违反条件
    if (A > B + tol) return 0;
    
    // 4. 满足条件，返回 1
    return 1;
}
