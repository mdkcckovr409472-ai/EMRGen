#include <cmath>
#include <cstddef>

int z1check(double* result1, double* result2, int n) {
    // 1. 防御性编程 (空指针与长度检查)
    if (result1 == NULL || result2 == NULL) return 0;
    if (n <= 0) return 1;

    // 2. 核心判断逻辑 (elementwise)
    for (int i = 0; i < n; i++) {
        // 根据条件 return.result1[] - return.result2[] - 1 == 0
        // 移项后：result1[i] == result2[i] + 1
        double A = result1[i];
        double B = result2[i] + 1.0;

        // 第三步：防御 NaN 和 Inf
        if (std::isnan(A) || std::isnan(B)) return 0;
        if (std::isinf(A) || std::isinf(B)) {
            // 对于无穷大，退化为精确比较
            if (!(A == B)) return 0;
            continue; // 当前元素比较通过，检查下一个
        }

        // 第一步：定义基础容差
        const double epsilon = 1e-5;
        // 第四步：计算动态宽容度
        double tol = epsilon * std::fmax(1.0, std::fmax(std::fabs(A), std::fabs(B)));

        // 第五步：带容差的相等判定 (==)
        // 若违反则 return 0
        if (std::fabs(A - B) > tol) {
            return 0;
        }
    }

    // 3. 全部满足，返回 1
    return 1;
}
