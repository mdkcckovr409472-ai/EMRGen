TestCase z2Change(TestCase t1) {
    TestCase follow_case = t1;
    // --- Memory handling code ---
    int* follow_case_values = (int*)malloc(t1.n * sizeof(int));
    memcpy(follow_case_values, t1.values, t1.n * sizeof(int));
    follow_case.values = follow_case_values;
    // -------------------------

    // Defensive check: ensure there is at least one element
    if (follow_case.n == 0) {
        return follow_case;
    }

    // Operation: subtract 5 from a random element
    int random_index = rand() % follow_case.n;
    follow_case.values[random_index] -= 5;

    return follow_case;
}
