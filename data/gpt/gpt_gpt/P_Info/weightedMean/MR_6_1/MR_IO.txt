TestCase change(TestCase tc) {
    TestCase tf = tc;
    // Allocate new memory for elements to modify safely without altering original input
    tf.elements = new double[tc.n];
    for (int i = 0; i < tc.n; i++) {
        tf.elements[i] = tc.elements[i];
    }
    // Find minimum element value
    double min_val = tf.elements[0];
    for (int i = 1; i < tc.n; i++) {
        if (tf.elements[i] < min_val) {
            min_val = tf.elements[i];
        }
    }
    // Set first element to minimum value
    tf.elements[0] = min_val;
    return tf;
}
