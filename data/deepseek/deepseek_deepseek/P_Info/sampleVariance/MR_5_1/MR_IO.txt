TestCase change(TestCase tc) {
    TestCase tf = tc;
    // 为排序分配新的数组，避免修改原始数据
    tf.elements = new double[tc.n];
    for (int i = 0; i < tc.n; ++i) {
        tf.elements[i] = tc.elements[i];
    }
    // 对数组进行降序排序
    std::sort(tf.elements, tf.elements + tf.n, std::greater<double>());
    return tf;
}
