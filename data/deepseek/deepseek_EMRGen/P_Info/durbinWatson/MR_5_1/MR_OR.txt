#include <math.h>
#include <stddef.h>

int z5check(double result1, double result2) {
    // 1. 防御NaN
    if (isnan(result1) || isnan(result2)) return 0;
    
    // 2. 处理无穷大（动态容差对无穷大无效）
    if (isinf(result1) || isinf(result2)) {
        // 移项后条件为 result1 == result2
        if (result1 == result2) return 1;
        else return 0;
    }
    
    // 3. 定义基础容差
    double epsilon = 1e-5;
    
    // 4. 移项提取A和B（条件为 result1 == result2）
    double A = result1;
    double B = result2;
    
    // 5. 计算动态容差
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    // 6. 带容差的相等判断
    if (fabs(A - B) > tol) {
        return 0;
    }
    
    // 7. 满足条件
    return 1;
}
