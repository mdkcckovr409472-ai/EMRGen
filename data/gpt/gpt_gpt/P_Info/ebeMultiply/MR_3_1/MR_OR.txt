int check(TestCase tc) {
    if (tc.a == NULL || tc.b == NULL || tc.n1 != tc.n2 || tc.n1 <= 0) {
        return 0;
    }

    double* Os = ebeMultiply(tc.a, tc.b, tc.n1, tc.n2);
    if (Os == NULL) {
        return 0;
    }

    TestCase Tf_case = change(tc);

    double* Of = NULL;
    if (Tf_case.a != tc.a) {
        Of = ebeMultiply(Tf_case.a, Tf_case.b, Tf_case.n1, Tf_case.n2);
    } else {
        // If change failed to allocate new a, follow-up input is same as source, call with original
        Of = ebeMultiply(tc.a, tc.b, tc.n1, tc.n2);
    }

    if (Of == NULL) {
        free(Os);
        if (Tf_case.a != tc.a) {
            free(Tf_case.a);
        }
        return 0;
    }

    int is_valid = 1;
    for (int i = 0; i < tc.n1; i++) {
        // Check if Of[i] == 2 * Os[i]
        if (Of[i] != 2.0 * Os[i]) {
            is_valid = 0;
            break;
        }
    }

    free(Os);
    free(Of);

    if (Tf_case.a != tc.a) {
        free(Tf_case.a);
    }

    return is_valid;
}
