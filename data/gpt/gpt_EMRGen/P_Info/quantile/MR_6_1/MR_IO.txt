TestCase z6Change(TestCase t1) {
    TestCase follow_case = t1;
    // Deep copy of sortedElements array
    double* follow_case_sortedElements = (double*)malloc(t1.n * sizeof(double));
    memcpy(follow_case_sortedElements, t1.sortedElements, t1.n * sizeof(double));
    follow_case.sortedElements = follow_case_sortedElements;

    // Defensive check: need at least one element
    if (follow_case.n == 0) {
        return follow_case;
    }

    // Set a random element to 100.0 if it is less than 100.0
    int random_index = rand() % follow_case.n;
    if (follow_case.sortedElements[random_index] < 100.0) {
        follow_case.sortedElements[random_index] = 100.0;
    }

    return follow_case;
}
