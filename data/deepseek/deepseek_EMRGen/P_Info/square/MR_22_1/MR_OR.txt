#include <cmath>
#include <cfloat>

int z22check(double* result1, double* result2, int n) {
    // 1. 防御性编程 (空指针与长度检查)
    if (result1 == NULL || result2 == NULL) return 0;
    if (n <= 0) return 1;

    // 2. 核心判断逻辑 (elementwise)
    for (int i = 0; i < n; i++) {
        // 提取移项后的操作数：A = result1[i], B = result2[i]
        double A = result1[i];
        double B = result2[i];

        // 防御 NaN
        if (std::isnan(A) || std::isnan(B)) return 0;

        // 处理无穷大
        if (std::isinf(A) || std::isinf(B)) {
            // 直接数学比较：A >= B
            if (!(A >= B)) return 0;
            continue; // 此对元素比较完毕，继续下一对
        }

        // 定义基础容差
        double epsilon = 1e-5;
        // 计算动态容差
        double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

        // 带容差的比较判定：检查是否违反 A >= B
        // 违反条件为：A < B - tol
        if (A < B - tol) {
            return 0; // 一旦违反条件，立即返回 0
        }
    }

    // 3. 全部满足，返回 1
    return 1;
}
