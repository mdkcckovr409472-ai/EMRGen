int check(TestCase tc) {
    if (tc.a == nullptr || tc.n <= 0) {
        return 0;
    }
    int* Os = dec_array(tc.a, tc.k, tc.n);
    if (Os == nullptr) {
        return 0;
    }
    TestCase Tf_case = change(tc);
    if (Tf_case.a == nullptr || Tf_case.n <= 0) {
        free(Os);
        if (Tf_case.a != tc.a) {
            free(Tf_case.a);
        }
        return 0;
    }
    int* Of = dec_array(Tf_case.a, Tf_case.k, Tf_case.n);
    if (Of == nullptr) {
        free(Os);
        if (Tf_case.a != tc.a) {
            free(Tf_case.a);
        }
        return 0;
    }
    int is_valid = 1;
    for (int i = 0; i < tc.n; i++) {
        // Check MR: Os[i] + n == Of[i]
        if (Os[i] + tc.n != Of[i]) {
            is_valid = 0;
            break;
        }
    }
    free(Os);
    free(Of);
    if (Tf_case.a != tc.a) {
        free(Tf_case.a);
    }
    return is_valid;
}
