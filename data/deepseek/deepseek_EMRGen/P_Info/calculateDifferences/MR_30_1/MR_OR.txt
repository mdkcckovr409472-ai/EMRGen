#include <cmath>
#include <cfloat>

int z30check(double* result1, double* result2, int n) {
    // 1. 防御性编程 (空指针与长度检查)
    if (result1 == NULL || result2 == NULL) return 0;
    if (n <= 0) return 1;

    // 2. 核心判断逻辑 (elementwise)
    for (int i = 0; i < n; i++) {
        // 移项：result1[i] - result2[i] - 1.5 == 0  ->  result1[i] == result2[i] + 1.5
        double A = result1[i];
        double B = result2[i] + 1.5;

        // 第三步：防御NaN，处理无穷大
        if (std::isnan(A) || std::isnan(B)) return 0;
        if (std::isinf(A) || std::isinf(B)) {
            // 无穷大时退化为直接数学比较
            if (A != B) return 0;
            continue; // 相等则继续检查下一个元素
        }

        // 第一步：定义基础容差
        double epsilon = 1e-5;
        // 第四步：计算动态宽容度
        double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

        // 第五步：带容差的相等判定 (移项后为 ==)
        if (fabs(A - B) > tol) {
            return 0; // 一旦违反条件，立即返回 0
        }
    }

    // 3. 全部满足，返回 1
    return 1;
}
