int check(TestCase tc) {
    double* Os = standardize(tc.data, tc.n);
    if (Os == nullptr) return 0;

    TestCase Tf_case = change(tc);
    double* Of = standardize(Tf_case.data, Tf_case.n);
    if (Of == nullptr) {
        free(Os);
        return 0;
    }

    int is_valid = 1;
    for (int i = 0; i < tc.n; i++) {
        if (fabs(Os[i] - Of[i]) > 1e-9) {
            is_valid = 0;
            break;
        }
    }

    free(Os);
    free(Of);

    return is_valid;
}
