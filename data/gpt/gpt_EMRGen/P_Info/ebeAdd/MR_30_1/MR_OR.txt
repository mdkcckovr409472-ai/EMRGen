int z30check(double* result1, double* result2, int n) {
    // 1. Defensive programming (Length check first)
    if (n <= 0) return 1;
    // 2. Null pointer check after length check
    if (result1 == NULL || result2 == NULL) return 0;

    const double epsilon = 1e-5;
    for (int i = 0; i < n; i++) {
        // Rearranged condition: result1[i] == result2[i] - 1
        // Let A = result1[i], B = result2[i] - 1
        double A = result1[i];
        double B = result2[i] - 1.0;

        if (std::isnan(A) || std::isnan(B)) return 0;
        if (std::isinf(A) || std::isinf(B)) {
            if (A == B) continue;
            else return 0;
        }

        double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
        if (fabs(A - B) > tol) return 0;
    }
    return 1;
}
