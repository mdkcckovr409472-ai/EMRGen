TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tf.n > 0 && tf.data != nullptr) {
        double min_val = tf.data[0];
        for (int i = 1; i < tf.n; i++) {
            if (tf.data[i] < min_val) {
                min_val = tf.data[i];
            }
        }
        tf.data[tf.n - 1] = min_val;
    }
    return tf;
}
