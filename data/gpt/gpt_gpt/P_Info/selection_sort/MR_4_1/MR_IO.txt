TestCase change(TestCase tc) {
    TestCase tf = tc;
    // Create a new list array for follow-up input, divide each element by 2
    int* new_list = (int*)malloc(tc.n * sizeof(int));
    for (int i = 0; i < tc.n; i++) {
        new_list[i] = tc.list[i] / 2;
    }
    tf.list = new_list;
    return tf;
}
