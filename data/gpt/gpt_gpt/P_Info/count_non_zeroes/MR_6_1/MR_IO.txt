TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tf.n > 1) {
        // Allocate new array for shifted elements
        int* shifted = new int[tf.n];
        for (int i = 0; i < tf.n - 1; i++) {
            shifted[i] = tf.a[i + 1];
        }
        shifted[tf.n - 1] = tf.a[0];
        tf.a = shifted;
    }
    // If n <= 1, reuse the original array without change
    return tf;
}
