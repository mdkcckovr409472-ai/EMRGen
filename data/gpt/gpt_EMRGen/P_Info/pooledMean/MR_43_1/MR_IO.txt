TestCase z43Change(TestCase t1) {
    TestCase follow_case = t1;
    // Memory handling for data2
    double* follow_case_data2 = (double*)malloc(t1.n2 * sizeof(double));
    memcpy(follow_case_data2, t1.data2, t1.n2 * sizeof(double));
    follow_case.data2 = follow_case_data2;
    // Defensive check for empty data2 array
    if (follow_case.n2 < 1) {
        return follow_case;
    }
    // Exponential operation on data2
    for (int i = 0; i < follow_case.n2; ++i) {
        follow_case.data2[i] = exp(follow_case.data2[i]);
    }
    return follow_case;
}
