int z36check(double result1, double result2) {
    double epsilon = 1e-5;

    // Algebraic rearrangement for: result1 < result2
    double A = result1;
    double B = result2;

    // NaN and Inf checks
    if (std::isnan(A) || std::isnan(B)) return 0;
    if (std::isinf(A) || std::isinf(B)) {
        if (A < B) return 1; else return 0;
    }

    // Dynamic tolerance calculation
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

    // Tolerance-based comparison: result1 < result2
    if (A >= B - tol) return 0;

    return 1;
}
