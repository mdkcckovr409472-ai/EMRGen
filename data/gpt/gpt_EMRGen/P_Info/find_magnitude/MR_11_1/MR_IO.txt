TestCase z11Change(TestCase t1) {
    TestCase follow_case = t1;
    // Memory handling for deep copy
    int* follow_case_a = (int*)malloc(t1.n * sizeof(int));
    memcpy(follow_case_a, t1.a, t1.n * sizeof(int));
    follow_case.a = follow_case_a;
    // Defensive check: need at least 2 elements to perform reversal
    if (follow_case.n < 2) {
        return follow_case;
    }
    // Reverse array
    int left = 0;
    int right = follow_case.n - 1;
    while (left < right) {
        int temp = follow_case.a[left];
        follow_case.a[left] = follow_case.a[right];
        follow_case.a[right] = temp;
        left++;
        right--;
    }
    return follow_case;
}
