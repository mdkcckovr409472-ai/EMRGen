TestCase change(TestCase tc) {
    TestCase tf = tc;

    // Allocate new array for tf.a to hold elements of tc.a minus 1
    if (tc.a == NULL || tc.n <= 0) {
        tf.a = NULL;
        return tf;
    }

    int* new_a = (int*)malloc(tc.n * sizeof(int));
    if (new_a == NULL) {
        tf.a = NULL;
        return tf;
    }

    for (int i = 0; i < tc.n; i++) {
        new_a[i] = tc.a[i] - 1;
    }

    tf.a = new_a;
    return tf;
}
