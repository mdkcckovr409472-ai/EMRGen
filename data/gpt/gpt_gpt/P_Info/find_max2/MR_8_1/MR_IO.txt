TestCase change(TestCase tc) {
    TestCase tf = tc;
    // We must create a new array to apply the change without modifying original input
    tf.a = new int[tf.n];
    for (int i = 0; i < tf.n; i++) {
        tf.a[i] = tc.a[i];
    }
    if (tf.n > 0) {
        tf.a[0] -= 2;
    }
    return tf;
}
