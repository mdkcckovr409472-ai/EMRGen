#include <cmath>
#include <cfloat>

int z2check(double* result1, double* result2, int n) {
    // 1. 防御性编程 (空指针与长度检查)
    if (result1 == NULL || result2 == NULL) return 0;
    if (n <= 0) return 1;

    // 2. 核心判断逻辑 (字典序比较)
    const double epsilon = 1e-5;
    
    for (int i = 0; i < n; i++) {
        double A = result1[i];
        double B = result2[i];
        
        // 防御 NaN
        if (std::isnan(A) || std::isnan(B)) return 0;
        
        // 处理无穷大情况
        if (std::isinf(A) || std::isinf(B)) {
            // 对于字典序比较，在第一个不相等的元素处决定结果
            if (A > B) {
                // result1[i] > result2[i]，满足 >= 条件，整个字典序比较成功
                return 1;
            } else if (A < B) {
                // result1[i] < result2[i]，违反 >= 条件
                return 0;
            }
            // 相等则继续比较下一个元素
            continue;
        }
        
        // 计算动态容差
        double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
        
        // 检查是否不相等（考虑容差）
        if (fabs(A - B) > tol) {
            // 找到第一个不相等的元素对
            if (A > B) {
                // result1[i] > result2[i]，满足 >= 条件
                return 1;
            } else {
                // result1[i] < result2[i]，违反 >= 条件
                return 0;
            }
        }
        // 在容差范围内视为相等，继续比较下一个元素
    }
    
    // 3. 所有元素在容差范围内相等，满足 >= 条件
    return 1;
}
