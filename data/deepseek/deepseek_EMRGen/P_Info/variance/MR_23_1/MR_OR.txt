#include <math.h>
#include <stddef.h>

int z23check(double result1, double result2) {
    // 1. 防御性编程 (标量无需空指针和长度检查)

    // 2. 核心判断逻辑 (标量直接比较)
    // 条件: result1 == result2
    // 移项后: A = result1, B = result2, 运算符 ==

    double A = result1;
    double B = result2;

    // 第三步: 防御 NaN 和 Inf
    if (isnan(A) || isnan(B)) return 0;
    if (isinf(A) || isinf(B)) {
        // 对于无穷大，退化为数学直接比较
        if (A == B) return 1;
        else return 0;
    }

    // 第一步 & 第四步: 定义基础容差并计算动态容差
    double epsilon = 1e-5;
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

    // 第五步: 带容差的相等判定
    if (fabs(A - B) > tol) {
        return 0;
    }

    // 3. 满足条件，返回 1
    return 1;
}
