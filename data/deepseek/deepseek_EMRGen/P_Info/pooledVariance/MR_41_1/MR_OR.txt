#include <math.h>
#include <stddef.h>

int z41check(double result1, double result2) {
    // 1. 防御性编程 (标量版本无需指针和长度检查)

    // 2. 核心判断逻辑 (标量直接比较)
    // 条件不变式: return.result1 == return.result2
    // 移项后: result1 == result2

    // 浮点数比较的特殊处理
    double epsilon = 1e-5;
    
    // 提取左右操作数 (已移项)
    double A = result1;
    double B = result2;
    
    // 防御 NaN
    if (isnan(A) || isnan(B)) return 0;
    
    // 处理无穷大情况
    if (isinf(A) || isinf(B)) {
        if (A == B) return 1;
        else return 0;
    }
    
    // 计算动态宽容度
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    // 带容差的相等比较
    if (fabs(A - B) > tol) {
        return 0;
    }
    
    // 3. 满足条件，返回 1
    return 1;
}
