TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tc.n > 0 && tc.elements != nullptr) {
        double* newElements = new double[tc.n];
        for (int i = 0; i < tc.n; i++) {
            newElements[i] = tc.elements[i] - 1.0;
        }
        tf.elements = newElements;
    }
    return tf;
}
