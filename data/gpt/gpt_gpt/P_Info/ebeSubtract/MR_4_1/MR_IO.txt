TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tf.a != NULL && tf.n1 > 0) {
        double* new_a = (double*)malloc(tf.n1 * sizeof(double));
        if (new_a == NULL) {
            // Allocation failed, return original tc unchanged
            return tf;
        }
        for (int i = 0; i < tf.n1; i++) {
            new_a[i] = tf.a[i];
        }
        new_a[0] = DBL_MIN;
        tf.a = new_a;
    }
    return tf;
}
