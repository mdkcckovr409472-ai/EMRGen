int check(TestCase tc) {
    double Os = evaluateNewton(tc.a, tc.c, tc.z, tc.nc);
    TestCase Tf_case = change(tc);
    double Of = evaluateNewton(Tf_case.a, Tf_case.c, Tf_case.z, Tf_case.nc);
    int is_valid = 0;
    double m = 2.0;
    if (fabs(Os / m - Of) < 1e-9) {
        is_valid = 1;
    }
    delete[] Tf_case.a;
    return is_valid;
}
