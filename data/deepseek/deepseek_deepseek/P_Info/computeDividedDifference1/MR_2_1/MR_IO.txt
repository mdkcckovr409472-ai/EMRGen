TestCase change(TestCase tc) {
    TestCase tf = tc;
    double k = 5.0; // 常数k，示例值，可根据需要调整
    if (tf.x != nullptr && tf.n > 0) {
        double* new_x = new double[tf.n];
        for (int i = 0; i < tf.n; i++) {
            new_x[i] = tf.x[i] - k;
        }
        tf.x = new_x;
    }
    return tf;
}
