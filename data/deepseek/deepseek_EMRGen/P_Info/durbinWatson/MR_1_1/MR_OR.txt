#include <math.h>
#include <stddef.h>

int z1check(double result1, double result2) {
    // 1. 防御性编程 (标量情况无需空指针和长度检查)

    // 2. 核心判断逻辑 (标量直接比较)
    // 条件: result1 == result2
    // 移项后: result1 == result2 (无需移项)
    double A = result1;
    double B = result2;

    // 3. 浮点数特殊处理
    // 防御 NaN
    if (isnan(A) || isnan(B)) return 0;
    
    // 处理无穷大
    if (isinf(A) || isinf(B)) {
        // 无穷大时退化为精确比较
        if (A == B) return 1;
        else return 0;
    }
    
    // 计算动态容差
    double epsilon = 1e-5;
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    // 带容差的相等判断
    if (fabs(A - B) > tol) {
        return 0;
    }

    // 4. 满足条件，返回 1
    return 1;
}
