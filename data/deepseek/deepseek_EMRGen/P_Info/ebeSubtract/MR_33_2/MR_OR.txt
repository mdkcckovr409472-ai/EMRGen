#include <cmath>
#include <cfloat>

int z33check(double* result1, double* result2, int n) {
    // 1. 防御性编程 (空指针与长度检查)
    if (result1 == NULL || result2 == NULL) return 0;
    if (n <= 0) return 1;

    // 2. 核心判断逻辑 (elementwise)
    for (int i = 0; i < n; i++) {
        // 提取移项后的操作数 A 和 B
        double A = result1[i];
        double B = result2[i];

        // 防御 NaN
        if (std::isnan(A) || std::isnan(B)) return 0;

        // 处理无穷大
        if (std::isinf(A) || std::isinf(B)) {
            // 移项后的条件为 A < B
            if (!(A < B)) return 0;
            continue; // 此对元素检查通过，继续下一对
        }

        // 定义基础容差
        double epsilon = 1e-5;
        // 计算动态宽容度
        double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

        // 移项后的条件为 A < B
        // 带容差的比较：若 A >= B - tol，则违反条件
        if (A >= B - tol) {
            return 0;
        }
    }

    // 3. 全部满足，返回 1
    return 1;
}
