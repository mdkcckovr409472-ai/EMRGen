TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tf.n > 1 && tf.elements != nullptr) {
        double* new_elements = new double[tf.n];
        new_elements[tf.n - 1] = tf.elements[0];
        for (int i = 0; i < tf.n - 1; ++i) {
            new_elements[i] = tf.elements[i + 1];
        }
        tf.elements = new_elements;
    }
    return tf;
}
