int check(TestCase tc) {
    // Source input
    int* a = tc.a;
    int k = tc.k;
    int n = tc.n;
    
    // Source output
    int* Os = dec_array(a, k, n);
    if (!Os) return 0;
    
    // Follow-up test case (Tf)
    TestCase Tf_case = change(tc);
    if (!Tf_case.a) {
        free(Os);
        return 0;
    }
    
    int* Of = dec_array(Tf_case.a, Tf_case.k, Tf_case.n);
    if (!Of) {
        free(Os);
        free(Tf_case.a);
        return 0;
    }
    
    // Verify MR: Os == Of (element wise equality)
    int is_valid = 1;
    for (int i = 0; i < n; i++) {
        if (Os[i] != Of[i]) {
            is_valid = 0;
            break;
        }
    }
    
    // Free all dynamically allocated memories
    free(Os);
    free(Of);
    free(Tf_case.a);
    
    return is_valid;
}
