TestCase z30Change(TestCase t1) {
    TestCase follow_case = t1;
    // Memory handling for target array y
    double* follow_case_y = (double*)malloc(t1.n * sizeof(double));
    memcpy(follow_case_y, t1.y, t1.n * sizeof(double));
    follow_case.y = follow_case_y;

    // Defensive check: ensure there is at least one element
    if (follow_case.n < 1) {
        return follow_case;
    }

    // Specific transformation: set last element to -1.0 if it is greater than -1.0
    if (follow_case.y[follow_case.n - 1] > -1.0) {
        follow_case.y[follow_case.n - 1] = -1.0;
    }

    return follow_case;
}
