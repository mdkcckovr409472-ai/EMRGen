TestCase z22Change(TestCase t1) {
    TestCase follow_case = t1;
    // Deep copy of the array
    double* follow_case_z = (double*)malloc(t1.n * sizeof(double));
    memcpy(follow_case_z, t1.z, t1.n * sizeof(double));
    follow_case.z = follow_case_z;

    // Defensive check for empty array
    if (follow_case.n <= 0) {
        return follow_case;
    }

    // Multiply elements at even indices by -1
    for (int i = 0; i < follow_case.n; ++i) {
        if (i % 2 == 0) {
            follow_case.z[i] = -follow_case.z[i];
        }
    }

    return follow_case;
}
