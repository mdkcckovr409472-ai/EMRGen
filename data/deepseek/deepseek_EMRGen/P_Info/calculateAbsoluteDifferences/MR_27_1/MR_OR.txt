#include <cmath>
#include <cfloat>

int z27check(double* result1, double* result2, int n) {
    // 1. 防御性编程 (空指针与长度检查)
    if (result1 == NULL || result2 == NULL) return 0;
    if (n <= 0) return 1;

    // 2. 核心判断逻辑 (elementwise)
    for (int i = 0; i < n; i++) {
        // 提取移项后的操作数 A = result1[i], B = result2[i] (条件为 result1[i] >= result2[i])
        double A = result1[i];
        double B = result2[i];

        // 防御 NaN
        if (std::isnan(A) || std::isnan(B)) return 0;

        // 处理无穷大
        if (std::isinf(A) || std::isinf(B)) {
            // 对于 >= 比较，直接使用数学比较
            if (!(A >= B)) return 0;
            continue; // 此对元素比较完成，继续下一对
        }

        // 定义基础容差
        double epsilon = 1e-5;
        // 计算动态容差
        double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

        // 带容差的 >= 比较: 若 A < B - tol 则违反条件
        if (A < B - tol) {
            return 0;
        }
    }

    // 3. 全部满足，返回 1
    return 1;
}
