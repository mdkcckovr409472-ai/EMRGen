int z25check(int* result1, int* result2, int n) {
    // 4.a) Length validity check
    if (n <= 0) return 1;
    // 4.b) Null pointer check
    if (result1 == NULL || result2 == NULL) return 0;

    // 3) Dynamic tolerance
    const double epsilon = 1e-5;

    for (int i = 0; i < n; i++) {
        int A = result1[i];
        int B = result2[i];
        // Lexicographical comparison for int: direct comparison
        if (A < B) return 1;
        else if (A > B) return 0;
        // else A == B, continue checking next index
    }
    // All elements equal, strict less than lexicographically means return 0
    return 0;
}
