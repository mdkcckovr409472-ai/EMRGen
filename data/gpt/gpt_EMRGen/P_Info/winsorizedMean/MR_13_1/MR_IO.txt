TestCase z13Change(TestCase t1) {
    TestCase follow_case = t1;
    // Memory handling for deep copy of elements
    double* follow_case_elements = (double*)malloc(t1.n * sizeof(double));
    memcpy(follow_case_elements, t1.elements, t1.n * sizeof(double));
    follow_case.elements = follow_case_elements;
    // Defensive check for empty array
    if (follow_case.n == 0) {
        return follow_case;
    }
    // Sort elements in descending order
    for (int i = 0; i < follow_case.n - 1; ++i) {
        int maxIdx = i;
        for (int j = i + 1; j < follow_case.n; ++j) {
            if (follow_case.elements[j] > follow_case.elements[maxIdx]) {
                maxIdx = j;
            }
        }
        if (maxIdx != i) {
            double temp = follow_case.elements[i];
            follow_case.elements[i] = follow_case.elements[maxIdx];
            follow_case.elements[maxIdx] = temp;
        }
    }
    return follow_case;
}
