int z17check(double result1, double result2) {
    double epsilon = 1e-5;

    // Algebraic rearrangement for return.result1 <= return.result2:
    // Directly: result1 <= result2
    double A = result1;
    double B = result2;

    if (std::isnan(A) || std::isnan(B)) return 0;

    if (std::isinf(A) || std::isinf(B)) {
        if (A <= B) return 1;
        else return 0;
    }

    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

    if (A > B + tol) return 0;

    return 1;
}
