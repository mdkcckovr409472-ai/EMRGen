TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tf.n > 0 && tf.a != nullptr) {
        int* reversed_a = (int*)malloc(tf.n * sizeof(int));
        if (reversed_a != nullptr) {
            for (int i = 0; i < tf.n; ++i) {
                reversed_a[i] = tf.a[tf.n - 1 - i];
            }
            tf.a = reversed_a;
        }
    }
    return tf;
}
