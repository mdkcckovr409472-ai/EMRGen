#include <math.h>
#include <stddef.h>

int z15check(double* result1, double* result2, int n) {
    // 1. 防御性编程 (空指针与长度检查)
    if (result1 == NULL || result2 == NULL) return 0;
    if (n <= 0) return 1;

    // 2. 核心判断逻辑 (字典序比较)
    double epsilon = 1e-5;
    for (int i = 0; i < n; i++) {
        double A = result1[i];
        double B = result2[i];

        // 防御 NaN
        if (isnan(A) || isnan(B)) return 0;

        // 处理无穷大
        if (isinf(A) || isinf(B)) {
            if (A > B) {
                // 第一个不相等元素满足 >，字典序上 result1 > result2，满足 >=
                return 1;
            } else if (A < B) {
                // 第一个不相等元素满足 <，字典序上 result1 < result2，不满足 >=
                return 0;
            }
            // 相等则继续循环
            continue;
        }

        // 计算动态容差
        double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

        // 比较当前元素对
        if (fabs(A - B) > tol) {
            // 找到第一个不相等的元素对
            if (A > B - tol) {
                // result1[i] > result2[i]，字典序上 result1 > result2，满足 >=
                return 1;
            } else {
                // result1[i] < result2[i]，字典序上 result1 < result2，不满足 >=
                return 0;
            }
        }
        // 相等则继续检查下一个元素
    }

    // 3. 所有元素在容差内相等，满足 >=
    return 1;
}
