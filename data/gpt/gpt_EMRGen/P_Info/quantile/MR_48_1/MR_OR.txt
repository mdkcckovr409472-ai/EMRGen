int z48check(double result1, double result2) {
    // Core evaluation logic (direct scalar comparison)
    // rearranged expression for result1 >= result2: A = result1, B = result2
    double A = result1;
    double B = result2;
    double epsilon = 1e-5;

    if (std::isnan(A) || std::isnan(B)) return 0;
    if (std::isinf(A) || std::isinf(B)) {
        if (A >= B) return 1;
        else return 0;
    }

    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    if (A < B - tol) return 0;

    return 1;
}
