TestCase z43Change(TestCase t1) {
    TestCase follow_case = t1;
    // Memory handling for array2 (deep copy)
    int* follow_case_array2 = (int*)malloc(t1.n * sizeof(int));
    memcpy(follow_case_array2, t1.array2, t1.n * sizeof(int));
    follow_case.array2 = follow_case_array2;

    // Defensive check: ensure there is at least one element to sort
    if (follow_case.n <= 0) {
        return follow_case;
    }

    // Sort array2 in ascending order
    for (int i = 0; i < follow_case.n - 1; ++i) {
        for (int j = 0; j < follow_case.n - i - 1; ++j) {
            if (follow_case.array2[j] > follow_case.array2[j + 1]) {
                int tmp = follow_case.array2[j];
                follow_case.array2[j] = follow_case.array2[j + 1];
                follow_case.array2[j + 1] = tmp;
            }
        }
    }

    return follow_case;
}
