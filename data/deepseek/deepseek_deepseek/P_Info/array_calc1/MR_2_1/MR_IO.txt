TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tf.n > 0 && tf.a != NULL) {
        int* reversed_a = (int*)malloc(tf.n * sizeof(int));
        if (reversed_a != NULL) {
            for (int i = 0; i < tf.n; i++) {
                reversed_a[i] = tc.a[tc.n - 1 - i];
            }
            tf.a = reversed_a;
        }
    }
    return tf;
}
