TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tf.n > 0) {
        double* newData = new double[tf.n];
        for (int i = 0; i < tf.n - 1; i++) {
            newData[i] = tf.data[i + 1];
        }
        newData[tf.n - 1] = tf.data[0];
        tf.data = newData;
    }
    return tf;
}
