#include <math.h>
#include <stddef.h>

int z31check(double result1, double result2) {
    // 1. 防御性编程 (针对浮点数的特殊处理)
    // 提取移项后的操作数
    double A = result1;
    double B = result2;
    
    // 检查 NaN
    if (isnan(A) || isnan(B)) return 0;
    
    // 处理无穷大情况
    if (isinf(A) || isinf(B)) {
        // 直接数学比较
        if (A >= B) return 1;
        else return 0;
    }
    
    // 2. 带容差的浮点数比较
    double epsilon = 1e-5;
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    // 检查条件: result1 >= result2
    if (A < B - tol) {
        return 0;
    }
    
    // 3. 满足条件，返回 1
    return 1;
}
