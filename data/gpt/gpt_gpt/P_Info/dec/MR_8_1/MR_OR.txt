int check(TestCase tc) {
    int n = tc.n;
    int* Os = dec(tc.array1, tc.array2, n);
    if (Os == NULL) {
        return 0;
    }
    TestCase Tf_case = change(tc);
    int* Of = dec(Tf_case.array1, Tf_case.array2, n);
    if (Of == NULL) {
        free(Os);
        return 0;
    }
    int is_valid = 1;
    for (int i = 0; i < n; i++) {
        if (Of[i] != Os[i] + n) {
            is_valid = 0;
            break;
        }
    }
    free(Os);
    free(Of);
    return is_valid;
}
