#include <math.h>
#include <stddef.h>

int z1check(double result1, double result2) {
    // 1. 浮点数比较特殊处理
    // 移项后的条件：result1 < result2
    double A = result1;
    double B = result2;
    
    // 防御NaN
    if (isnan(A) || isnan(B)) return 0;
    
    // 处理无穷大情况
    if (isinf(A) || isinf(B)) {
        if (A < B) return 1;
        else return 0;
    }
    
    // 动态容差计算
    double epsilon = 1e-5;
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    // 带容差的比较：A < B
    // 违反条件：A >= B - tol
    if (A >= B - tol) return 0;
    
    // 2. 满足条件，返回1
    return 1;
}
