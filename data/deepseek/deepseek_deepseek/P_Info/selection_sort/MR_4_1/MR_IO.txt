TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tf.n <= 1) {
        return tf;
    }
    int p = 1; // 循环左移1个位置
    int* new_list = (int*)malloc(tf.n * sizeof(int));
    if (new_list == nullptr) {
        return tf;
    }
    for (int i = 0; i < tf.n; i++) {
        new_list[i] = tf.list[(i + p) % tf.n];
    }
    tf.list = new_list;
    return tf;
}
