int check(TestCase tc) {
    double* Os = standardize(tc.data, tc.n);
    TestCase Tf_case = change(tc);
    double* Of = standardize(Tf_case.data, Tf_case.n);
    int is_valid = 1;
    for (int i = 0; i < tc.n; i++) {
        if (fabs(Os[i] - (-1.0 * Of[i])) > 1e-9) {
            is_valid = 0;
            break;
        }
    }
    free(Os);
    free(Of);
    free(Tf_case.data);
    return is_valid;
}
