TestCase z50Change(TestCase t1) {
    TestCase follow_case = t1;
    // --- Memory handling code ---
    double* follow_case_data2 = (double*)malloc(t1.n2 * sizeof(double));
    memcpy(follow_case_data2, t1.data2, t1.n2 * sizeof(double));
    follow_case.data2 = follow_case_data2;
    // ------------------------------------
    
    // Defensive check: need at least 1 element to sort
    if (follow_case.n2 < 1) {
        return follow_case;
    }
    
    // Sort data2 in descending order
    for (int i = 0; i < follow_case.n2 - 1; ++i) {
        int maxIdx = i;
        for (int j = i + 1; j < follow_case.n2; ++j) {
            if (follow_case.data2[j] > follow_case.data2[maxIdx]) {
                maxIdx = j;
            }
        }
        if (maxIdx != i) {
            double temp = follow_case.data2[i];
            follow_case.data2[i] = follow_case.data2[maxIdx];
            follow_case.data2[maxIdx] = temp;
        }
    }
    
    return follow_case;
}
