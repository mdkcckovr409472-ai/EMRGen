int check(TestCase tc) {
    double Os = find_magnitude(tc.a, tc.n);
    TestCase Tf_case = change(tc);
    double Of = find_magnitude(Tf_case.a, Tf_case.n);
    int is_valid = 0;
    const double EPSILON = 1e-8;
    if (fabs(Of - 2 * Os) < EPSILON) {
        is_valid = 1;
    }
    delete[] Tf_case.a;
    return is_valid;
}
