int z44check(double result1, double result2) {
    // Core evaluation logic (direct scalar comparison) with floating-point tolerance
    double epsilon = 1e-5;

    // Rearrange the invariant condition "result1 >= result2" as A >= B
    // Here A = result1, B = result2
    double A = result1;
    double B = result2;

    if (std::isnan(A) || std::isnan(B)) return 0;
    if (std::isinf(A) || std::isinf(B)) {
        if (A >= B) return 1;
        else return 0;
    }

    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

    if (A < B - tol) return 0;

    return 1;
}
