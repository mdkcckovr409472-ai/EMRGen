TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tf.n <= 0 || tf.data == nullptr) {
        return tf;
    }
    double sum = 0.0;
    for (int i = 0; i < tf.n; ++i) {
        sum += tf.data[i];
    }
    double mean = sum / tf.n;
    double* new_data = new double[tf.n];
    for (int i = 0; i < tf.n; ++i) {
        new_data[i] = tf.data[i] + mean;
    }
    tf.data = new_data;
    return tf;
}
