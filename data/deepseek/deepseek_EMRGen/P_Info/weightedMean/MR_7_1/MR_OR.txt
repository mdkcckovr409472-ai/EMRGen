#include <cmath>
#include <cstddef>

int z7check(double result1, double result2) {
    // 1. 核心判断逻辑 (标量直接比较)
    // 条件: result1 + result2 == 0
    // 移项: result1 == -result2
    // 提取 A = result1, B = -result2

    double A = result1;
    double B = -result2;

    // 浮点数比较特殊处理
    // 第一步：定义基础容差
    double epsilon = 1e-5;

    // 第二步：防御 NaN
    if (std::isnan(A) || std::isnan(B)) return 0;

    // 第三步：处理无穷大
    if (std::isinf(A) || std::isinf(B)) {
        // 无穷大时退化为精确数学比较
        if (A == B) return 1;
        else return 0;
    }

    // 第四步：计算动态宽容度
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

    // 第五步：带容差的相等判定
    // 条件为 A == B (即 result1 == -result2)
    if (fabs(A - B) > tol) {
        return 0;
    }

    // 2. 满足条件，返回 1
    return 1;
}
