int z40check(int* result1, int* result2, int n) {
    if (n <= 0) return 1;
    if (result1 == NULL || result2 == NULL) return 0;

    for (int i = 0; i < n; i++) {
        int A = result1[i];
        int B = result2[i] + 1;
        if (A != B) {
            return 0;
        }
    }

    return 1;
}
