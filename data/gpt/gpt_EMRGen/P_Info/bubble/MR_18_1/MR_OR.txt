int z18check(int* result1, int* result2, int n) {
    // 1. Defensive programming (Length check first)
    if (n <= 0) return 1;
    // 2. Defensive programming (Null pointer check second)
    if (result1 == NULL || result2 == NULL) return 0;

    // 3. Core evaluation logic (elementwise check: result1[i] == result2[i])
    for (int i = 0; i < n; i++) {
        if (!(result1[i] == result2[i])) {
            return 0; // Return 0 immediately upon any violation
        }
    }

    // 4. Fully satisfied, return 1
    return 1;
}
