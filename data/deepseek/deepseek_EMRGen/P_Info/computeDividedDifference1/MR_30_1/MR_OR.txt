#include <cmath>
#include <cfloat>

int z30check(double* result1, double* result2, int n) {
    // 1. 防御性编程 (空指针与长度检查)
    if (result1 == NULL || result2 == NULL) return 0;
    if (n <= 0) return 1;

    // 2. 字典序比较逻辑 (lexically)
    for (int i = 0; i < n; i++) {
        double A = result1[i];
        double B = result2[i];

        // 防御 NaN
        if (std::isnan(A) || std::isnan(B)) return 0;

        // 处理无穷大
        if (std::isinf(A) || std::isinf(B)) {
            // 对于 > 比较，直接使用数学比较
            if (A > B) {
                // 在第一个不相等的元素处，result1[i] > result2[i]，满足字典序 >，可以提前返回成功
                return 1;
            } else if (A < B) {
                // 在第一个不相等的元素处，result1[i] < result2[i]，违反字典序 >，立即失败
                return 0;
            }
            // 如果相等，继续检查下一个元素
            continue;
        }

        // 动态容差比较（用于处理非无穷大的浮点数）
        double epsilon = 1e-5;
        double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

        // 判断 A > B 是否成立（带容差）
        if (A > B + tol) {
            // 在第一个不相等的元素处，result1[i] > result2[i]，满足字典序 >，可以提前返回成功
            return 1;
        } else if (A < B - tol) {
            // 在第一个不相等的元素处，result1[i] < result2[i]，违反字典序 >，立即失败
            return 0;
        }
        // 如果两数在容差内视为相等，则继续检查下一个元素
    }

    // 3. 循环结束，所有元素在容差内都相等，不满足严格的 > 关系
    return 0;
}
