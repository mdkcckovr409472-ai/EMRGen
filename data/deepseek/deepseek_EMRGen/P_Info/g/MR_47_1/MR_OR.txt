#include <math.h>
#include <stddef.h>

int z47check(double result1, double result2) {
    // 1. 核心判断逻辑 (标量直接比较)
    // 条件: result1 <= result2
    // 移项后: result1 <= result2 (已满足形式)

    // 浮点数特殊处理
    double epsilon = 1e-5;
    
    // 提取左右操作数 (已移项)
    double A = result1;
    double B = result2;
    
    // 防御 NaN
    if (isnan(A) || isnan(B)) return 0;
    
    // 处理无穷大情况
    if (isinf(A) || isinf(B)) {
        // 无穷大时直接数学比较
        if (A <= B) return 1;
        else return 0;
    }
    
    // 计算动态容差
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    // 带容差的 <= 比较 (若违反则返回0)
    // 条件: A <= B
    // 违反条件: A > B
    // 带容差: A > B + tol
    if (A > B + tol) {
        return 0;
    }
    
    // 2. 满足条件，返回 1
    return 1;
}
