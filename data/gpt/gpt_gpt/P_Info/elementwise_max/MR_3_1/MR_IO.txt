TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tf.a != NULL && tf.n > 0) {
        tf.a = (int*)malloc(tf.n * sizeof(int));
        if (tf.a != NULL) {
            for (int i = 0; i < tf.n; i++) {
                tf.a[i] = tc.a[i];
            }
            tf.a[0] = 2147483647;
        }
    }
    return tf;
}
