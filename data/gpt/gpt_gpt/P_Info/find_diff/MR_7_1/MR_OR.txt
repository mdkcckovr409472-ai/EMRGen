int check(TestCase tc) {
    int* Os = find_diff(tc.a, tc.b, tc.na);
    TestCase Tf_case = change(tc);
    int* Of = find_diff(tc.a, Tf_case.b, Tf_case.na);
    int is_valid = 1;
    for (int i = 0; i < tc.na; i++) {
        if (Of[i] != Os[i] + 1) {
            is_valid = 0;
            break;
        }
    }
    delete[] Tf_case.b;
    return is_valid;
}
