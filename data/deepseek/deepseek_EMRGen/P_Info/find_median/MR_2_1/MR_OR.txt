#include <cmath>
#include <cfloat>

int z2check(double result1, double result2) {
    // 1. 核心判断逻辑 (标量直接比较)
    // 移项处理：return.result1 - return.result2 - 1 == 0
    // 等价于：result1 - result2 == 1
    // 进一步移项：result1 == result2 + 1
    // 令 A = result1, B = result2 + 1
    
    double A = result1;
    double B = result2 + 1.0;
    
    // 防御 NaN
    if (std::isnan(A) || std::isnan(B)) return 0;
    
    // 处理无穷大情况
    if (std::isinf(A) || std::isinf(B)) {
        if (A == B) return 1;
        else return 0;
    }
    
    // 动态容差比较
    double epsilon = 1e-5;
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    if (fabs(A - B) > tol) {
        return 0;
    }
    
    // 2. 满足条件，返回 1
    return 1;
}
