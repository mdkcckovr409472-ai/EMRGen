TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tf.n > 1) {
        int idx1 = rand() % tf.n;
        int idx2 = rand() % tf.n;
        while (idx2 == idx1) {
            idx2 = rand() % tf.n;
        }
        int temp = tf.a[idx1];
        tf.a[idx1] = tf.a[idx2];
        tf.a[idx2] = temp;
    }
    return tf;
}
