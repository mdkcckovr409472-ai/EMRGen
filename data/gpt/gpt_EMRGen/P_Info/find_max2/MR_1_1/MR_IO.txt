TestCase z1Change(TestCase t1) {
    TestCase follow_case = t1;
    // Memory handling: deep copy of array 'a'
    int* follow_case_a = (int*)malloc(t1.n * sizeof(int));
    memcpy(follow_case_a, t1.a, t1.n * sizeof(int));
    follow_case.a = follow_case_a;

    // Defensive check: ensure there is at least one element
    if (follow_case.n == 0) {
        return follow_case;
    }

    // Operation: add 1 to all elements
    for (int i = 0; i < follow_case.n; ++i) {
        follow_case.a[i] += 1;
    }

    return follow_case;
}
