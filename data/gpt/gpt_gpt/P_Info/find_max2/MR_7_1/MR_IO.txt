TestCase change(TestCase tc) {
    TestCase tf = tc;
    // Allocate new array to avoid modifying original array per test case reuse principle
    int* new_arr = new int[tc.n];
    for (int i = 0; i < tc.n; i++) {
        new_arr[i] = tc.a[i];
    }
    new_arr[0] += 2; // add 2 to the first element only
    tf.a = new_arr;
    return tf;
}
