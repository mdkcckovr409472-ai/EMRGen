#include <cmath>
#include <cfloat>

int z16check(double result1, double result2) {
    // 1. 防御性编程 (标量版本无需空指针和长度检查)
    
    // 2. 核心判断逻辑 (标量直接比较)
    // 条件为 return.result1 == return.result2，即 result1 == result2
    // 由于数据类型为 double，必须使用浮点数特殊处理
    
    // 第一步：定义基础容差
    double epsilon = 1e-5;
    
    // 第二步：提取左右操作数 A 和 B
    // 条件 result1 == result2 已经是标准形式，无需移项
    double A = result1;
    double B = result2;
    
    // 第三步：防御 NaN，并对 Inf 进行特殊放行
    if (std::isnan(A) || std::isnan(B)) return 0;
    
    // 处理无穷大情况
    if (std::isinf(A) || std::isinf(B)) {
        // 对于无穷大，直接使用数学比较
        if (A == B) return 1;
        else return 0;
    }
    
    // 第四步：计算动态宽容度
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    // 第五步：带容差的相等判定
    // 移项后的条件是相等 (==)
    if (fabs(A - B) > tol) {
        return 0;
    }
    
    // 3. 满足条件，返回 1
    return 1;
}
