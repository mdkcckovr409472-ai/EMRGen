int check(TestCase tc) {
    int Os = add_values(tc.a, tc.n);
    TestCase Tf_case = change(tc);
    int Of = add_values(Tf_case.a, Tf_case.n);
    int k = 5;
    int n = tc.n;
    int half_n_ceil = (n + 1) / 2;
    int expected_Of = Os + half_n_ceil * k;

        is_valid = 1;
    }

}
