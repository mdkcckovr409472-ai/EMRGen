#include <math.h>
#include <stddef.h>

int z10check(double result1, double result2) {
    // 1. 核心判断逻辑 (标量直接比较)
    // 条件: 4 * return.result1 - return.result2 == 0
    // 移项恢复量级: 4 * result1 == result2
    
    double A = 4.0 * result1;
    double B = result2;
    
    // 防御 NaN
    if (isnan(A) || isnan(B)) return 0;
    
    // 处理无穷大情况
    if (isinf(A) || isinf(B)) {
        // 无穷大时退化为精确比较
        if (A == B) return 1;
        else return 0;
    }
    
    // 动态容差比较
    double epsilon = 1e-5;
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    if (fabs(A - B) > tol) {
        return 0;
    }
    
    // 2. 满足条件，返回 1
    return 1;
}
