int check(TestCase tc) {
    if (tc.a == NULL || tc.n <= 0) return 0;

    // Source output Os
    int* Os = set_min_val(tc.a, tc.k, tc.n);
    if (Os == NULL) return 0;

    // Follow-up test case
    TestCase Tf_case = change(tc);
    if (Tf_case.a == NULL) {
        free(Os);
        return 0;
    }
    // Follow-up output Of
    int* Of = set_min_val(Tf_case.a, Tf_case.k, Tf_case.n);
    if (Of == NULL) {
        free(Os);
        free(Tf_case.a);
        return 0;
    }

    int is_valid = 0;
    if (Of[tc.n - 1] >= tc.k || Os[tc.n - 1] >= Of[tc.n - 1]) {
        is_valid = 1;
    }

    free(Os);
    free(Of);
    free(Tf_case.a);

    return is_valid;
}
