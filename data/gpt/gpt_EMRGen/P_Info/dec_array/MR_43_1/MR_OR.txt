int z43check(int* result1, int* result2, int n) {
    // Defensive programming: length check first
    if (n <= 0) return 1;
    // Null pointer check after length check
    if (result1 == NULL || result2 == NULL) return 0;

    // Core evaluation logic (elementwise): check result1[i] >= result2[i]
    for (int i = 0; i < n; i++) {
        if (!(result1[i] >= result2[i])) {
            return 0;
        }
    }

    // Fully satisfied
    return 1;
}
