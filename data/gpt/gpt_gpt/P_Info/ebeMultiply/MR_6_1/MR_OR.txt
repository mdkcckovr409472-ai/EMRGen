int check(TestCase tc) {
    if (tc.a == NULL || tc.b == NULL || tc.n1 <= 0 || tc.n2 != tc.n1) {
        return 0;
    }

    double* Os = ebeMultiply(tc.a, tc.b, tc.n1, tc.n2);
    if (Os == NULL) {
        return 0;
    }

    TestCase Tf_case = change(tc);
    if (Tf_case.b == NULL) {
        free(Os);
        return 0;
    }

    double* Of = ebeMultiply(Tf_case.a, Tf_case.b, Tf_case.n1, Tf_case.n2);
    if (Of == NULL) {
        free(Os);
        if (Tf_case.b != tc.b) free(Tf_case.b);
        return 0;
    }

    int is_valid = 1;
    for (int i = 0; i < tc.n1; i++) {
        double expected = Os[i] + tc.a[i];
        if (Of[i] != expected) {
            is_valid = 0;
            break;
        }
    }

    free(Os);
    free(Of);
    if (Tf_case.b != tc.b) free(Tf_case.b);

    return is_valid;
}
