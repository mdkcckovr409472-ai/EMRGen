int check(TestCase tc) {
    double* Os = scale(tc.val, tc.arr, tc.n);
    TestCase Tf_case = change(tc);
    double* Of = scale(Tf_case.val, Tf_case.arr, Tf_case.n);

    int is_valid = 1;
    for (int i = 0; i < tc.n; i++) {
        if (Os[i] == 0) {
            if (Of[i] != 0) {
                is_valid = 0;
                break;
            }
        } else {
            // Check Of[i] == Os[i] / 2 with floating precision tolerance
            double expected = Os[i] / 2.0;
            double diff = Of[i] - expected;
            if (diff < -1e-9 || diff > 1e-9) {
                is_valid = 0;
                break;
            }
        }
    }

    delete[] Tf_case.arr;

    return is_valid;
}
