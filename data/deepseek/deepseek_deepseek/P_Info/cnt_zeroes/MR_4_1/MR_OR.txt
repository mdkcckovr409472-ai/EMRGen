int check(TestCase tc) {
    // 获取初始输出 Os
    int Os = cnt_zeroes(tc.a, tc.n);
    
    // 构造衍生用例
    TestCase Tf_case = change(tc);
    
    // 获取衍生输出 Of
    int Of = cnt_zeroes(Tf_case.a, Tf_case.n);
    
    // 验证蜕变关系 Os == Of
    int is_valid = 0;
    if (Os == Of) {
        is_valid = 1;
    }
    
    // 释放衍生用例中动态分配的内存
    delete[] Tf_case.a;
    
    return is_valid;
}
