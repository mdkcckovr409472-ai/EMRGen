int check(TestCase tc) {
    // Source output
    int* Os = dec(tc.array1, tc.array2, tc.n);

    // Construct follow-up test case
    TestCase Tf_case = change(tc);

    // Follow-up output
    int* Of = dec(Tf_case.array1, Tf_case.array2, Tf_case.n);

    // Compare Os and Of elements for equality
    int is_valid = 1;
    for (int i = 0; i < tc.n; i++) {
        if (Os[i] != Of[i]) {
            is_valid = 0;
            break;
        }
    }

    // Free dynamically allocated memory
    free(Os);
    free(Of);
    free(Tf_case.array1);

    return is_valid;
}
