TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tf.n > 0 && tf.a != nullptr) {
        double min_val = tf.a[0];
        for (int i = 1; i < tf.n; i++) {
            if (tf.a[i] < min_val) {
                min_val = tf.a[i];
            }
        }
        double k = min_val > 0 ? min_val / 2.0 : 1.0;
        double* new_a = new double[tf.n];
        for (int i = 0; i < tf.n; i++) {
            new_a[i] = tf.a[i] - k;
        }
        tf.a = new_a;
    }
    return tf;
}
