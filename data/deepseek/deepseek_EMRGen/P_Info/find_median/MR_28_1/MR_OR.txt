#include <math.h>
#include <stddef.h>

int z28check(double result1, double result2) {
    // 1. 防御性编程 (标量情况无需空指针和长度检查)

    // 2. 核心判断逻辑 (标量直接比较)
    // 条件: result1 <= result2
    // 移项后: result1 <= result2 (无需移项)

    // 3. 浮点数比较的特殊处理
    double epsilon = 1e-5;
    
    // 提取左右操作数 (根据移项后的条件)
    double A = result1;
    double B = result2;
    
    // 防御 NaN
    if (isnan(A) || isnan(B)) return 0;
    
    // 处理无穷大情况
    if (isinf(A) || isinf(B)) {
        // 无穷大时退化为数学直接比较
        if (A <= B) return 1;
        else return 0;
    }
    
    // 计算动态容差
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    // 带容差的比较判定 (条件: A <= B)
    if (A > B + tol) {
        return 0; // 违反条件
    }
    
    // 3. 满足条件，返回 1
    return 1;
}
