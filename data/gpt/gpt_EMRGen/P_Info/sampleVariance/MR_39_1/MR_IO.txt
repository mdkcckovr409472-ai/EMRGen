TestCase z39Change(TestCase t1) {
    TestCase follow_case = t1;
    
    // Defensive check to avoid division by zero
    if (follow_case.mean == 0) {
        return follow_case;
    }
    
    // Operation: take reciprocal of mean
    follow_case.mean = 1.0 / follow_case.mean;
    
    return follow_case;
}
