TestCase z33Change(TestCase t1) {
    TestCase follow_case = t1;
    // Memory handling for target parameter 'b'
    int* follow_case_b = (int*)malloc(t1.na * sizeof(int));
    memcpy(follow_case_b, t1.b, t1.na * sizeof(int));
    follow_case.b = follow_case_b;

    // Defensive check: ensure there is at least one element to sort
    if (follow_case.na <= 0) {
        return follow_case;
    }

    // Sort 'b' in descending order
    for (int i = 0; i < follow_case.na - 1; ++i) {
        for (int j = i + 1; j < follow_case.na; ++j) {
            if (follow_case.b[i] < follow_case.b[j]) {
                int temp = follow_case.b[i];
                follow_case.b[i] = follow_case.b[j];
                follow_case.b[j] = temp;
            }
        }
    }

    return follow_case;
}
