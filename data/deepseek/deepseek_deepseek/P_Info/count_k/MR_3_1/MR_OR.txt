int check(TestCase tc) {
    // 计算初始输出 Os
    int Os = count_k(tc.a, tc.k, tc.n);
    
    // 构造衍生用例
    TestCase Tf_case = change(tc);
    
    // 计算衍生输出 Of
    int Of = count_k(Tf_case.a, Tf_case.k, Tf_case.n);
    
    int is_valid = 0;
    // 验证蜕变关系 Os == Of
    if (Os == Of) {
        is_valid = 1;
    }
    
    // 释放衍生用例中动态分配的数组内存
    delete[] Tf_case.a;
    
    return is_valid;
}
