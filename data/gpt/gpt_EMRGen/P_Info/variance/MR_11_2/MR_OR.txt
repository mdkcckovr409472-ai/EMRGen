int z11check(double result1, double result2) {
    // Step 1: define epsilon
    double epsilon = 1e-5;

    // Step 2: rearrangement: return.result1 - return.result2 == 0 
    // means result1 == result2
    double A = result1;
    double B = result2;

    // Step 3: NaN and Inf checks
    if (std::isnan(A) || std::isnan(B)) return 0;
    if (std::isinf(A) || std::isinf(B)) {
        // Direct equality comparison for infinite values
        if (A == B) return 1;
        else return 0;
    }

    // Step 4: dynamic tolerance
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

    // Step 5: tolerance based comparison for equality
    if (fabs(A - B) > tol) return 0;

    // satisfied
    return 1;
}
