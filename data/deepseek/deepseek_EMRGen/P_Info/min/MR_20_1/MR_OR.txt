#include <math.h>
#include <stddef.h>

int z20check(double result1, double result2) {
    // 1. 核心判断逻辑 (标量直接比较)
    // 浮点数比较需要特殊处理
    double epsilon = 1e-5;
    
    // 移项处理：条件 "result1 == result2" 已经是标准形式
    // A = result1, B = result2
    double A = result1;
    double B = result2;
    
    // 防御NaN
    if (isnan(A) || isnan(B)) return 0;
    
    // 处理无穷大情况
    if (isinf(A) || isinf(B)) {
        if (A == B) return 1;
        else return 0;
    }
    
    // 计算动态容差
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    // 带容差的相等判断
    if (fabs(A - B) > tol) {
        return 0;
    }
    
    // 2. 满足条件，返回 1
    return 1;
}
