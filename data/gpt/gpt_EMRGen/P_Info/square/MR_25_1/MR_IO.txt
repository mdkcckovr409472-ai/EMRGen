TestCase z25Change(TestCase t1) {
    TestCase follow_case = t1;
    // Memory handling: deep copy of the array
    double* follow_case_data = (double*)malloc(t1.n * sizeof(double));
    memcpy(follow_case_data, t1.data, t1.n * sizeof(double));
    follow_case.data = follow_case_data;
    // Defensive check: ensure there is at least one element
    if (follow_case.n <= 0) {
        return follow_case;
    }
    // Apply absolute value to each element (subarray = whole array)
    for (int i = 0; i < follow_case.n; ++i) {
        if (follow_case.data[i] < 0) {
            follow_case.data[i] = -follow_case.data[i];
        }
    }
    return follow_case;
}
