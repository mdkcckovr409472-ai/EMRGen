int z4check(double result1, double result2) {
    // Step 1: Define base tolerance
    double epsilon = 1e-5;

    // Step 2: Rearrange equation return.result1 - 3 * return.result2 == 0 as:
    // result1 == 3 * result2
    double A = result1;
    double B = 3 * result2;

    // Step 3: Defend against NaN and Inf
    if (std::isnan(A) || std::isnan(B)) return 0;
    if (std::isinf(A) || std::isinf(B)) {
        if (A == B) return 1;
        else return 0;
    }

    // Step 4: Calculate dynamic tolerance
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

    // Step 5: Tolerance-based comparison
    if (fabs(A - B) > tol) return 0;

    return 1;
}
