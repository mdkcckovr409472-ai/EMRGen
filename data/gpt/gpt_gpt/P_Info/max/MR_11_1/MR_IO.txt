TestCase change(TestCase tc) {
    TestCase tf = tc;
    // Allocate new memory to modify elements for follow-up input
    double* new_elements = new double[tf.n];
    for (int i = 0; i < tf.n; i++) {
        new_elements[i] = tf.elements[i];
    }
    // Set the last element to a value greater than the original max element (Os) + 1
    // Find the max in original elements first
    double max_value = tf.elements[0];
    for (int i = 1; i < tf.n; i++) {
        if (tf.elements[i] > max_value) {
            max_value = tf.elements[i];
        }
    }
    new_elements[tf.n - 1] = max_value + 1.0;
    tf.elements = new_elements;
    return tf;
}
