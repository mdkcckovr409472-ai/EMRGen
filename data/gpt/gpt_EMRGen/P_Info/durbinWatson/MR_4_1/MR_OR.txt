int z4check(double result1, double result2) {
    const double epsilon = 1e-5;

    // Algebraic rearrangement of return.result1 - 4 * return.result2 == 0
    // Here the equation is exactly: result1 == 4 * result2
    // So A = result1, B = 4 * result2

    double A = result1;
    double B = 4.0 * result2;

    if (std::isnan(A) || std::isnan(B)) return 0;
    if (std::isinf(A) || std::isinf(B)) {
        if (A == B) return 1;
        else return 0;
    }

    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

    if (fabs(A - B) > tol) return 0;

    return 1;
}
