#include <cmath>
#include <cfloat>

int z16check(double result1, double result2) {
    // 1. 防御性编程 (标量版本无需指针和长度检查)
    
    // 2. 核心判断逻辑 (标量直接比较)
    // 条件: result1 >= result2
    // 移项后: A = result1, B = result2, 比较符: >=
    
    double A = result1;
    double B = result2;
    
    // 第三步: 防御NaN，处理Inf
    if (std::isnan(A) || std::isnan(B)) return 0;
    
    if (std::isinf(A) || std::isinf(B)) {
        // 无穷大时退化为数学直接比较
        if (A >= B) return 1;
        else return 0;
    }
    
    // 第一步: 定义基础容差
    double epsilon = 1e-5;
    
    // 第四步: 计算动态宽容度
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));
    
    // 第五步: 带容差的比较判定
    // 条件: A >= B  (即 result1 >= result2)
    // 违反条件: A < B - tol
    if (A < B - tol) {
        return 0;
    }
    
    // 3. 满足条件，返回 1
    return 1;
}
