int z7check(double result1, double result2) {
    // Core evaluation logic (direct scalar comparison)

    // Step 1: Define tolerance
    double epsilon = 1e-5;

    // Step 2: Rearrange inequality result1 <= result2 as A <= B where A = result1, B = result2
    double A = result1;
    double B = result2;

    // Step 3: NaN and Inf handling
    if (std::isnan(A) || std::isnan(B)) return 0;
    if (std::isinf(A) || std::isinf(B)) {
        if (A <= B) return 1;
        else return 0;
    }

    // Step 4: Calculate dynamic tolerance
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

    // Step 5: Tolerance comparison for <=
    if (A > B + tol) return 0;

    // Condition satisfied
    return 1;
}
