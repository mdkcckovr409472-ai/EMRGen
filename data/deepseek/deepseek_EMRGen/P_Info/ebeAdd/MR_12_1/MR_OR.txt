#include <cmath>
#include <cfloat>

int z12check(double* result1, double* result2, int n) {
    // 1. 防御性编程 (空指针与长度检查)
    if (result1 == NULL || result2 == NULL) return 0;
    if (n <= 0) return 1;

    // 2. 字典序比较逻辑 (lexically)
    for (int i = 0; i < n; i++) {
        double A = result1[i];
        double B = result2[i];

        // 浮点数特殊处理
        // 防御 NaN
        if (std::isnan(A) || std::isnan(B)) return 0;

        // 处理无穷大（动态容差对Inf无效）
        if (std::isinf(A) || std::isinf(B)) {
            // 根据移项后的条件 result1 >= result2 直接比较
            if (A < B) {
                // 在第一个不相等的位置，result1[i] < result2[i]，违反 >= 条件
                return 0;
            } else if (A > B) {
                // 在第一个不相等的位置，result1[i] > result2[i]，满足 >= 条件
                return 1;
            }
            // 如果相等，继续检查下一个元素
            continue;
        }

        // 计算动态容差
        double epsilon = 1e-5;
        double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

        // 检查是否不相等（考虑容差）
        if (fabs(A - B) > tol) {
            // 找到第一个不相等的元素对
            // 根据条件 result1 >= result2 判断
            if (A < B - tol) {
                // 违反 >= 条件
                return 0;
            } else {
                // 满足 >= 条件
                return 1;
            }
        }
        // 如果相等（在容差范围内），继续检查下一个元素
    }

    // 3. 循环结束，所有元素都相等（在容差范围内），满足 >= 条件
    return 1;
}
