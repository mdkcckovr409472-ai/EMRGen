TestCase change(TestCase tc) {
    TestCase tf = tc;
    if (tf.n1 > 1 && tf.data1 != nullptr) {
        double* new_data1 = new double[tf.n1];
        new_data1[0] = tf.data1[1];
        for (int i = 1; i < tf.n1 - 1; ++i) {
            new_data1[i] = tf.data1[i + 1];
        }
        new_data1[tf.n1 - 1] = tf.data1[0];
        tf.data1 = new_data1;
    }
    return tf;
}
