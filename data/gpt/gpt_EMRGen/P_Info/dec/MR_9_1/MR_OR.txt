int z9check(int* result1, int* result2, int n) {
    // Defensive programming and boundary checks
    if (n <= 0) return 1;
    if (result1 == NULL || result2 == NULL) return 0;

    for (int i = 0; i < n; i++) {
        if (result1[i] == result2[i]) {
            continue;
        } else if (result1[i] > result2[i]) {
            return 1;
        } else {
            return 0;
        }
    }
    // arrays are equal, strictly greater not satisfied
    return 0;
}
