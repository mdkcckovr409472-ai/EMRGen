int z29check(double result1, double result2) {
    // Core evaluation logic (direct scalar comparison)
    // Invariant: return.result1 == return.result2
    // Step 1: Base tolerance
    const double epsilon = 1e-5;
    double A = result1;
    double B = result2;

    // Step 3: NaN and Inf handling
    if (std::isnan(A) || std::isnan(B)) return 0;
    if (std::isinf(A) || std::isinf(B)) {
        if (A == B) return 1;
        else return 0;
    }

    // Step 4: Dynamic tolerance
    double tol = epsilon * fmax(1.0, fmax(fabs(A), fabs(B)));

    // Step 5: Tolerance-based == comparison
    if (fabs(A - B) > tol) return 0;

    return 1;
}
